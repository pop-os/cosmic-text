2129pub fn bsearch_range_value_table<T: Copy>(c: char, r: &'static [(char, char, T)]) -> Option<T> {
2130    use core::cmp::Ordering::{Equal, Less, Greater};
2131    match r.binary_search_by(|&(lo, hi, _)| {
2132        if lo <= c && c <= hi { Equal }
2133        else if hi < c { Less }
2134        else { Greater }
2135    }) {
2136        Ok(idx) => {
2137            let (_, _, cat) = r[idx];
2138            Some(cat)
2139        }
2140        Err(_) => None
2141    }
2142}
