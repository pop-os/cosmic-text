27/// Parse as raw config parts use this when you want custom handling config file
28///
29/// Otherwise, you may want [`FontConfig::merge_config`]
30pub fn parse_config_parts(s: &str) -> Result<Vec<ConfigPart>> {
31    crate::parser::parse_config(&roxmltree::Document::parse_with_options(
32        s,
33        roxmltree::ParsingOptions {
34            allow_dtd: true,
35            ..Default::default()
36        },
37    )?)?
38    .collect()
39}
