11/// Errors that may occur when drawing glyphs.
12#[derive(Clone, Debug)]
13pub enum DrawError {
14    /// No viable sources were available.
15    NoSources,
16    /// The requested glyph was not present in the font.
17    GlyphNotFound(GlyphId),
18    /// Exceeded memory limits when loading a glyph.
19    InsufficientMemory,
20    /// Exceeded a recursion limit when loading a glyph.
21    RecursionLimitExceeded(GlyphId),
22    /// Error occurred during hinting.
23    HintingFailed(HintError),
24    /// An anchor point had invalid indices.
25    InvalidAnchorPoint(GlyphId, u16),
26    /// Error occurred while loading a PostScript (CFF/CFF2) glyph.
27    PostScript(CffError),
28    /// Conversion from outline to path failed.
29    ToPath(ToPathError),
30    /// Error occurred when reading font data.
31    Read(ReadError),
32    /// HarfBuzz style drawing with hints is not supported
33    // Error rather than silently returning unhinted per f2f discussion.
34    HarfBuzzHintingUnsupported,
35}
36
37impl From<HintError> for DrawError {
38    fn from(value: HintError) -> Self {
39        Self::HintingFailed(value)
40    }
41}
42
43impl From<ToPathError> for DrawError {
44    fn from(e: ToPathError) -> Self {
45        Self::ToPath(e)
46    }
47}
48
49impl From<ReadError> for DrawError {
50    fn from(e: ReadError) -> Self {
51        Self::Read(e)
52    }
53}
54
55impl From<CffError> for DrawError {
56    fn from(value: CffError) -> Self {
57        Self::PostScript(value)
58    }
59}
60
61impl fmt::Display for DrawError {
62    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
63        match self {
64            Self::NoSources => write!(f, "No glyph sources are available for the given font"),
65            Self::GlyphNotFound(gid) => write!(f, "Glyph {gid} was not found in the given font"),
66            Self::InsufficientMemory => write!(f, "exceeded memory limits"),
67            Self::RecursionLimitExceeded(gid) => write!(
68                f,
69                "Recursion limit ({}) exceeded when loading composite component {gid}",
70                super::GLYF_COMPOSITE_RECURSION_LIMIT,
71            ),
72            Self::HintingFailed(e) => write!(f, "{e}"),
73            Self::InvalidAnchorPoint(gid, index) => write!(
74                f,
75                "Invalid anchor point index ({index}) for composite glyph {gid}",
76            ),
77            Self::PostScript(e) => write!(f, "{e}"),
78            Self::ToPath(e) => write!(f, "{e}"),
79            Self::Read(e) => write!(f, "{e}"),
80            Self::HarfBuzzHintingUnsupported => write!(
81                f,
82                "HarfBuzz style paths with hinting is not (yet?) supported"
83            ),
84        }
85    }
86}
