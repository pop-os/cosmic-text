831        let range = self.shape.version_byte_range();

841    /// To compute: set it to 0, sum the entire font as uint32, then
842    /// store 0xB1B0AFBA - sum. If the font is used as a component in a
843    /// font collection file, the value of this field will be
844    /// invalidated by changes to the file structure and font table
845    /// directory, and must be ignored.
