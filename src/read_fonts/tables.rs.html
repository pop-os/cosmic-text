52/// Computes the table checksum for the given data.
53///
54/// See the OpenType [specification](https://learn.microsoft.com/en-us/typography/opentype/spec/otff#calculating-checksums)
55/// for details.
56pub fn compute_checksum(table: &[u8]) -> u32 {
57    let mut sum = 0u32;
58    let mut iter = table.chunks_exact(4);
59    for quad in &mut iter {
60        // this can't fail, and we trust the compiler to avoid a branch
61        let array: [u8; 4] = quad.try_into().unwrap_or_default();
62        sum = sum.wrapping_add(u32::from_be_bytes(array));
63    }
64
65    let rem = match *iter.remainder() {
66        [a] => u32::from_be_bytes([a, 0, 0, 0]),
67        [a, b] => u32::from_be_bytes([a, b, 0, 0]),
68        [a, b, c] => u32::from_be_bytes([a, b, c, 0]),
69        _ => 0,
70    };
71
72    sum.wrapping_add(rem)
73}
