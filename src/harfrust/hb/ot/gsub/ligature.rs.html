14impl WouldApply for Ligature<'_> {
15    fn would_apply(&self, ctx: &WouldApplyContext) -> bool {
16        let components = self.component_glyph_ids();
17        ctx.glyphs.len() == components.len() + 1
18            && components
19                .iter()
20                .map(|comp| GlyphId::from(comp.get()))
21                .enumerate()
22                .all(|(i, comp)| ctx.glyphs[i + 1] == comp)
23    }
24}

26impl Apply for Ligature<'_> {
27    fn apply(&self, ctx: &mut hb_ot_apply_context_t) -> Option<()> {
28        // Special-case to make it in-place and not consider this
29        // as a "ligated" substitution.
30        let components = self.component_glyph_ids();
31        if components.is_empty() {
32            ctx.replace_glyph(self.ligature_glyph().into());
33            Some(())
34        } else {
35            let f = |info: &mut GlyphInfo, index| {
36                let value = components.get(index as usize).unwrap().get().to_u16();
37                match_glyph(info, value)
38            };
39
40            let mut match_end = 0;
41            let mut total_component_count = 0;
42
43            if !match_input(
44                ctx,
45                components.len() as u16,
46                f,
47                &mut match_end,
48                Some(&mut total_component_count),
49            ) {
50                ctx.buffer
51                    .unsafe_to_concat(Some(ctx.buffer.idx), Some(match_end));
52                return None;
53            }
54            let count = components.len() + 1;
55            ligate_input(
56                ctx,
57                count,
58                match_end,
59                total_component_count,
60                self.ligature_glyph().into(),
61            );
62            Some(())
63        }
64    }
65}

80impl ApplyLigatureSet for LigatureSet<'_> {
81    fn apply(&self, ctx: &mut hb_ot_apply_context_t, seconds: &hb_set_digest_t) -> Option<()> {
82        let mut second = GlyphId::new(u32::MAX);
83        let mut unsafe_to = 0;
84        let ligatures = self.ligatures();
85        let slow_path = if ligatures.len() <= 1 {
86            true
87        } else {
88            let mut iter = skipping_iterator_t::with_match_fn(ctx, false, Some(match_always));
89            iter.reset(iter.buffer.idx);
90            let matched = iter.next(Some(&mut unsafe_to));
91            if !matched {
92                true
93            } else {
94                second = iter.buffer.info[iter.index()].glyph_id.into();
95                unsafe_to = iter.index() + 1;
96
97                // Can't use the fast path if eg. the next char is a default-ignorable
98                // or other skippable.
99                iter.may_skip(&iter.buffer.info[iter.index()]) != may_skip_t::SKIP_NO
100            }
101        };
102
103        if slow_path {
104            // Slow path
105            for lig in ligatures.iter().filter_map(Result::ok) {
106                if lig.apply(ctx).is_some() {
107                    return Some(());
108                }
109            }
110        } else {
111            // Fast path
112            if !seconds.may_have(second.into()) {
113                return None;
114            }
115            let mut unsafe_to_concat = false;
116            for lig in ligatures.iter().filter_map(|lig| lig.ok()) {
117                let components = lig.component_glyph_ids();
118                if components.is_empty() || components[0].get() == second {
119                    if lig.apply(ctx).is_some() {
120                        if unsafe_to_concat {
121                            ctx.buffer
122                                .unsafe_to_concat(Some(ctx.buffer.idx), Some(unsafe_to));
123                        }
124                        return Some(());
125                    }
126                } else if !components.is_empty() {
127                    unsafe_to_concat = true;
128                }
129            }
130            if unsafe_to_concat {
131                ctx.buffer
132                    .unsafe_to_concat(Some(ctx.buffer.idx), Some(unsafe_to));
133            }
134        }
135        None
136    }
137}

139impl WouldApply for LigatureSubstFormat1<'_> {
140    fn would_apply(&self, ctx: &WouldApplyContext) -> bool {
141        self.coverage()
142            .ok()
143            .and_then(|coverage| coverage.get(ctx.glyphs[0]))
144            .and_then(|index| self.ligature_sets().get(index as usize).ok())
145            .is_some_and(|set| set.would_apply(ctx))
146    }
147}

149impl Apply for LigatureSubstFormat1<'_> {
150    fn apply_with_external_cache(
151        &self,
152        ctx: &mut hb_ot_apply_context_t,
153        external_cache: &SubtableExternalCache,
154    ) -> Option<()> {
155        let glyph = ctx.buffer.cur(0).as_glyph();
156
157        let (index, seconds) = match external_cache {
158            SubtableExternalCache::LigatureSubstFormat1Cache(cache) => (
159                coverage_index_cached(
160                    |gid| self.coverage().ok()?.get(gid),
161                    glyph,
162                    &cache.coverage,
163                )?,
164                &cache.seconds,
165            ),
166            SubtableExternalCache::LigatureSubstFormat1SmallCache(cache) => (
167                cache.coverage.index(&self.offset_data(), glyph)?,
168                &cache.seconds,
169            ),
170            _ => (
171                coverage_index(self.coverage(), glyph)?,
172                &hb_set_digest_t::full(),
173            ),
174        };
175        self.ligature_sets()
176            .get(index as usize)
177            .ok()
178            .and_then(|set| set.apply(ctx, seconds))
179    }
180
181    fn external_cache_create(&self, mode: SubtableExternalCacheMode) -> SubtableExternalCache {
182        match mode {
183            SubtableExternalCacheMode::Full => SubtableExternalCache::LigatureSubstFormat1Cache(
184                Box::new(LigatureSubstFormat1Cache::new(collect_seconds(self))),
185            ),
186            SubtableExternalCacheMode::Small => {
187                if let Some(coverage) =
188                    CoverageInfo::new(&self.offset_data(), self.coverage_offset().to_u32() as u16)
189                {
190                    SubtableExternalCache::LigatureSubstFormat1SmallCache(
191                        LigatureSubstFormat1SmallCache {
192                            coverage,
193                            seconds: collect_seconds(self),
194                        },
195                    )
196                } else {
197                    SubtableExternalCache::None
198                }
199            }
200            SubtableExternalCacheMode::None => SubtableExternalCache::None,
201        }
202    }
203}

205fn collect_seconds(lig_subst: &LigatureSubstFormat1) -> hb_set_digest_t {
206    let mut seconds = hb_set_digest_t::new();
207    lig_subst
208        .ligature_sets()
209        .iter()
210        .filter_map(Result::ok)
211        .for_each(|lig_set| {
212            lig_set
213                .ligatures()
214                .iter()
215                .filter_map(Result::ok)
216                .for_each(|lig| {
217                    if let Some(gid) = lig.component_glyph_ids().first() {
218                        seconds.add(gid.get().into());
219                    } else {
220                        seconds = hb_set_digest_t::full();
221                    };
222                });
223        });
224    seconds
