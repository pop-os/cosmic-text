20impl MarkArrayExt for MarkArray<'_> {
21    fn apply(
22        &self,
23        ctx: &mut hb_ot_apply_context_t,
24        base_anchor: &AnchorTable,
25        mark_anchor: &AnchorTable,
26        glyph_pos: usize,
27    ) -> Option<()> {
28        // If this subtable doesn't have an anchor for this base and this class
29        // return `None` such that the subsequent subtables have a chance at it.
30
31        let (base_x, base_y) = ctx.face.ot_tables.resolve_anchor(base_anchor);
32        let (mark_x, mark_y) = ctx.face.ot_tables.resolve_anchor(mark_anchor);
33
34        ctx.buffer
35            .unsafe_to_break(Some(glyph_pos), Some(ctx.buffer.idx + 1));
36
37        let idx = ctx.buffer.idx;
38        let pos = ctx.buffer.cur_pos_mut();
39        pos.x_offset = base_x - mark_x;
40        pos.y_offset = base_y - mark_y;
41        pos.set_attach_type(attach_type::MARK);
42        pos.set_attach_chain((glyph_pos as isize - idx as isize) as i16);
43
44        ctx.buffer.scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_GPOS_ATTACHMENT;
45        ctx.buffer.idx += 1;
46
47        Some(())
48    }
49}

51impl Apply for MarkBasePosFormat1<'_> {
52    fn apply(&self, ctx: &mut hb_ot_apply_context_t) -> Option<()> {
53        let mark_glyph = ctx.buffer.cur(0).as_glyph();
54        let mark_index = self.mark_coverage().ok()?.get(mark_glyph)?;
55
56        let base_coverage = self.base_coverage().ok()?;
57        let last_base_until = ctx.last_base_until;
58        let mut last_base = ctx.last_base;
59
60        // Due to borrowing rules, we have this piece of code before creating the
61        // iterator, unlike in harfbuzz.
62        if ctx.last_base_until > ctx.buffer.idx as u32 {
63            ctx.last_base_until = 0;
64            ctx.last_base = -1;
65        }
66
67        // Now we search backwards for a non-mark glyph
68        // We don't use skippy_iter.prev() to avoid O(n^2) behavior.
69        let mut iter = skipping_iterator_t::new(ctx, false);
70        iter.set_lookup_props(u32::from(lookup_flags::IGNORE_MARKS));
71
72        let mut j = iter.buffer.idx;
73        while j > last_base_until as usize {
74            let mut _match = iter.match_at(j - 1, MatchSource::Info);
75            if _match == match_t::MATCH {
76                // https://github.com/harfbuzz/harfbuzz/issues/4124
77                if !accept(iter.buffer, j - 1)
78                    && base_coverage
79                        .get(iter.buffer.info[j - 1].as_glyph())
80                        .is_none()
81                {
82                    _match = match_t::SKIP;
83                }
84            }
85
86            if _match == match_t::MATCH {
87                last_base = j as i32 - 1;
88                break;
89            }
90
91            j -= 1;
92        }
93        ctx.last_base_until = ctx.buffer.idx as u32;
94        ctx.last_base = last_base;
95
96        if ctx.last_base == -1 {
97            ctx.buffer
98                .unsafe_to_concat_from_outbuffer(Some(0), Some(ctx.buffer.idx + 1));
99            return None;
100        }
101
102        let idx = ctx.last_base as u32;
103
104        let info = &ctx.buffer.info;
105
106        // Checking that matched glyph is actually a base glyph by GDEF is too strong; disabled
107        let base_glyph = info[idx as usize].as_glyph();
108        let Some(base_index) = self.base_coverage().ok()?.get(base_glyph) else {
109            ctx.buffer
110                .unsafe_to_concat_from_outbuffer(Some(idx as usize), Some(ctx.buffer.idx + 1));
111            return None;
112        };
113
114        let mark_array = self.mark_array().ok()?;
115        let mark_record = mark_array.mark_records().get(mark_index as usize)?;
116        let mark_anchor = mark_record.mark_anchor(mark_array.offset_data()).ok()?;
117
118        let base_array = self.base_array().ok()?;
119        let base_record = base_array.base_records().get(base_index as usize).ok()?;
120        let base_anchor = base_record
121            .base_anchors(base_array.offset_data())
122            .get(mark_record.mark_class() as usize)?
123            .ok()?;
124
125        mark_array.apply(ctx, &base_anchor, &mark_anchor, idx as usize)
126    }
127}

129fn accept(buffer: &hb_buffer_t, idx: usize) -> bool {
130    /* We only want to attach to the first of a MultipleSubst sequence.
131     * https://github.com/harfbuzz/harfbuzz/issues/740
132     * Reject others...
133     * ...but stop if we find a mark in the MultipleSubst sequence:
134     * https://github.com/harfbuzz/harfbuzz/issues/1020 */
135    !buffer.info[idx].multiplied()
136        || 0 == buffer.info[idx].lig_comp()
137        || (idx == 0
138            || buffer.info[idx - 1].is_mark()
139            || !buffer.info[idx - 1].multiplied()
140            || buffer.info[idx].lig_id() != buffer.info[idx - 1].lig_id()
141            || buffer.info[idx].lig_comp() != buffer.info[idx - 1].lig_comp() + 1)
142}

144impl Apply for MarkMarkPosFormat1<'_> {
145    fn apply(&self, ctx: &mut hb_ot_apply_context_t) -> Option<()> {
146        let mark1_glyph = ctx.buffer.cur(0).as_glyph();
147        let mark1_index = self.mark1_coverage().ok()?.get(mark1_glyph)?;
148        let lookup_props = ctx.lookup_props;
149        // Now we search backwards for a suitable mark glyph until a non-mark glyph
150        let mut iter = skipping_iterator_t::new(ctx, false);
151        iter.reset_fast(iter.buffer.idx);
152        iter.set_lookup_props(lookup_props & !u32::from(lookup_flags::IGNORE_FLAGS));
153
154        let mut unsafe_from = 0;
155        if !iter.prev(Some(&mut unsafe_from)) {
156            iter.buffer
157                .unsafe_to_concat_from_outbuffer(Some(unsafe_from), Some(iter.buffer.idx + 1));
158            return None;
159        }
160
161        let iter_idx = iter.index();
162        if !ctx.buffer.info[iter_idx].is_mark() {
163            ctx.buffer
164                .unsafe_to_concat_from_outbuffer(Some(iter_idx), Some(ctx.buffer.idx + 1));
165            return None;
166        }
167
168        let id1 = ctx.buffer.cur(0).lig_id();
169        let id2 = ctx.buffer.info[iter_idx].lig_id();
170        let comp1 = ctx.buffer.cur(0).lig_comp();
171        let comp2 = ctx.buffer.info[iter_idx].lig_comp();
172
173        let matches = if id1 == id2 {
174            // Marks belonging to the same base
175            // or marks belonging to the same ligature component.
176            id1 == 0 || comp1 == comp2
177        } else {
178            // If ligature ids don't match, it may be the case that one of the marks
179            // itself is a ligature.  In which case match.
180            (id1 > 0 && comp1 == 0) || (id2 > 0 && comp2 == 0)
181        };
182
183        if !matches {
184            ctx.buffer
185                .unsafe_to_concat_from_outbuffer(Some(iter_idx), Some(ctx.buffer.idx + 1));
186            return None;
187        }
188
189        let mark2_glyph = ctx.buffer.info[iter_idx].as_glyph();
190        let mark2_index = self.mark2_coverage().ok()?.get(mark2_glyph)?;
191
192        let mark1_array = self.mark1_array().ok()?;
193        let mark1_record = mark1_array.mark_records().get(mark1_index as usize)?;
194        let mark1_anchor = mark1_record.mark_anchor(mark1_array.offset_data()).ok()?;
195
196        let base_array = self.mark2_array().ok()?;
197        let base_record = base_array.mark2_records().get(mark2_index as usize).ok()?;
198        let base_anchor = base_record
199            .mark2_anchors(base_array.offset_data())
200            .get(mark1_record.mark_class() as usize)?
201            .ok()?;
202
203        mark1_array.apply(ctx, &base_anchor, &mark1_anchor, iter_idx)
204    }
205}

207impl Apply for MarkLigPosFormat1<'_> {
208    fn apply(&self, ctx: &mut hb_ot_apply_context_t) -> Option<()> {
209        let mark_glyph = ctx.buffer.cur(0).as_glyph();
210        let mark_index = self.mark_coverage().ok()?.get(mark_glyph)? as usize;
211
212        // Due to borrowing rules, we have this piece of code before creating the
213        // iterator, unlike in harfbuzz.
214        if ctx.last_base_until > ctx.buffer.idx as u32 {
215            ctx.last_base_until = 0;
216            ctx.last_base = -1;
217        }
218
219        let last_base_until = ctx.last_base_until;
220        let mut last_base = ctx.last_base;
221
222        // Now we search backwards for a non-mark glyph
223        let mut iter = skipping_iterator_t::new(ctx, false);
224        iter.set_lookup_props(u32::from(lookup_flags::IGNORE_MARKS));
225
226        let mut j = iter.buffer.idx;
227        while j > last_base_until as usize {
228            let mut _match = iter.match_at(j - 1, MatchSource::Info);
229            if _match == match_t::MATCH {
230                last_base = j as i32 - 1;
231                break;
232            }
233            j -= 1;
234        }
235
236        ctx.last_base_until = ctx.buffer.idx as u32;
237        ctx.last_base = last_base;
238
239        if ctx.last_base == -1 {
240            ctx.buffer
241                .unsafe_to_concat_from_outbuffer(Some(0), Some(ctx.buffer.idx + 1));
242            return None;
243        }
244
245        let idx = ctx.last_base as usize;
246
247        // Checking that matched glyph is actually a ligature by GDEF is too strong; disabled
248
249        let lig_glyph = ctx.buffer.info[idx].as_glyph();
250        let Some(lig_index) = self.ligature_coverage().ok()?.get(lig_glyph) else {
251            ctx.buffer
252                .unsafe_to_concat_from_outbuffer(Some(idx), Some(ctx.buffer.idx + 1));
253            return None;
254        };
255        let lig_attach = self
256            .ligature_array()
257            .ok()?
258            .ligature_attaches()
259            .get(lig_index as usize)
260            .ok()?;
261
262        // Find component to attach to
263        let comp_count = lig_attach.component_count();
264        if comp_count == 0 {
265            ctx.buffer
266                .unsafe_to_concat_from_outbuffer(Some(idx), Some(ctx.buffer.idx + 1));
267            return None;
268        }
269
270        // We must now check whether the ligature ID of the current mark glyph
271        // is identical to the ligature ID of the found ligature.  If yes, we
272        // can directly use the component index.  If not, we attach the mark
273        // glyph to the last component of the ligature.
274        let lig_id = ctx.buffer.info[idx].lig_id();
275        let mark_id = ctx.buffer.cur(0).lig_id();
276        let mark_comp = u16::from(ctx.buffer.cur(0).lig_comp());
277        let matches = lig_id != 0 && lig_id == mark_id && mark_comp > 0;
278        let comp_index = if matches {
279            mark_comp.min(comp_count)
280        } else {
281            comp_count
282        } - 1;
283
284        let mark_array = self.mark_array().ok()?;
285        let mark_record = mark_array.mark_records().get(mark_index)?;
286        let mark_anchor = mark_record.mark_anchor(mark_array.offset_data()).ok()?;
287
288        let base_record = lig_attach
289            .component_records()
290            .get(comp_index as usize)
291            .ok()?;
292        let base_anchor = base_record
293            .ligature_anchors(lig_attach.offset_data())
294            .get(mark_record.mark_class() as usize)?
295            .ok()?;
296
297        mark_array.apply(ctx, &base_anchor, &mark_anchor, idx)
298    }
