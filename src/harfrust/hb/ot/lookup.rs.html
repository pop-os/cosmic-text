32pub struct LookupData<'a> {
33    /// Offset of the lookup from the base of the layout table.
34    offset: usize,
35    /// True if the lookup comes from GSUB.
36    is_subst: bool,
37    /// Data of the layout table.
38    table_data: FontData<'a>,
39}

41pub trait LookupHost<'a> {
42    fn lookup_count(&self) -> u16;
43    fn lookup_data(&self, index: u16) -> Result<LookupData<'a>, ReadError>;
44}
45
46impl<'a> LookupHost<'a> for Gsub<'a> {
47    fn lookup_count(&self) -> u16 {
48        self.lookup_list()
49            .map(|list| list.lookup_count())
50            .unwrap_or_default()
51    }
52
53    fn lookup_data(&self, index: u16) -> Result<LookupData<'a>, ReadError> {
54        let list = self.lookup_list()?;
55        let offset = list
56            .lookup_offsets()
57            .get(index as usize)
58            .ok_or(ReadError::OutOfBounds)?
59            .get()
60            .to_usize()
61            + self.lookup_list_offset().to_usize();
62        Ok(LookupData {
63            offset,
64            is_subst: true,
65            table_data: self.offset_data(),
66        })
67    }
68}

108            Self { lookups }

114                .get_or_init(|| {

116                        .ok()
117                        .and_then(|data| LookupInfo::new(&data))
118                        .map(Box::new)
119                })
120                .as_ref()
121                .map(|v| &**v)
122        }

157fn is_reversed(table_data: FontData, lookup: &Lookup<()>, lookup_offset: usize) -> Option<bool> {
158    match lookup.lookup_type() {
159        // Reverse chain context
160        8 => Some(true),
161        // Extension table
162        7 => {
163            let offset = lookup_offset + lookup.subtable_offsets().first()?.get().to_usize();
164            let data = table_data.split_off(offset)?;
165            let ext = ExtensionSubstFormat1::<()>::read(data).ok()?;
166            Some(ext.extension_lookup_type() == 8)
167        }
168        _ => Some(false),
169    }

172/// Cached information about a lookup.
173#[derive(Default)]
174pub struct LookupInfo {
175    pub props: u32,
176    pub is_subst: bool,
177    pub is_reversed: bool,
178    pub digest: hb_set_digest_t,
179    pub subtable_cache_user_idx: Option<usize>,
180    pub subtables: Vec<SubtableInfo>,
181}
182
183impl LookupInfo {
184    pub fn new(data: &LookupData) -> Option<Self> {
185        let mut info = Self {
186            is_subst: data.is_subst,
187            ..Default::default()
188        };
189        let lookup_data = data.table_data.split_off(data.offset)?;
190        let lookup: Lookup<()> = Lookup::read(lookup_data).ok()?;
191        let lookup_type = lookup.lookup_type();
192        let lookup_flag = lookup.lookup_flag();
193        info.props = u32::from(lookup.lookup_flag().to_bits());
194        if lookup_flag.to_bits() & LookupFlag::USE_MARK_FILTERING_SET.to_bits() != 0 {
195            info.props |= (lookup.mark_filtering_set().unwrap_or_default() as u32) << 16;
196        }
197        if data.is_subst {
198            info.is_reversed =
199                is_reversed(data.table_data, &lookup, data.offset).unwrap_or_default();
200        }
201        let mut subtable_cache_user_cost = 0;
202        info.subtables.reserve(lookup.sub_table_count() as usize);
203        for (idx, subtable_offset) in lookup.subtable_offsets().iter().enumerate() {
204            let cache_mode = if idx < 8 {
205                SubtableExternalCacheMode::Full
206            } else {
207                SubtableExternalCacheMode::Small
208            };
209            let subtable_offset = subtable_offset.get().to_usize() + data.offset;
210            if let Some((subtable_info, cache_cost)) = SubtableInfo::new(
211                data.table_data,
212                subtable_offset as u32,
213                data.is_subst,
214                lookup_type as u8,
215                cache_mode,
216            ) {
217                info.digest.union(&subtable_info.digest);
218                if cache_cost > subtable_cache_user_cost {
219                    info.subtable_cache_user_idx = Some(info.subtables.len());
220                    subtable_cache_user_cost = cache_cost;
221                }
222                info.subtables.push(subtable_info);
223            }
224        }
225        info.subtables.shrink_to_fit();
226        Some(info)
227    }
228
229    pub fn props(&self) -> u32 {
230        self.props
231    }
232
233    pub fn is_reverse(&self) -> bool {
234        self.is_reversed
235    }
236
237    pub fn digest(&self) -> &hb_set_digest_t {
238        &self.digest
239    }
240}

242impl LookupInfo {
243    #[inline]
244    pub(crate) fn apply(
245        &self,
246        ctx: &mut hb_ot_apply_context_t,
247        table_data: &[u8],
248        use_hot_subtable_cache: bool,
249    ) -> Option<()> {
250        let glyph = ctx.buffer.cur(0).glyph_id;
251        for (subtable_idx, subtable_info) in self.subtables.iter().enumerate() {
252            if !subtable_info.digest.may_have(glyph) {
253                continue;
254            }
255            let is_cached =
256                use_hot_subtable_cache & (self.subtable_cache_user_idx == Some(subtable_idx));
257            if subtable_info.apply(ctx, table_data, is_cached).is_some() {
258                return Some(());
259            }
260        }
261        None
262    }
263
264    pub(crate) fn cache_enter(&self, ctx: &mut hb_ot_apply_context_t) -> bool {
265        let Some(idx) = self.subtable_cache_user_idx else {
266            return false;
267        };
268        let Some(subtable_info) = self.subtables.get(idx) else {
269            return false;
270        };
271        if matches!(
272            subtable_info.kind,
273            SubtableKind::ContextFormat2 | SubtableKind::ChainedContextFormat2
274        ) {
275            cache_enter(ctx)
276        } else {
277            false
278        }
279    }
280    pub(crate) fn cache_leave(&self, ctx: &mut hb_ot_apply_context_t) {
281        let Some(idx) = self.subtable_cache_user_idx else {
282            return;
283        };
284        let Some(subtable_info) = self.subtables.get(idx) else {
285            return;
286        };
287        if matches!(
288            subtable_info.kind,
289            SubtableKind::ContextFormat2 | SubtableKind::ChainedContextFormat2
290        ) {
291            cache_leave(ctx);
292        }
293    }
294}

296impl LookupInfo {
297    pub fn would_apply(&self, face: &hb_font_t, ctx: &WouldApplyContext) -> Option<bool> {
298        let glyph = ctx.glyphs[0];
299        if !self.digest.may_have(glyph.into()) {
300            return Some(false);
301        }
302        let table_index = if self.is_subst {
303            TableIndex::GSUB
304        } else {
305            TableIndex::GPOS
306        };
307        let table_data = face.ot_tables.table_data(table_index)?;
308        for subtable_info in &self.subtables {
309            if !subtable_info.digest.may_have(glyph.into()) {
310                continue;
311            }
312            let Some(data) = table_data.get(subtable_info.offset as usize..) else {
313                continue;
314            };
315            let data = FontData::new(data);
316            let result = match subtable_info.kind {
317                SubtableKind::SingleSubst1 => {
318                    SingleSubstFormat1::read(data).map(|t| t.would_apply(ctx))
319                }
320                SubtableKind::SingleSubst2 => {
321                    SingleSubstFormat2::read(data).map(|t| t.would_apply(ctx))
322                }
323                SubtableKind::MultipleSubst1 => {
324                    MultipleSubstFormat1::read(data).map(|t| t.would_apply(ctx))
325                }
326                SubtableKind::AlternateSubst1 => {
327                    AlternateSubstFormat1::read(data).map(|t| t.would_apply(ctx))
328                }
329                SubtableKind::LigatureSubst1 => {
330                    LigatureSubstFormat1::read(data).map(|t| t.would_apply(ctx))
331                }
332                SubtableKind::ReverseChainContext => {
333                    ReverseChainSingleSubstFormat1::read(data).map(|t| t.would_apply(ctx))
334                }
335                SubtableKind::ContextFormat1 => {
336                    SequenceContextFormat1::read(data).map(|t| t.would_apply(ctx))
337                }
338                SubtableKind::ContextFormat2 => {
339                    SequenceContextFormat2::read(data).map(|t| t.would_apply(ctx))
340                }
341                SubtableKind::ContextFormat3 => {
342                    SequenceContextFormat3::read(data).map(|t| t.would_apply(ctx))
343                }
344                SubtableKind::ChainedContextFormat1 => {
345                    ChainedSequenceContextFormat1::read(data).map(|t| t.would_apply(ctx))
346                }
347                SubtableKind::ChainedContextFormat2 => {
348                    ChainedSequenceContextFormat2::read(data).map(|t| t.would_apply(ctx))
349                }
350                SubtableKind::ChainedContextFormat3 => {
351                    ChainedSequenceContextFormat3::read(data).map(|t| t.would_apply(ctx))
352                }
353                _ => continue,
354            };
355            if result == Ok(true) {
356                return Some(true);
357            }
358        }
359        None
360    }
361}

363/// Cached information about a subtable.
364pub struct SubtableInfo {
365    /// The fully resolved type of the subtable.
366    pub kind: SubtableKind,
367    /// Byte offset to the subtable from the base of the GSUB or GPOS
368    /// table.
369    pub offset: u32,
370    pub digest: hb_set_digest_t,
371    pub apply_fns: [SubtableApplyFn; 2],
372    pub external_cache: SubtableExternalCache,
373}
374
375pub type SubtableApplyFn =
376    fn(&mut hb_ot_apply_context_t, &SubtableExternalCache, &[u8]) -> Option<()>;
377
378impl SubtableInfo {
379    #[inline]
380    pub(crate) fn apply(
381        &self,
382        ctx: &mut hb_ot_apply_context_t,
383        table_data: &[u8],
384        is_cached: bool,
385    ) -> Option<()> {
386        let subtable_data = table_data.get(self.offset as usize..)?;
387        self.apply_fns[is_cached as usize](ctx, &self.external_cache, subtable_data)
388    }
389}
390
391macro_rules! apply_fns {
392    ($apply:ident, $apply_cached:ident, $ty:ident) => {
393        fn $apply(
394            ctx: &mut hb_ot_apply_context_t,
395            external_cache: &SubtableExternalCache,
396            table_data: &[u8],
397        ) -> Option<()> {
398            let t = $ty::read(FontData::new(table_data)).ok()?;
399            t.apply_with_external_cache(ctx, external_cache)
400        }
401
402        fn $apply_cached(
403            ctx: &mut hb_ot_apply_context_t,
404            external_cache: &SubtableExternalCache,
405            table_data: &[u8],
406        ) -> Option<()> {
407            let t = $ty::read(FontData::new(table_data)).ok()?;
408            t.apply_cached(ctx, external_cache)
409        }
410    };
411}

487impl SubtableInfo {
488    fn new(
489        table_data: FontData,
490        subtable_offset: u32,
491        is_subst: bool,
492        lookup_type: u8,
493        cache_mode: SubtableExternalCacheMode,
494    ) -> Option<(Self, u32)> {
495        let data = table_data.split_off(subtable_offset as usize)?;
496        let maybe_external_cache = |s: &dyn Apply| s.external_cache_create(cache_mode);
497        let (kind, (external_cache, cache_cost, coverage), apply_fns): (
498            SubtableKind,
499            (SubtableExternalCache, u32, CoverageTable),
500            [SubtableApplyFn; 2],
501        ) = match (is_subst, lookup_type) {
502            (true, 1) => match SingleSubst::read(data).ok()? {
503                SingleSubst::Format1(s) => (
504                    SubtableKind::SingleSubst1,
505                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
506                    [single_subst1, single_subst1_cached as _],
507                ),
508                SingleSubst::Format2(s) => (
509                    SubtableKind::SingleSubst2,
510                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
511                    [single_subst2, single_subst2_cached as _],
512                ),
513            },
514            (false, 1) => match SinglePos::read(data).ok()? {
515                SinglePos::Format1(s) => (
516                    SubtableKind::SinglePos1,
517                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
518                    [single_pos1, single_pos1_cached as _],
519                ),
520                SinglePos::Format2(s) => (
521                    SubtableKind::SinglePos2,
522                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
523                    [single_pos2, single_pos2_cached as _],
524                ),
525            },
526            (true, 2) => (
527                SubtableKind::MultipleSubst1,
528                MultipleSubstFormat1::read(data).ok().and_then(|t| {
529                    Some((maybe_external_cache(&t), t.cache_cost(), t.coverage().ok()?))
530                })?,
531                [multiple_subst1, multiple_subst1_cached as _],
532            ),
533            (false, 2) => match PairPos::read(data).ok()? {
534                PairPos::Format1(s) => (
535                    SubtableKind::PairPos1,
536                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
537                    [pair_pos1, pair_pos1_cached as _],
538                ),
539                PairPos::Format2(s) => (
540                    SubtableKind::PairPos2,
541                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
542                    [pair_pos2, pair_pos2_cached as _],
543                ),
544            },
545            (true, 3) => (
546                SubtableKind::AlternateSubst1,
547                AlternateSubstFormat1::read(data).ok().and_then(|t| {
548                    Some((maybe_external_cache(&t), t.cache_cost(), t.coverage().ok()?))
549                })?,
550                [alternate_subst1, alternate_subst1_cached as _],
551            ),
552            (false, 3) => (
553                SubtableKind::CursivePos1,
554                CursivePosFormat1::read(data).ok().and_then(|t| {
555                    Some((maybe_external_cache(&t), t.cache_cost(), t.coverage().ok()?))
556                })?,
557                [cursive_pos1, cursive_pos1_cached as _],
558            ),
559            (true, 4) => (
560                SubtableKind::LigatureSubst1,
561                LigatureSubstFormat1::read(data).ok().and_then(|t| {
562                    Some((maybe_external_cache(&t), t.cache_cost(), t.coverage().ok()?))
563                })?,
564                [ligature_subst1, ligature_subst1_cached as _],
565            ),
566            (false, 4) => (
567                SubtableKind::MarkBasePos1,
568                MarkBasePosFormat1::read(data).ok().and_then(|t| {
569                    Some((
570                        maybe_external_cache(&t),
571                        t.cache_cost(),
572                        t.mark_coverage().ok()?,
573                    ))
574                })?,
575                [mark_base_pos1, mark_base_pos1_cached as _],
576            ),
577            (true, 5) | (false, 7) => match SequenceContext::read(data).ok()? {
578                SequenceContext::Format1(s) => (
579                    SubtableKind::ContextFormat1,
580                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
581                    [context1, context1_cached as _],
582                ),
583                SequenceContext::Format2(s) => (
584                    SubtableKind::ContextFormat2,
585                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
586                    [context2, context2_cached as _],
587                ),
588                SequenceContext::Format3(s) => (
589                    SubtableKind::ContextFormat3,
590                    (
591                        maybe_external_cache(&s),
592                        s.cache_cost(),
593                        s.coverages().get(0).ok()?,
594                    ),
595                    [context3, context3_cached as _],
596                ),
597            },
598            (false, 5) => (
599                SubtableKind::MarkLigPos1,
600                MarkLigPosFormat1::read(data).ok().and_then(|t| {
601                    Some((
602                        maybe_external_cache(&t),
603                        t.cache_cost(),
604                        t.mark_coverage().ok()?,
605                    ))
606                })?,
607                [mark_lig_pos1, mark_lig_pos1_cached as _],
608            ),
609            (true, 6) | (false, 8) => match ChainedSequenceContext::read(data).ok()? {
610                ChainedSequenceContext::Format1(s) => (
611                    SubtableKind::ChainedContextFormat1,
612                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
613                    [chained_context1, chained_context1_cached as _],
614                ),
615                ChainedSequenceContext::Format2(s) => (
616                    SubtableKind::ChainedContextFormat2,
617                    (maybe_external_cache(&s), s.cache_cost(), s.coverage().ok()?),
618                    [chained_context2, chained_context2_cached as _],
619                ),
620                ChainedSequenceContext::Format3(s) => (
621                    SubtableKind::ChainedContextFormat3,
622                    (
623                        maybe_external_cache(&s),
624                        s.cache_cost(),
625                        s.input_coverages().get(0).ok()?,
626                    ),
627                    [chained_context3, chained_context3_cached as _],
628                ),
629            },
630            (true, 7) | (false, 9) => {
631                let ext = ExtensionSubstFormat1::<'_, ()>::read(data).ok()?;
632                let ext_type = ext.extension_lookup_type() as u8;
633                let ext_offset = ext.extension_offset().to_u32();
634                return Self::new(
635                    table_data,
636                    subtable_offset.checked_add(ext_offset)?,
637                    is_subst,
638                    ext_type,
639                    cache_mode,
640                );
641            }
642            (false, 6) => (
643                SubtableKind::MarkMarkPos1,
644                MarkMarkPosFormat1::read(data).ok().and_then(|t| {
645                    Some((
646                        maybe_external_cache(&t),
647                        t.cache_cost(),
648                        t.mark1_coverage().ok()?,
649                    ))
650                })?,
651                [mark_mark_pos1, mark_mark_pos1_cached as _],
652            ),
653            (true, 8) => (
654                SubtableKind::ReverseChainContext,
655                ReverseChainSingleSubstFormat1::read(data)
656                    .ok()
657                    .and_then(|t| {
658                        Some((maybe_external_cache(&t), t.cache_cost(), t.coverage().ok()?))
659                    })?,
660                [rev_chain_single_subst1, rev_chain_single_subst1_cached as _],
661            ),
662            _ => return None,
663        };
664        let mut digest = hb_set_digest_t::new();
665        digest.add_coverage(&coverage);
666        Some((
667            SubtableInfo {
668                kind,
669                offset: subtable_offset,
670                digest,
671                apply_fns,
672                external_cache,
673            },
674            cache_cost,
675        ))
676    }
677}
678
679fn cache_enter(ctx: &mut hb_ot_apply_context_t) -> bool {
680    if !ctx.buffer.try_allocate_var(GlyphInfo::SYLLABLE_VAR) {
681        return false;
682    }
683    for info in &mut ctx.buffer.info {
684        info.set_syllable(255);
685    }
686    ctx.new_syllables = Some(255);
687    true
688}
689
690fn cache_leave(ctx: &mut hb_ot_apply_context_t) {
691    ctx.new_syllables = None;
692    ctx.buffer.deallocate_var(GlyphInfo::SYLLABLE_VAR);
693}
