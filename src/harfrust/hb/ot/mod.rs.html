36impl OtCache {
37    pub fn new(font: &FontRef) -> Self {
38        let gsub = font
39            .gsub()
40            .map(|t| LookupCache::new(&t))
41            .unwrap_or_default();
42        let gpos = font
43            .gpos()
44            .map(|t| LookupCache::new(&t))
45            .unwrap_or_default();
46        let mut gdef_mark_set_digests = Vec::new();
47        if let Ok(gdef) = font.gdef() {
48            if let Some(Ok(mark_sets)) = gdef.mark_glyph_sets_def() {
49                gdef_mark_set_digests.extend(mark_sets.coverages().iter().map(|set| {
50                    set.ok()
51                        .map(|coverage| hb_set_digest_t::from_coverage(&coverage))
52                        .unwrap_or_default()
53                }));
54            }
55        }
56        Self {
57            gsub,
58            gpos,
59            gdef_glyph_props_cache: MappingCache::new(),
60            gdef_mark_set_digests,
61        }
62    }
63}

103impl<'a> GdefTable<'a> {

105        if let Some(gdef) = table_ranges.gdef.resolve_table::<Gdef>(font) {
106            let classes = gdef.glyph_class_def().transpose().ok().flatten();
107            let mark_classes = gdef.mark_attach_class_def().transpose().ok().flatten();
108            let mark_sets = gdef
109                .mark_glyph_sets_def()

112                .flatten()

114            Self {
115                table: Some(gdef),
116                classes,
117                mark_classes,
118                mark_sets,
119            }
120        } else {
121            Self::default()
122        }

203    pub(crate) fn glyph_props(&self, glyph: GlyphId) -> u16 {
204        let glyph = glyph.to_u32();
205
206        if let Some(props) = self.gdef_glyph_props_cache.get(glyph) {
207            return props as u16;
208        }
209
210        let props = match self.glyph_class(glyph) {
211            1 => GlyphPropsFlags::BASE_GLYPH.bits(),
212            2 => GlyphPropsFlags::LIGATURE.bits(),
213            3 => {
214                let class = self.glyph_mark_attachment_class(glyph);
215                (class << 8) | GlyphPropsFlags::MARK.bits()
216            }
217            _ => 0,
218        };
219
220        self.gdef_glyph_props_cache.set(glyph, props as u32);
221
222        props
223    }

276    pub(super) fn resolve_anchor(&self, anchor: &AnchorTable) -> (i32, i32) {
277        let mut x = anchor.x_coordinate() as i32;
278        let mut y = anchor.y_coordinate() as i32;
279        if let Some(vs) = self.var_store.as_ref() {
280            let delta = |val: Option<Result<DeviceOrVariationIndex<'_>, ReadError>>| match val {
281                Some(Ok(DeviceOrVariationIndex::VariationIndex(varix))) => vs
282                    .compute_delta(
283                        DeltaSetIndex {
284                            outer: varix.delta_set_outer_index(),
285                            inner: varix.delta_set_inner_index(),
286                        },
287                        self.coords,
288                    )
289                    .unwrap_or_default(),
290                _ => 0,
291            };
292            x += delta(anchor.x_device());
293            y += delta(anchor.y_device());
294        }
295        (x, y)
296    }

365    pub(crate) fn feature_variation_index(&self, coords: &[F2Dot14]) -> Option<u32> {
366        let feature_variations = self.feature_variations()?;
367        for (index, rec) in feature_variations
368            .feature_variation_records()
369            .iter()
370            .enumerate()
371        {
372            // If the ConditionSet offset is 0, this is treated as the
373            // universal condition: all contexts are matched.
374            if rec.condition_set_offset().is_null() {
375                return Some(index as u32);
376            }
377            let Some(Ok(condition_set)) = rec.condition_set(feature_variations.offset_data())
378            else {
379                continue;
380            };
381            // Otherwise, all conditions must be satisfied.
382            if condition_set
383                .conditions()
384                .iter()
385                // .. except we ignore errors
386                .filter_map(Result::ok)
387                .all(|cond| match cond {
388                    Condition::Format1AxisRange(format1) => {
389                        let coord = coords
390                            .get(format1.axis_index() as usize)
391                            .copied()
392                            .unwrap_or_default();
393                        coord >= format1.filter_range_min_value()
394                            && coord <= format1.filter_range_max_value()
395                    }
396                    _ => false,
397                })
398            {
399                return Some(index as u32);
400            }
401        }
402        None
403    }

405    pub(crate) fn feature_substitution(
406        &self,
407        variation_index: u32,
408        feature_index: u16,
409    ) -> Option<Feature<'a>> {
410        let feature_variations = self.feature_variations()?;
411        let record = feature_variations
412            .feature_variation_records()
413            .get(variation_index as usize)?;
414        let subst_table = record
415            .feature_table_substitution(feature_variations.offset_data())?
416            .ok()?;
417        let subst_records = subst_table.substitutions();
418        match subst_records.binary_search_by_key(&feature_index, |subst| subst.feature_index()) {
419            Ok(ix) => Some(
420                subst_records
421                    .get(ix)?
422                    .alternate_feature(subst_table.offset_data())
423                    .ok()?,
424            ),
425            _ => None,
426        }
427    }

517fn coverage_index_cached(
518    coverage: impl Fn(GlyphId) -> Option<u16>,
519    gid: GlyphId,
520    cache: &MappingCache,
521) -> Option<u16> {
522    if let Some(index) = cache.get(gid.into()) {
523        if index == MappingCache::MAX_VALUE {
524            None
525        } else {
526            Some(index as u16)
527        }
528    } else {
529        let index = coverage(gid);
530        if let Some(index) = index {
531            if (index as u32) < MappingCache::MAX_VALUE {
532                cache.set(gid.into(), index as u32);
533            }
534            Some(index)
535        } else {
536            cache.set(gid.into(), MappingCache::MAX_VALUE);
537            None
538        }
539    }

589#[derive(Copy, Clone, Default, Debug)]
590pub(crate) struct CoverageInfo {
591    pub offset: u16,
592    pub format: u16,
593    pub count: u16,
594}
595
596impl CoverageInfo {
597    pub fn new(parent_data: &FontData, offset: u16) -> Option<Self> {
598        if offset == 0 {
599            return None;
600        }
601        let format = parent_data.read_at::<u16>(offset as usize).ok()?;
602        if format != 1 && format != 2 {
603            return None;
604        }
605        let count = parent_data.read_at::<u16>(offset as usize + 2).ok()?;
606        Some(Self {
607            offset,
608            format,
609            count,
610        })
611    }
612
613    pub fn index(&self, parent_data: &FontData, gid: GlyphId) -> Option<u16> {
614        if self.offset == 0 {
615            return None;
616        }
617        let gid = gid.to_u32();
618        let data_offset = self.offset as usize + 4;
619        let len = self.count as usize;
620        if self.format == 1 {
621            let glyphs = parent_data
622                .read_array::<BigEndian<GlyphId16>>(data_offset..data_offset + len * 2)
623                .ok()?;
624            glyphs
625                .binary_search_by_key(&gid, |g| g.get().to_u32())
626                .ok()
627                .map(|idx| idx as _)
628        } else {
629            use core::cmp::Ordering;
630            let records = parent_data
631                .read_array::<RangeRecord>(
632                    data_offset..data_offset + len * size_of::<RangeRecord>(),
633                )
634                .ok()?;
635            records
636                .binary_search_by(|rec| {
637                    if rec.end_glyph_id().to_u32() < gid {
638                        Ordering::Less
639                    } else if rec.start_glyph_id().to_u32() > gid {
640                        Ordering::Greater
641                    } else {
642                        Ordering::Equal
643                    }
644                })
645                .ok()
646                .map(|idx| {
647                    let rec = &records[idx];
648                    (rec.start_coverage_index() as u32 + gid - rec.start_glyph_id().to_u32()) as u16
649                })
650        }
651    }
652}

654#[derive(Copy, Clone, Default, Debug)]
655pub(crate) struct ClassDefInfo {
656    pub offset: u16,
657    pub format: u16,
658    // For format 1 only
659    pub start_glyph_id: u16,
660    pub count: u16,
661}
662
663impl ClassDefInfo {
664    pub fn new(parent_data: &FontData, offset: u16) -> Option<Self> {
665        if offset == 0 {
666            return None;
667        }
668        let format = parent_data.read_at::<u16>(offset as usize).ok()?;
669        if format != 1 && format != 2 {
670            return None;
671        }
672        let (start_glyph_id, count) = if format == 1 {
673            let start_glyph_id = parent_data.read_at::<u16>(offset as usize + 2).ok()?;
674            let count = parent_data.read_at::<u16>(offset as usize + 4).ok()?;
675            (start_glyph_id, count)
676        } else if format == 2 {
677            let count = parent_data.read_at::<u16>(offset as usize + 2).ok()?;
678            (0, count)
679        } else {
680            return None;
681        };
682        Some(Self {
683            offset,
684            format,
685            start_glyph_id,
686            count,
687        })
688    }
689
690    pub fn class(&self, parent_data: &FontData, gid: GlyphId) -> u16 {
691        let offset = self.offset as usize;
692        if offset == 0 {
693            return 0;
694        }
695        let gid = gid.to_u32();
696        if self.format == 1 {
697            let Some(idx) = gid.checked_sub(self.start_glyph_id as u32) else {
698                return 0;
699            };
700            if idx >= self.count as u32 {
701                return 0;
702            }
703            parent_data
704                .read_at::<u16>(offset + 6 + idx as usize * 2)
705                .unwrap_or(0)
706        } else {
707            use core::cmp::Ordering;
708            let start = offset + 4;
709            let end = start + self.count as usize * size_of::<ClassRangeRecord>();
710            let Ok(records) = parent_data.read_array::<ClassRangeRecord>(start..end) else {
711                return 0;
712            };
713            records
714                .binary_search_by(|rec| {
715                    if rec.end_glyph_id().to_u32() < gid {
716                        Ordering::Less
717                    } else if rec.start_glyph_id().to_u32() > gid {
718                        Ordering::Greater
719                    } else {
720                        Ordering::Equal
721                    }
722                })
723                .ok()
724                .map_or(0, |idx| records[idx].class())
725        }
726    }
