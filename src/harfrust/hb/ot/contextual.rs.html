278fn get_class_cached(class_def: &impl Fn(GlyphId) -> u16, info: &mut GlyphInfo) -> u16 {
279    let mut klass = info.syllable() as u16;
280    if klass < 255 {
281        return klass;
282    }
283    klass = class_def(info.as_glyph());
284    if klass < 255 {
285        info.set_syllable(klass as u8);
286    }
287
288    klass
289}

292    class_def: impl Fn(GlyphId) -> u16 + 'a,
293) -> impl Fn(&mut GlyphInfo, u16) -> bool + 'a {
294    move |info: &mut GlyphInfo, value| get_class_cached(&class_def, info) == value
295}

297fn get_class_cached1(class_def: &impl Fn(GlyphId) -> u16, info: &mut GlyphInfo) -> u16 {
298    let mut klass = (info.syllable() & 0x0F) as u16;
299    if klass < 15 {
300        return klass;
301    }
302
303    klass = class_def(info.as_glyph());
304
305    if klass < 15 {
306        info.set_syllable((info.syllable() & 0xF0) | klass as u8);
307    }
308
309    klass
310}

318fn get_class_cached2(class_def: &impl Fn(GlyphId) -> u16, info: &mut GlyphInfo) -> u16 {
319    let mut klass = (info.syllable() & 0xF0) as u16 >> 4;
320    if klass < 15 {
321        return klass;
322    }
323    klass = class_def(info.as_glyph());
324    if klass < 15 {
325        info.set_syllable((info.syllable() & 0x0F) | ((klass as u8) << 4));
326    }
327    klass
328}

429impl Apply for ChainedSequenceContextFormat3<'_> {
430    fn apply(&self, ctx: &mut hb_ot_apply_context_t) -> Option<()> {
431        let glyph = ctx.buffer.cur(0).as_glyph();
432
433        let input_coverages = self.input_coverages();
434        input_coverages.get(0).ok()?.get(glyph)?;
435
436        let backtrack_coverages = self.backtrack_coverages();
437        let lookahead_coverages = self.lookahead_coverages();
438
439        let back = |info: &mut GlyphInfo, index: u16| {
440            backtrack_coverages
441                .get(index as usize)
442                .is_ok_and(|cov| cov.get(info.glyph_id).is_some())
443        };
444
445        let ahead = |info: &mut GlyphInfo, index: u16| {
446            lookahead_coverages
447                .get(index as usize)
448                .is_ok_and(|cov| cov.get(info.glyph_id).is_some())
449        };
450
451        let input = |info: &mut GlyphInfo, index: u16| {
452            input_coverages
453                .get(index as usize + 1)
454                .is_ok_and(|cov| cov.get(info.glyph_id).is_some())
455        };
456
457        let mut end_index = ctx.buffer.idx;
458        let mut match_end = 0;
459
460        let input_matches = match_input(
461            ctx,
462            input_coverages.len() as u16 - 1,
463            input,
464            &mut match_end,
465            None,
466        );
467
468        if input_matches {
469            end_index = match_end;
470        }
471
472        if !(input_matches
473            && match_lookahead(
474                ctx,
475                lookahead_coverages.len() as u16,
476                ahead,
477                match_end,
478                &mut end_index,
479            ))
480        {
481            ctx.buffer
482                .unsafe_to_concat(Some(ctx.buffer.idx), Some(end_index));
483            return None;
484        }
485
486        let mut start_index = ctx.buffer.out_len;
487
488        if !match_backtrack(
489            ctx,
490            backtrack_coverages.len() as u16,
491            back,
492            &mut start_index,
493        ) {
494            ctx.buffer
495                .unsafe_to_concat_from_outbuffer(Some(start_index), Some(end_index));
496            return None;
497        }
498
499        ctx.buffer
500            .unsafe_to_break_from_outbuffer(Some(start_index), Some(end_index));
501        apply_lookup(
502            ctx,
503            input_coverages.len() - 1,
504            match_end,
505            self.seq_lookup_records(),
506        );
507
508        Some(())
509    }
510}

528trait ContextRule<'a>: FontRead<'a> {
529    type Input: ToU16 + 'a;
530
531    fn input(&self) -> &'a [Self::Input];
532    fn lookup_records(&self) -> &'a [SequenceLookupRecord];
533
534    fn apply(
535        &self,
536        ctx: &mut hb_ot_apply_context_t,
537        match_func: &impl Fn(&mut GlyphInfo, u16) -> bool,
538    ) -> Option<()> {
539        let inputs = self.input();
540        let match_func = |info: &mut GlyphInfo, index| {
541            let value = inputs.get(index as usize).unwrap().to_u16();
542            match_func(info, value)
543        };
544
545        let mut match_end = 0;
546
547        if match_input(ctx, inputs.len() as _, match_func, &mut match_end, None) {
548            ctx.buffer
549                .unsafe_to_break(Some(ctx.buffer.idx), Some(match_end));
550            apply_lookup(ctx, inputs.len(), match_end, self.lookup_records());
551            return Some(());
552        }
553        None
554    }
555}

581fn apply_context_rules<'a, 'b, R: ContextRule<'a>>(
582    ctx: &mut hb_ot_apply_context_t,
583    rules: &'b ArrayOfOffsets<'a, R, Offset16>,
584    match_func: impl Fn(&mut GlyphInfo, u16) -> bool,
585) -> Option<()> {
586    // TODO: In HarfBuzz, the following condition makes NotoNastaliqUrdu
587    // faster. But our lookup code is slower, so NOT using this condition
588    // makes us faster.  Reconsider when lookup code is faster.
589    //if rules.len() <= 4 {
590    if false {
591        for rule in rules.iter().filter_map(|r| r.ok()) {
592            if rule.apply(ctx, &match_func).is_some() {
593                return Some(());
594            };
595        }
596        return None;
597    }
598    // This version is optimized for speed by matching the first & second
599    // components of the rule here, instead of calling into the matching code.
600    let mut skippy_iter = skipping_iterator_t::with_match_fn(ctx, true, Some(match_always));
601    skippy_iter.reset(skippy_iter.buffer.idx);
602    skippy_iter.set_glyph_data(0);
603    let mut unsafe_to;
604    let mut unsafe_to2 = 0;
605    let mut second = None;
606    let first = if skippy_iter.next(None) {
607        let g1 = skippy_iter.index();
608        unsafe_to = Some(skippy_iter.index() + 1);
609        if skippy_iter.may_skip(&skippy_iter.buffer.info[g1]) != may_skip_t::SKIP_NO {
610            // Can't use the fast path if eg. the next char is a default-ignorable
611            // or other skippable.
612            for rule in rules.iter().filter_map(|r| r.ok()) {
613                if rule.apply(ctx, &match_func).is_some() {
614                    return Some(());
615                };
616            }
617            return None;
618        }
619        g1
620    } else {
621        // Failed to match a next glyph. Only try applying rules that have no
622        // further impact.
623        for rule in rules
624            .iter()
625            .filter_map(|r| r.ok())
626            .filter(|r| r.input().len() <= 1)
627        {
628            if rule.apply(ctx, &match_func).is_some() {
629                return Some(());
630            };
631        }
632        return None;
633    };
634    let matched = skippy_iter.next(None);
635    let g2 = skippy_iter.index();
636    if matched && skippy_iter.may_skip(&skippy_iter.buffer.info[g2]) == may_skip_t::SKIP_NO {
637        second = Some(g2);
638        unsafe_to2 = skippy_iter.index() + 1;
639    }
640    for rule in rules.iter().filter_map(|r| r.ok()) {
641        let inputs = rule.input();
642        let match_func2 = |info: &mut GlyphInfo, index| {
643            if let Some(value) = inputs.get(index as usize).map(|v| v.to_u16()) {
644                match_func(info, value)
645            } else {
646                false
647            }
648        };
649        if inputs.len() <= 1 || match_func2(&mut ctx.buffer.info[first], 0) {
650            if second.is_none()
651                || (inputs.len() <= 2 || match_func2(&mut ctx.buffer.info[second.unwrap()], 1))
652            {
653                if rule.apply(ctx, &match_func).is_some() {
654                    if let Some(unsafe_to) = unsafe_to {
655                        ctx.buffer
656                            .unsafe_to_concat(Some(ctx.buffer.idx), Some(unsafe_to));
657                    }
658                    return Some(());
659                }
660            } else {
661                unsafe_to = Some(unsafe_to2);
662            }
663        }
664    }
665    if let Some(unsafe_to) = unsafe_to {
666        ctx.buffer
667            .unsafe_to_concat(Some(ctx.buffer.idx), Some(unsafe_to));
668    }
669    None
670}

672trait ChainContextRule<'a>: ContextRule<'a> {
673    fn backtrack(&self) -> &'a [Self::Input];
674    fn lookahead(&self) -> &'a [Self::Input];
675
676    #[inline(always)]
677    fn apply_chain<
678        F1: Fn(&mut GlyphInfo, u16) -> bool,
679        F2: Fn(&mut GlyphInfo, u16) -> bool,
680        F3: Fn(&mut GlyphInfo, u16) -> bool,
681    >(
682        &self,
683        ctx: &mut hb_ot_apply_context_t,
684        match_funcs: &(F1, F2, F3),
685    ) -> Option<()> {
686        let input = self.input();
687        let backtrack = self.backtrack();
688        let lookahead = self.lookahead();
689
690        // NOTE: Whenever something in this method changes, we also need to
691        // change it in the `apply` implementation for ChainedContextLookup.
692        let f1 = |info: &mut GlyphInfo, index| {
693            let value = (*backtrack.get(index as usize).unwrap()).to_u16();
694            match_funcs.0(info, value)
695        };
696
697        let f2 = |info: &mut GlyphInfo, index| {
698            let value = (*lookahead.get(index as usize).unwrap()).to_u16();
699            match_funcs.2(info, value)
700        };
701
702        let f3 = |info: &mut GlyphInfo, index| {
703            let value = (*input.get(index as usize).unwrap()).to_u16();
704            match_funcs.1(info, value)
705        };
706
707        let mut end_index = ctx.buffer.idx;
708        let mut match_end = 0;
709
710        let input_matches = match_input(ctx, input.len() as u16, f3, &mut match_end, None);
711
712        if input_matches {
713            end_index = match_end;
714        }
715
716        if !(input_matches
717            && match_lookahead(ctx, lookahead.len() as u16, f2, match_end, &mut end_index))
718        {
719            ctx.buffer
720                .unsafe_to_concat(Some(ctx.buffer.idx), Some(end_index));
721            return None;
722        }
723
724        let mut start_index = ctx.buffer.out_len;
725
726        if !match_backtrack(ctx, backtrack.len() as u16, f1, &mut start_index) {
727            ctx.buffer
728                .unsafe_to_concat_from_outbuffer(Some(start_index), Some(end_index));
729            return None;
730        }
731
732        ctx.buffer
733            .unsafe_to_break_from_outbuffer(Some(start_index), Some(end_index));
734        apply_lookup(ctx, input.len(), match_end, self.lookup_records());
735
736        Some(())
737    }
738}

784fn apply_chain_context_rules<
785    'a,
786    'b,
787    R: ChainContextRule<'a>,
788    F1: Fn(&mut GlyphInfo, u16) -> bool,
789    F2: Fn(&mut GlyphInfo, u16) -> bool,
790    F3: Fn(&mut GlyphInfo, u16) -> bool,
791>(
792    ctx: &mut hb_ot_apply_context_t,
793    rules: &'b ArrayOfOffsets<'a, R, Offset16>,
794    match_funcs: (F1, F2, F3),
795) -> Option<()> {
796    // If the input skippy has non-auto joiners behavior (as in Indic shapers),
797    // skip this fast path, as we don't distinguish between input & lookahead
798    // matching in the fast path.
799    // https://github.com/harfbuzz/harfbuzz/issues/4813
800    if rules.len() <= 4 || !ctx.auto_zwnj || !ctx.auto_zwj {
801        for rule in rules.iter().filter_map(|r| r.ok()) {
802            if rule.apply_chain(ctx, &match_funcs).is_some() {
803                return Some(());
804            };
805        }
806        return None;
807    }
808    // This version is optimized for speed by matching the first & second
809    // components of the rule here, instead of calling into the matching code.
810    let mut skippy_iter = skipping_iterator_t::with_match_fn(ctx, true, Some(match_always));
811    skippy_iter.reset(skippy_iter.buffer.idx);
812    skippy_iter.set_glyph_data(0);
813    let mut unsafe_to;
814    let mut unsafe_to2 = 0;
815    let mut second = None;
816    let first = if skippy_iter.next(None) {
817        let g1 = skippy_iter.index();
818        unsafe_to = Some(skippy_iter.index() + 1);
819        if skippy_iter.may_skip(&skippy_iter.buffer.info[g1]) != may_skip_t::SKIP_NO {
820            // Can't use the fast path if eg. the next char is a default-ignorable
821            // or other skippable.
822            for rule in rules.iter().filter_map(|r| r.ok()) {
823                if rule.apply_chain(ctx, &match_funcs).is_some() {
824                    return Some(());
825                };
826            }
827            return None;
828        }
829        g1
830    } else {
831        // Failed to match a next glyph. Only try applying rules that have no
832        // further impact.
833        for rule in rules
834            .iter()
835            .filter_map(|r| r.ok())
836            .filter(|r| r.input().len() <= 1 && r.lookahead().is_empty())
837        {
838            if rule.apply_chain(ctx, &match_funcs).is_some() {
839                return Some(());
840            };
841        }
842        return None;
843    };
844    let matched = skippy_iter.next(None);
845    let g2 = skippy_iter.index();
846    if matched && skippy_iter.may_skip(&skippy_iter.buffer.info[g2]) == may_skip_t::SKIP_NO {
847        second = Some(g2);
848        unsafe_to2 = skippy_iter.index() + 1;
849    }
850    for rule in rules.iter().filter_map(|r| r.ok()) {
851        let input = rule.input();
852        let lookahead = rule.lookahead();
853        let match_input = |info: &mut GlyphInfo, index: usize| {
854            input
855                .get(index)
856                .is_some_and(|v| match_funcs.1(info, v.to_u16()))
857        };
858        let match_lookahead = |info: &mut GlyphInfo, index: usize| {
859            lookahead
860                .get(index)
861                .is_some_and(|v| match_funcs.2(info, v.to_u16()))
862        };
863        let len_p1 = input.len() + 1;
864        let matched_first = if len_p1 > 1 {
865            match_input(&mut ctx.buffer.info[first], 0)
866        } else {
867            lookahead.is_empty() || match_lookahead(&mut ctx.buffer.info[first], 0)
868        };
869        if matched_first {
870            let matched_second = if let Some(second) = second {
871                if len_p1 > 2 {
872                    match_input(&mut ctx.buffer.info[second], 1)
873                } else {
874                    (lookahead.len() <= 2 - len_p1)
875                        || match_lookahead(&mut ctx.buffer.info[second], 2 - len_p1)
876                }
877            } else {
878                true
879            };
880            if matched_second {
881                if rule.apply_chain(ctx, &match_funcs).is_some() {
882                    if let Some(unsafe_to) = unsafe_to {
883                        ctx.buffer
884                            .unsafe_to_concat(Some(ctx.buffer.idx), Some(unsafe_to));
885                    }
886                    return Some(());
887                }
888            } else {
889                unsafe_to = Some(unsafe_to2);
890            }
891        }
892    }
893    if let Some(unsafe_to) = unsafe_to {
894        ctx.buffer
895            .unsafe_to_concat(Some(ctx.buffer.idx), Some(unsafe_to));
896    }
897    None
898}
