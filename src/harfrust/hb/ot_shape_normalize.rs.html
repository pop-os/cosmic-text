53// HIGHLEVEL DESIGN:
54//
55// This file exports one main function: normalize().
56//
57// This function closely reflects the Unicode Normalization Algorithm,
58// yet it's different.
59//
60// Each shaper specifies whether it prefers decomposed (NFD) or composed (NFC).
61// The logic however tries to use whatever the font can support.
62//
63// In general what happens is that: each grapheme is decomposed in a chain
64// of 1:2 decompositions, marks reordered, and then recomposed if desired,
65// so far it's like Unicode Normalization.  However, the decomposition and
66// recomposition only happens if the font supports the resulting characters.
67//
68// The goals are:
69//
70//   - Try to render all canonically equivalent strings similarly.  To really
71//     achieve this we have to always do the full decomposition and then
72//     selectively recompose from there.  It's kinda too expensive though, so
73//     we skip some cases.  For example, if composed is desired, we simply
74//     don't touch 1-character clusters that are supported by the font, even
75//     though their NFC may be different.
76//
77//   - When a font has a precomposed character for a sequence but the 'ccmp'
78//     feature in the font is not adequate, use the precomposed character
79//     which typically has better mark positioning.
80//
81//   - When a font does not support a combining mark, but supports it precomposed
82//     with previous base, use that.  This needs the itemizer to have this
83//     knowledge too.  We need to provide assistance to the itemizer.
84//
85//   - When a font does not support a character but supports its canonical
86//     decomposition, well, use the decomposition.
87//
88//   - The shapers can customize the compose and decompose functions to
89//     offload some of their requirements to the normalizer.  For example, the
90//     Indic shaper may want to disallow recomposing of two matras.
91

113fn output_char(buffer: &mut hb_buffer_t, unichar: u32, glyph: u32) {
114    // This is very confusing indeed.
115    buffer.cur_mut(0).set_normalizer_glyph_index(glyph);
116    buffer.output_glyph(unichar);
117    // TODO: should be _hb_glyph_info_set_unicode_props (&buffer->prev(), buffer);
118    let mut flags = buffer.scratch_flags;
119    buffer.prev_mut().init_unicode_props(&mut flags);
120    buffer.scratch_flags = flags;
121}

132/// Returns 0 if didn't decompose, number of resulting characters otherwise.
133fn decompose(ctx: &mut hb_ot_shape_normalize_context_t, shortest: bool, ab: Codepoint) -> u32 {
134    let Some((a, b)) = (ctx.decompose)(ctx, ab) else {
135        return 0;
136    };
137
138    let a_glyph = ctx.face.get_nominal_glyph(a);
139    let b_glyph = if b != 0 {
140        match ctx.face.get_nominal_glyph(b) {
141            Some(glyph_id) => Some(glyph_id),
142            None => return 0,
143        }
144    } else {
145        None
146    };
147
148    if !shortest || a_glyph.is_none() {
149        let ret = decompose(ctx, shortest, a);
150        if ret != 0 {
151            if let Some(b_glyph) = b_glyph {
152                output_char(ctx.buffer, b, u32::from(b_glyph));
153                return ret + 1;
154            }
155            return ret;
156        }
157    }
158
159    if let Some(a_glyph) = a_glyph {
160        // Output a and b.
161        output_char(ctx.buffer, a, u32::from(a_glyph));
162        if let Some(b_glyph) = b_glyph {
163            output_char(ctx.buffer, b, u32::from(b_glyph));
164            return 2;
165        }
166        return 1;
167    }
168
169    0
170}

172fn decompose_current_character(ctx: &mut hb_ot_shape_normalize_context_t, shortest: bool) {
173    let u = ctx.buffer.cur(0).as_codepoint();
174    let glyph = ctx.face.get_nominal_glyph(u);
175
176    // TODO: different to harfbuzz, sync
177    if !shortest || glyph.is_none() {
178        if decompose(ctx, shortest, u) > 0 {
179            skip_char(ctx.buffer);
180            return;
181        }
182    }
183
184    // TODO: different to harfbuzz, sync
185    if let Some(glyph) = glyph {
186        next_char(ctx.buffer, u32::from(glyph));
187        return;
188    }
189
190    if ctx.buffer.cur(0).is_unicode_space() {
191        let space_type = u.space_fallback();
192        if space_type != hb_unicode_funcs_t::NOT_SPACE {
193            let space_glyph = ctx.face.get_nominal_glyph(0x0020).or(ctx.buffer.invisible);
194
195            if let Some(space_glyph) = space_glyph {
196                ctx.buffer
197                    .cur_mut(0)
198                    .set_unicode_space_fallback_type(space_type);
199                next_char(ctx.buffer, u32::from(space_glyph));
200                ctx.buffer.scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_SPACE_FALLBACK;
201                return;
202            }
203        }
204    }
205
206    // U+2011 is the only sensible character that is a no-break version of another character
207    // and not a space.  The space ones are handled already.  Handle this lone one.
208    if u == 0x2011 {
209        if let Some(other_glyph) = ctx.face.get_nominal_glyph(0x2010) {
210            next_char(ctx.buffer, u32::from(other_glyph));
211            return;
212        }
213    }
214
215    // Insert a .notdef glyph if decomposition failed.
216    next_char(ctx.buffer, 0);
217}

219fn handle_variation_selector_cluster(
220    ctx: &mut hb_ot_shape_normalize_context_t,
221    end: usize,
222    _: bool,
223) {
224    let face = ctx.face;
225
226    // Currently if there's a variation-selector we give-up on normalization, it's just too hard.
227    let buffer = &mut ctx.buffer;
228    while buffer.idx < end - 1 && buffer.successful {
229        if buffer.cur(1).as_codepoint().is_variation_selector() {
230            if let Some(glyph_id) = face.get_nominal_variant_glyph(
231                buffer.cur(0).as_codepoint(),
232                buffer.cur(1).as_codepoint(),
233            ) {
234                buffer
235                    .cur_mut(0)
236                    .set_normalizer_glyph_index(u32::from(glyph_id));
237                let unicode = buffer.cur(0).glyph_id;
238                buffer.replace_glyphs(2, 1, &[unicode]);
239            } else {
240                // Just pass on the two characters separately, let GSUB do its magic.
241                set_glyph(buffer.cur_mut(0), face);
242                buffer.next_glyph();
243
244                buffer.scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_VARIATION_SELECTOR_FALLBACK;
245
246                buffer.cur_mut(0).set_variation_selector(true);
247
248                if buffer.not_found_variation_selector.is_some() {
249                    buffer.cur_mut(0).clear_default_ignorable();
250                }
251
252                set_glyph(buffer.cur_mut(0), face);
253                buffer.next_glyph();
254            }
255
256            // Skip any further variation selectors.
257            while buffer.idx < end && buffer.cur(0).as_codepoint().is_variation_selector() {
258                set_glyph(buffer.cur_mut(0), face);
259                buffer.next_glyph();
260            }
261        } else {
262            set_glyph(buffer.cur_mut(0), face);
263            buffer.next_glyph();
264        }
265    }
266
267    if ctx.buffer.idx < end {
268        set_glyph(ctx.buffer.cur_mut(0), face);
269        ctx.buffer.next_glyph();
270    }
271}

273fn decompose_multi_char_cluster(
274    ctx: &mut hb_ot_shape_normalize_context_t,
275    end: usize,
276    short_circuit: bool,
277) {
278    let mut i = ctx.buffer.idx;
279    while i < end && ctx.buffer.successful {
280        if ctx.buffer.info[i].as_codepoint().is_variation_selector() {
281            handle_variation_selector_cluster(ctx, end, short_circuit);
282            return;
283        }
284        i += 1;
285    }
286
287    while ctx.buffer.idx < end && ctx.buffer.successful {
288        decompose_current_character(ctx, short_circuit);
289    }
290}

292fn compare_combining_class(pa: &GlyphInfo, pb: &GlyphInfo) -> bool {
293    let a = pa.modified_combining_class();
294    let b = pb.modified_combining_class();
295    a > b
296}

298pub fn _hb_ot_shape_normalize(
299    plan: &hb_ot_shape_plan_t,
300    buffer: &mut hb_buffer_t,
301    face: &hb_font_t,
302) {
303    if buffer.is_empty() {
304        return;
305    }
306
307    buffer.assert_unicode_vars();
308
309    let mut mode = plan.shaper.normalization_preference;
310    if mode == HB_OT_SHAPE_NORMALIZATION_MODE_AUTO {
311        if plan.has_gpos_mark {
312            // https://github.com/harfbuzz/harfbuzz/issues/653#issuecomment-423905920
313            // mode = Some(HB_OT_SHAPE_NORMALIZATION_MODE_DECOMPOSED);
314            mode = HB_OT_SHAPE_NORMALIZATION_MODE_COMPOSED_DIACRITICS;
315        } else {
316            mode = HB_OT_SHAPE_NORMALIZATION_MODE_COMPOSED_DIACRITICS;
317        }
318    }
319
320    let mut ctx = hb_ot_shape_normalize_context_t {
321        plan,
322        buffer,
323        face,
324        decompose: decompose_unicode,
325        compose: compose_unicode,
326    };
327    ctx.override_decompose_and_compose(plan.shaper.decompose, plan.shaper.compose);
328
329    let mut buffer = &mut ctx.buffer;
330
331    let always_short_circuit = mode == HB_OT_SHAPE_NORMALIZATION_MODE_NONE;
332    let might_short_circuit = always_short_circuit
333        || (mode != HB_OT_SHAPE_NORMALIZATION_MODE_DECOMPOSED
334            && mode != HB_OT_SHAPE_NORMALIZATION_MODE_COMPOSED_DIACRITICS_NO_SHORT_CIRCUIT);
335
336    // We do a fairly straightforward yet custom normalization process in three
337    // separate rounds: decompose, reorder, recompose (if desired).  Currently
338    // this makes two buffer swaps.  We can make it faster by moving the last
339    // two rounds into the inner loop for the first round, but it's more readable
340    // this way.
341
342    // First round, decompose
343    let mut all_simple = true;
344    {
345        buffer.clear_output();
346        let count = buffer.len;
347        buffer.idx = 0;
348        loop {
349            let mut end = buffer.idx + 1;
350            while end < count && !buffer.info[end].is_unicode_mark() {
351                end += 1;
352            }
353
354            if end < count {
355                // Leave one base for the marks to cluster with.
356                end -= 1;
357            }
358
359            // From idx to end are simple clusters.
360            if might_short_circuit {
361                let len = end - buffer.idx;
362                let mut done = 0;
363                while done < len {
364                    let cur = buffer.cur_mut(done);
365                    cur.set_normalizer_glyph_index(match face.get_nominal_glyph(cur.glyph_id) {
366                        Some(glyph_id) => u32::from(glyph_id),
367                        None => break,
368                    });
369                    done += 1;
370                }
371                buffer.next_glyphs(done);
372            }
373
374            while buffer.idx < end && buffer.successful {
375                decompose_current_character(&mut ctx, might_short_circuit);
376                buffer = &mut ctx.buffer;
377            }
378
379            if buffer.idx == count || !buffer.successful {
380                break;
381            }
382
383            all_simple = false;
384
385            // Find all the marks now.
386            end = buffer.idx + 1;
387            while end < count && buffer.info[end].is_unicode_mark() {
388                end += 1;
389            }
390
391            // idx to end is one non-simple cluster.
392            decompose_multi_char_cluster(&mut ctx, end, always_short_circuit);
393            buffer = &mut ctx.buffer;
394
395            if buffer.idx >= count || !buffer.successful {
396                break;
397            }
398        }
399
400        buffer.sync();
401    }
402
403    // Second round, reorder (inplace)
404    if !all_simple {
405        let count = buffer.len;
406        let mut i = 0;
407        while i < count {
408            if buffer.info[i].modified_combining_class() == 0 {
409                i += 1;
410                continue;
411            }
412
413            let mut end = i + 1;
414            while end < count && buffer.info[end].modified_combining_class() != 0 {
415                end += 1;
416            }
417
418            // We are going to do a O(n^2).  Only do this if the sequence is short.
419            if end - i <= MAX_COMBINING_MARKS {
420                buffer.sort(i, end, compare_combining_class);
421
422                if let Some(reorder_marks) = ctx.plan.shaper.reorder_marks {
423                    reorder_marks(ctx.plan, buffer, i, end);
424                }
425            }
426
427            i = end + 1;
428        }
429    }
430    if buffer.scratch_flags & HB_BUFFER_SCRATCH_FLAG_HAS_CGJ != 0 {
431        // For all CGJ, check if it prevented any reordering at all.
432        // If it did NOT, then make it skippable.
433        // https://github.com/harfbuzz/harfbuzz/issues/554
434        for i in 1..buffer.len.saturating_sub(1) {
435            if buffer.info[i].glyph_id == 0x034F
436            /* CGJ */
437            {
438                let last = buffer.info[i - 1].modified_combining_class();
439                let next = buffer.info[i + 1].modified_combining_class();
440                if next == 0 || last <= next {
441                    buffer.info[i].unhide();
442                }
443            }
444        }
445    }
446
447    // Third round, recompose
448    if !all_simple
449        && buffer.successful
450        && (mode == HB_OT_SHAPE_NORMALIZATION_MODE_COMPOSED_DIACRITICS
451            || mode == HB_OT_SHAPE_NORMALIZATION_MODE_COMPOSED_DIACRITICS_NO_SHORT_CIRCUIT)
452    {
453        // As noted in the comment earlier, we don't try to combine
454        // ccc=0 chars with their previous Starter.
455
456        let count = buffer.len;
457        let mut starter = 0;
458        buffer.clear_output();
459        buffer.next_glyph();
460        while buffer.idx < count && buffer.successful {
461            // We don't try to compose a non-mark character with it's preceding starter.
462            // This is both an optimization to avoid trying to compose every two neighboring
463            // glyphs in most scripts AND a desired feature for Hangul.  Apparently Hangul
464            // fonts are not designed to mix-and-match pre-composed syllables and Jamo.
465            let cur = buffer.cur(0);
466            if cur.is_unicode_mark() &&
467                // If there's anything between the starter and this char, they should have CCC
468                // smaller than this character's.
469                (starter == buffer.out_len - 1
470                    || buffer.prev().modified_combining_class() < cur.modified_combining_class())
471            {
472                let a = buffer.out_info()[starter].as_codepoint();
473                let b = cur.as_codepoint();
474                if let Some(composed) = (ctx.compose)(&ctx, a, b) {
475                    if let Some(glyph_id) = face.get_nominal_glyph(composed) {
476                        // Copy to out-buffer.
477                        buffer = &mut ctx.buffer;
478                        buffer.next_glyph();
479                        if !buffer.successful {
480                            return;
481                        }
482
483                        // Merge and remove the second composable.
484                        buffer.merge_out_clusters(starter, buffer.out_len);
485                        buffer.out_len -= 1;
486
487                        // Modify starter and carry on.
488                        let mut flags = buffer.scratch_flags;
489                        let info = &mut buffer.out_info_mut()[starter];
490                        info.glyph_id = composed;
491                        info.set_normalizer_glyph_index(u32::from(glyph_id));
492                        info.init_unicode_props(&mut flags);
493                        buffer.scratch_flags = flags;
494
495                        continue;
496                    }
497                }
498            }
499
500            // Blocked, or doesn't compose.
501            buffer = &mut ctx.buffer;
502            buffer.next_glyph();
503
504            if buffer.prev().modified_combining_class() == 0 {
505                starter = buffer.out_len - 1;
506            }
507        }
508
509        buffer.sync();
510    }
