385    }

391    fn space_fallback(self) -> hb_unicode_funcs_t::space_t {
392        use hb_unicode_funcs_t::*;
393
394        // All GC=Zs chars that can use a fallback.
395        match self {
396            0x0020 => SPACE,             // SPACE
397            0x00A0 => SPACE,             // NO-BREAK SPACE
398            0x2000 => SPACE_EM_2,        // EN QUAD
399            0x2001 => SPACE_EM,          // EM QUAD
400            0x2002 => SPACE_EM_2,        // EN SPACE
401            0x2003 => SPACE_EM,          // EM SPACE
402            0x2004 => SPACE_EM_3,        // THREE-PER-EM SPACE
403            0x2005 => SPACE_EM_4,        // FOUR-PER-EM SPACE
404            0x2006 => SPACE_EM_6,        // SIX-PER-EM SPACE
405            0x2007 => SPACE_FIGURE,      // FIGURE SPACE
406            0x2008 => SPACE_PUNCTUATION, // PUNCTUATION SPACE

418    }

420    fn modified_combining_class(self) -> u8 {
421        let u = self;
422
423        // Reorder SAKOT to ensure it comes after any tone marks.
424        if u == 0x1A60 {
425            return 254;
426        }
427
428        // Reorder PADMA to ensure it comes after any vowel marks.
429        if u == 0x0FC6 {
430            return 254;
431        }
432
433        // Reorder TSA -PHRU to reorder before U+0F74
434        if u == 0x0F39 {
435            return 127;
436        }
437
438        let k = u.combining_class();
439
440        MODIFIED_COMBINING_CLASS[k as usize]
441    }

443    fn mirrored(self) -> Option<Codepoint> {
444        let delta = _hb_ucd_bmg(self as usize);
445        if delta == 0 {
446            None
447        } else {
448            Some(((self as i32).wrapping_add(delta as i32)) as u32)
449        }
450    }

615    /// Default_Ignorable codepoints:
616    ///
617    /// Note: While U+115F, U+1160, U+3164 and U+FFA0 are Default_Ignorable,
618    /// we do NOT want to hide them, as the way Uniscribe has implemented them
619    /// is with regular spacing glyphs, and that's the way fonts are made to work.
620    /// As such, we make exceptions for those four.
621    /// Also ignoring U+1BCA0..1BCA3. https://github.com/harfbuzz/harfbuzz/issues/503
622    ///
623    /// Unicode 14.0:
624    /// $ grep '; Default_Ignorable_Code_Point ' DerivedCoreProperties.txt | sed 's/;.*#/#/'
625    /// 00AD          # Cf       SOFT HYPHEN
626    /// 034F          # Mn       COMBINING GRAPHEME JOINER
627    /// 061C          # Cf       ARABIC LETTER MARK
628    /// 115F..1160    # Lo   [2] HANGUL CHOSEONG FILLER..HANGUL JUNGSEONG FILLER
629    /// 17B4..17B5    # Mn   [2] KHMER VOWEL INHERENT AQ..KHMER VOWEL INHERENT AA
630    /// 180B..180D    # Mn   [3] MONGOLIAN FREE VARIATION SELECTOR ONE..MONGOLIAN FREE VARIATION SELECTOR THREE
631    /// 180E          # Cf       MONGOLIAN VOWEL SEPARATOR
632    /// 180F          # Mn       MONGOLIAN FREE VARIATION SELECTOR FOUR
633    /// 200B..200F    # Cf   [5] ZERO WIDTH SPACE..RIGHT-TO-LEFT MARK
634    /// 202A..202E    # Cf   [5] LEFT-TO-RIGHT EMBEDDING..RIGHT-TO-LEFT OVERRIDE
635    /// 2060..2064    # Cf   [5] WORD JOINER..INVISIBLE PLUS
636    /// 2065          # Cn       <reserved-2065>
637    /// 2066..206F    # Cf  [10] LEFT-TO-RIGHT ISOLATE..NOMINAL DIGIT SHAPES
638    /// 3164          # Lo       HANGUL FILLER
639    /// FE00..FE0F    # Mn  [16] VARIATION SELECTOR-1..VARIATION SELECTOR-16
640    /// FEFF          # Cf       ZERO WIDTH NO-BREAK SPACE
641    /// FFA0          # Lo       HALFWIDTH HANGUL FILLER
642    /// FFF0..FFF8    # Cn   [9] <reserved-FFF0>..<reserved-FFF8>
643    /// 1BCA0..1BCA3  # Cf   [4] SHORTHAND FORMAT LETTER OVERLAP..SHORTHAND FORMAT UP STEP
644    /// 1D173..1D17A  # Cf   [8] MUSICAL SYMBOL BEGIN BEAM..MUSICAL SYMBOL END PHRASE
645    /// E0000         # Cn       <reserved-E0000>
646    /// E0001         # Cf       LANGUAGE TAG
647    /// E0002..E001F  # Cn  [30] <reserved-E0002>..<reserved-E001F>
648    /// E0020..E007F  # Cf  [96] TAG SPACE..CANCEL TAG
649    /// E0080..E00FF  # Cn [128] <reserved-E0080>..<reserved-E00FF>
650    /// E0100..E01EF  # Mn [240] VARIATION SELECTOR-17..VARIATION SELECTOR-256
651    /// E01F0..E0FFF  # Cn [3600] <reserved-E01F0>..<reserved-E0FFF>
652    fn is_default_ignorable(self) -> bool {
653        let ch = self;
654        let plane = ch >> 16;
655        if plane == 0 {
656            // BMP
657            let page = ch >> 8;
658            match page {
659                0x00 => ch == 0x00AD,
660                0x03 => ch == 0x034F,
661                0x06 => ch == 0x061C,
662                0x17 => (0x17B4..=0x17B5).contains(&ch),
663                0x18 => (0x180B..=0x180E).contains(&ch),
664                0x20 => {
665                    (0x200B..=0x200F).contains(&ch)
666                        || (0x202A..=0x202E).contains(&ch)
667                        || (0x2060..=0x206F).contains(&ch)
668                }
669                0xFE => (0xFE00..=0xFE0F).contains(&ch) || ch == 0xFEFF,
670                0xFF => (0xFFF0..=0xFFF8).contains(&ch),
671                _ => false,
672            }
673        } else {
674            // Other planes
675            match plane {
676                0x01 => (0x1D173..=0x1D17A).contains(&ch),
677                0x0E => (0xE0000..=0xE0FFF).contains(&ch),
678                _ => false,
679            }
680        }
681    }

683    fn is_variation_selector(self) -> bool {
684        // U+180B..180D, U+180F MONGOLIAN FREE VARIATION SELECTORs are handled in the
685        //Arabic shaper. No need to match them here.
686        (0x0FE00..=0x0FE0F).contains(&self) || // VARIATION SELECTOR - 1..16
687        (0xE0100..=0xE01EF).contains(&self) // VARIATION SELECTOR - 17..256
688    }

742const S_BASE: u32 = 0xAC00;
743const L_BASE: u32 = 0x1100;
744const V_BASE: u32 = 0x1161;
745const T_BASE: u32 = 0x11A7;
746const L_COUNT: u32 = 19;
747const V_COUNT: u32 = 21;
748const T_COUNT: u32 = 28;
749const N_COUNT: u32 = V_COUNT * T_COUNT;
750const S_COUNT: u32 = L_COUNT * N_COUNT;
751
752pub fn compose(a: Codepoint, b: Codepoint) -> Option<Codepoint> {
753    // Hangul is handled algorithmically.
754    if let Some(ab) = compose_hangul(a, b) {
755        return Some(ab);
756    }
757
758    let u: u32;
759
760    if (a & 0xFFFF_F800) == 0x0000 && (b & 0xFFFF_FF80) == 0x0300 {
761        /* If "a" is small enough and "b" is in the U+0300 range,
762         * the composition data is encoded in a 32bit array sorted
763         * by "a,b" pair. */
764        let k = HB_CODEPOINT_ENCODE3_11_7_14(a, b, 0);
765        let v = _hb_ucd_dm2_u32_map
766            .binary_search_by(|probe| {
767                let key = probe & HB_CODEPOINT_ENCODE3_11_7_14(0x001F_FFFF, 0x001F_FFFF, 0);
768                key.cmp(&k)
769            })
770            .ok()
771            .map(|index| _hb_ucd_dm2_u32_map[index]);
772
773        if let Some(value) = v {
774            u = HB_CODEPOINT_DECODE3_11_7_14_3(value);
775        } else {
776            return None;
777        }
778    } else {
779        /* Otherwise it is stored in a 64bit array sorted by
780         * "a,b" pair. */
781        let k = HB_CODEPOINT_ENCODE3(a, b, 0);
782        let v = _hb_ucd_dm2_u64_map
783            .binary_search_by(|probe| {
784                let key = probe & HB_CODEPOINT_ENCODE3(0x001F_FFFF, 0x001F_FFFF, 0);
785                key.cmp(&k)
786            })
787            .ok()
788            .map(|index| _hb_ucd_dm2_u64_map[index]);
789
790        if let Some(value) = v {
791            u = HB_CODEPOINT_DECODE3_3(value);
792        } else {
793            return None;
794        }
795    }
796
797    if u == 0 {
798        None
799    } else {
800        Some(u)
801    }
802}
803
804fn compose_hangul(a: Codepoint, b: Codepoint) -> Option<Codepoint> {
805    let l = a;
806    let v = b;
807    if L_BASE <= l && l < (L_BASE + L_COUNT) && V_BASE <= v && v < (V_BASE + V_COUNT) {
808        let r = S_BASE + (l - L_BASE) * N_COUNT + (v - V_BASE) * T_COUNT;
809        Some(r)
810    } else if S_BASE <= l
811        && l <= (S_BASE + S_COUNT - T_COUNT)
812        && T_BASE <= v
813        && v < (T_BASE + T_COUNT)
814        && (l - S_BASE) % T_COUNT == 0
815    {
816        let r = l + (v - T_BASE);
817        Some(r)
818    } else {
819        None
820    }
821}

823pub fn decompose(ab: Codepoint) -> Option<(Codepoint, Codepoint)> {
824    if let Some((a, b)) = decompose_hangul(ab) {
825        return Some((a, b));
826    }
827
828    let mut i = _hb_ucd_dm(ab as usize) as usize;
829
830    // If no data, there's no decomposition.
831    if i == 0 {
832        return None;
833    }
834    i -= 1;
835
836    if i < _hb_ucd_dm1_p0_map.len() + _hb_ucd_dm1_p2_map.len() {
837        let a = if i < _hb_ucd_dm1_p0_map.len() {
838            _hb_ucd_dm1_p0_map[i] as u32
839        } else {
840            let j = i - _hb_ucd_dm1_p0_map.len();
841            0x20000 | _hb_ucd_dm1_p2_map[j] as u32
842        };
843        return Some((a, 0));
844    }
845
846    i -= _hb_ucd_dm1_p0_map.len() + _hb_ucd_dm1_p2_map.len();
847
848    if i < _hb_ucd_dm2_u32_map.len() {
849        let v = _hb_ucd_dm2_u32_map[i];
850        let a = HB_CODEPOINT_DECODE3_11_7_14_1(v);
851        let b = HB_CODEPOINT_DECODE3_11_7_14_2(v);
852        return Some((a, b));
853    }
854
855    i -= _hb_ucd_dm2_u32_map.len();
856
857    let v = _hb_ucd_dm2_u64_map[i];
858    let a = HB_CODEPOINT_DECODE3_1(v);
859    let b = HB_CODEPOINT_DECODE3_2(v);
860    Some((a, b))
861}
862
863pub fn decompose_hangul(ab: Codepoint) -> Option<(Codepoint, Codepoint)> {
864    let si = ab.wrapping_sub(S_BASE);
865    if si >= S_COUNT {
866        return None;
867    }
868
869    let (a, b) = if si % T_COUNT != 0 {
870        // LV,T
871        (S_BASE + (si / T_COUNT) * T_COUNT, T_BASE + (si % T_COUNT))
872    } else {
873        // L,V
874        (L_BASE + (si / N_COUNT), V_BASE + (si % N_COUNT) / T_COUNT)
875    };
876    Some((a, b))
877}
