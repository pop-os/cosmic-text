13pub struct Charmap<'a> {
14    subtable: Option<(SelectedCmapSubtable, CmapSubtable<'a>)>,
15    vs_subtable: Option<Cmap14<'a>>,
16    cache: &'a cache_t,
17}
18
19impl<'a> Charmap<'a> {
20    pub fn new(font: &FontRef<'a>, table_offsets: &TableOffsets, cache: &'a cache_t) -> Self {
21        if let Some(cmap) = table_offsets.cmap.resolve_table::<Cmap>(font) {
22            let data = cmap.offset_data();
23            let records = cmap.encoding_records();
24            let subtable = table_offsets
25                .cmap_subtable
26                .and_then(|s| Some((s, records.get(s.index as usize)?.subtable(data).ok()?)));
27            let vs_subtable = table_offsets
28                .cmap_vs_subtable
29                .and_then(|index| records.get(index as usize))
30                .and_then(|rec| rec.subtable(data).ok())
31                .and_then(|subtable| match subtable {
32                    CmapSubtable::Format14(table) => Some(table),
33                    _ => None,
34                });
35            Self {
36                subtable,
37                vs_subtable,
38                cache,
39            }
40        } else {
41            Self {
42                subtable: None,
43                vs_subtable: None,
44                cache,
45            }
46        }
47    }
48
49    fn map_impl(&self, mut c: u32) -> Option<GlyphId> {
50        let subtable = self.subtable.as_ref()?;
51        if subtable.0.is_mac_roman && c > 0x7F {
52            c = unicode_to_macroman(c);
53        }
54        let result = match &subtable.1 {
55            CmapSubtable::Format0(table) => table.map_codepoint(c),
56            CmapSubtable::Format6(table) => table.map_codepoint(c),
57            CmapSubtable::Format10(table) => {
58                if let Some(index) = c.checked_sub(table.start_char_code()) {
59                    if index < table.num_chars() {
60                        table
61                            .glyph_id_array()
62                            .get(index as usize)
63                            .map(|gid| GlyphId::from(gid.get()))
64                    } else {
65                        None
66                    }
67                } else {
68                    None
69                }
70            }
71            CmapSubtable::Format4(table) => table.map_codepoint(c),
72            CmapSubtable::Format12(table) => table.map_codepoint(c),
73            CmapSubtable::Format13(table) => table.map_codepoint(c),
74            _ => None,
75        };
76        if result.is_none() && subtable.0.is_symbol && c <= 0x00FF {
77            // For symbol-encoded OpenType fonts, we duplicate the
78            // U+F000..F0FF range at U+0000..U+00FF.  That's what
79            // Windows seems to do, and that's hinted about at:
80            // https://docs.microsoft.com/en-us/typography/opentype/spec/recom
81            // under "Non-Standard (Symbol) Fonts".
82            return self.map(0xF000 + c);
83        }
84        result
85    }
86
87    #[inline(always)]
88    pub fn map(&self, c: u32) -> Option<GlyphId> {
89        if let Some(gid) = self.cache.get(c) {
90            return Some(GlyphId::new(gid));
91        }
92        let gid = self.map_impl(c);
93        if let Some(gid) = gid {
94            self.cache.set(c, gid.to_u32());
95        }
96        gid
97    }
98
99    pub fn map_variant(&self, c: u32, vs: u32) -> Option<GlyphId> {
100        let subtable = self.vs_subtable.as_ref()?;
101        match subtable.map_variant(c, vs)? {
102            MapVariant::UseDefault => self.map(c),
103            MapVariant::Variant(gid) => Some(gid),
104        }
105    }
106}

128fn unicode_to_macroman(c: u32) -> u32 {
129    let u = c as u16;
130    let Some(index) = UNICODE_TO_MACROMAN.iter().position(|m| *m == u) else {
131        return 0;
132    };
133    (0x80 + index) as u32
