10trait SmallVecExt {
11    fn left(&self) -> usize;
12    fn is_full(&self) -> bool;
13}

15impl<A: smallvec::Array> SmallVecExt for SmallVec<A> {
16    fn left(&self) -> usize {
17        self.inline_size() - self.len()
18    }
19
20    fn is_full(&self) -> bool {
21        self.len() == self.inline_size()
22    }
23}

25/// Converts an `Script` and an `Language` to script and language tags.
26pub fn tags_from_script_and_language(
27    script: Option<Script>,
28    language: Option<&Language>,
29) -> (ThreeTags, ThreeTags) {
30    let mut needs_script = true;
31    let mut scripts = SmallVec::new();
32    let mut languages = SmallVec::new();
33
34    let mut private_use_subtag = None;
35    let mut prefix = "";
36    if let Some(language) = language {
37        let language = language.as_str();
38        if language.starts_with("x-") {
39            private_use_subtag = Some(language);
40        } else {
41            let bytes = language.as_bytes();
42            let mut i = 1;
43            while i < bytes.len() {
44                if bytes.get(i - 1) == Some(&b'-') && bytes.get(i + 1) == Some(&b'-') {
45                    if bytes[i] == b'x' {
46                        private_use_subtag = Some(&language[i..]);
47                        if prefix.is_empty() {
48                            prefix = &language[..i - 1];
49                        }
50
51                        break;
52                    } else {
53                        prefix = &language[..i - 1];
54                    }
55                }
56
57                i += 1;
58            }
59
60            if prefix.is_empty() {
61                prefix = &language[..i];
62            }
63        }
64
65        needs_script = !parse_private_use_subtag(
66            private_use_subtag,
67            "-hbsc",
68            u8::to_ascii_lowercase,
69            &mut scripts,
70        );
71
72        let needs_language = !parse_private_use_subtag(
73            private_use_subtag,
74            "-hbot",
75            u8::to_ascii_uppercase,
76            &mut languages,
77        );
78
79        if needs_language {
80            if let Ok(prefix) = Language::from_str(prefix) {
81                tags_from_language(&prefix, &mut languages);
82            }
83        }
84    }
85
86    if needs_script {
87        all_tags_from_script(script, &mut scripts);
88    }
89
90    (scripts, languages)
91}

93fn parse_private_use_subtag(
94    private_use_subtag: Option<&str>,
95    prefix: &str,
96    normalize: fn(&u8) -> u8,
97    tags: &mut ThreeTags,
98) -> bool {
99    let Some(private_use_subtag) = private_use_subtag else {
100        return false;
101    };
102
103    let private_use_subtag = match private_use_subtag.find(prefix) {
104        Some(idx) => &private_use_subtag[idx + prefix.len()..],
105        None => return false,
106    };
107
108    let mut tag = SmallVec::<[u8; 4]>::new();
109    for c in private_use_subtag.bytes().take(4) {
110        if c.is_ascii_alphanumeric() {
111            tag.push((normalize)(&c));
112        } else {
113            break;
114        }
115    }
116
117    if tag.is_empty() {
118        return false;
119    }
120
121    let mut tag = hb_tag_t::from_bytes_lossy(tag.as_slice());
122
123    // Some bits magic from HarfBuzz...
124    if tag.as_u32() & 0xDFDF_DFDF == hb_tag_t::default_script().as_u32() {
125        tag = hb_tag_t::from_u32(tag.as_u32() ^ !0xDFDF_DFDF);
126    }
127
128    tags.push(tag);
129
130    true
131}

133fn lang_cmp(s1: &str, s2: &str) -> core::cmp::Ordering {
134    let da = s1.find('-').unwrap_or(s1.len());
135    let db = s2.find('-').unwrap_or(s2.len());
136    let n = core::cmp::max(da, db);
137    let ea = core::cmp::min(n, s1.len());
138    let eb = core::cmp::min(n, s2.len());
139    s1[..ea].cmp(&s2[..eb])
140}

142fn tags_from_language(language: &Language, tags: &mut ThreeTags) {
143    let language = language.as_str();
144
145    // Check for matches of multiple subtags.
146    if tag_table::tags_from_complex_language(language, tags) {
147        return;
148    }
149
150    let mut sublang = language;
151
152    // Find a language matching in the first component.
153    if let Some(i) = language.find('-') {
154        // If there is an extended language tag, use it.
155        if language.len() >= 6 {
156            let extlang = match language[i + 1..].find('-') {
157                Some(idx) => idx == 3,
158                None => language.len() - i - 1 == 3,
159            };
160
161            if extlang && language.as_bytes()[i + 1].is_ascii_alphabetic() {
162                sublang = &language[i + 1..];
163            }
164        }
165    }
166
167    use tag_table::OPEN_TYPE_LANGUAGES as LANGUAGES;
168
169    if let Ok(mut idx) = LANGUAGES.binary_search_by(|v| lang_cmp(v.language, sublang)) {
170        while idx != 0 && LANGUAGES[idx].language == LANGUAGES[idx - 1].language {
171            idx -= 1;
172        }
173
174        let len = core::cmp::min(tags.left(), LANGUAGES.len() - idx - 1);
175        for i in 0..len {
176            if LANGUAGES[idx + i].language != LANGUAGES[idx].language {
177                break;
178            }
179
180            if LANGUAGES[idx + i].tag.is_null() {
181                break;
182            }
183
184            if tags.is_full() {
185                break;
186            }
187
188            tags.push(LANGUAGES[idx + i].tag);
189        }
190
191        return;
192    }
193
194    if language.len() == 3 {
195        tags.push(hb_tag_t::from_bytes_lossy(language.as_bytes()).to_uppercase());
196    }
197}

199fn all_tags_from_script(script: Option<Script>, tags: &mut ThreeTags) {
200    if let Some(script) = script {
201        if let Some(tag) = new_tag_from_script(script) {
202            // Script::Myanmar maps to 'mym2', but there is no 'mym3'.
203            if tag != hb_tag_t::new(b"mym2") {
204                let mut tag3 = tag.to_be_bytes();
205                tag3[3] = b'3';
206                tags.push(hb_tag_t::new(&tag3));
207            }
208
209            if !tags.is_full() {
210                tags.push(tag);
211            }
212        }
213
214        if !tags.is_full() {
215            tags.push(old_tag_from_script(script));
216        }
217    }
218}

220fn new_tag_from_script(script: Script) -> Option<hb_tag_t> {
221    match script {
222        script::BENGALI => Some(hb_tag_t::new(b"bng2")),
223        script::DEVANAGARI => Some(hb_tag_t::new(b"dev2")),
224        script::GUJARATI => Some(hb_tag_t::new(b"gjr2")),
225        script::GURMUKHI => Some(hb_tag_t::new(b"gur2")),
226        script::KANNADA => Some(hb_tag_t::new(b"knd2")),
227        script::MALAYALAM => Some(hb_tag_t::new(b"mlm2")),
228        script::ORIYA => Some(hb_tag_t::new(b"ory2")),
229        script::TAMIL => Some(hb_tag_t::new(b"tml2")),
230        script::TELUGU => Some(hb_tag_t::new(b"tel2")),
231        script::MYANMAR => Some(hb_tag_t::new(b"mym2")),
232        _ => None,
233    }
234}

236fn old_tag_from_script(script: Script) -> hb_tag_t {
237    // This seems to be accurate as of end of 2012.
238    match script {
239        script::MATH => hb_tag_t::new(b"math"),
240
241        // Katakana and Hiragana both map to 'kana'.
242        script::HIRAGANA => hb_tag_t::new(b"kana"),
243
244        // Spaces at the end are preserved, unlike ISO 15924.
245        script::LAO => hb_tag_t::new(b"lao "),
246        script::YI => hb_tag_t::new(b"yi  "),
247        // Unicode-5.0 additions.
248        script::NKO => hb_tag_t::new(b"nko "),
249        // Unicode-5.1 additions.
250        script::VAI => hb_tag_t::new(b"vai "),
251
252        // Else, just change first char to lowercase and return.
253        _ => hb_tag_t::from_u32(script.tag().as_u32() | 0x2000_0000),
254    }
255}

282    fn old_tag_to_script(tag: hb_tag_t) -> Option<Script> {
283        if tag == hb_tag_t::default_script() {
284            return None;
285        }
286
287        let mut bytes = tag.to_be_bytes();
288
289        // This side of the conversion is fully algorithmic.
290
291        // Any spaces at the end of the tag are replaced by repeating the last
292        // letter.  Eg 'nko ' -> 'Nkoo'
293        if bytes[2] == b' ' {
294            bytes[2] = bytes[1];
295        }
296        if bytes[3] == b' ' {
297            bytes[3] = bytes[2];
298        }
299
300        // Change first char to uppercase.
301        bytes[0] = bytes[0].to_ascii_uppercase();
302
303        Some(Script(hb_tag_t::new(&bytes)))
304    }
305
306    fn tag_to_script(tag: hb_tag_t) -> Option<Script> {
307        let bytes = tag.to_be_bytes();
308        if bytes[3] == b'2' || bytes[3] == b'3' {
309            let mut tag2 = bytes;
310            tag2[3] = b'2';
311            return new_tag_to_script(hb_tag_t::new(&tag2));
312        }
313
314        old_tag_to_script(tag)
315    }

628    macro_rules! test_tags {
629        ($name:ident, $script:expr, $lang:expr, $scripts:expr, $langs:expr) => {
630            #[test]
631            fn $name() {
632                let (scripts, languages) = tags_from_script_and_language(
633                    $script, Language::from_str($lang).ok().as_ref(),
634                );
635
636                let exp_scripts: Vec<hb_tag_t> = $scripts.iter().map(|v| hb_tag_t::from_bytes_lossy(*v)).collect();
637                let exp_langs: Vec<hb_tag_t> = $langs.iter().map(|v| hb_tag_t::from_bytes_lossy(*v)).collect();
638
639                assert_eq!(exp_scripts, scripts.as_slice());
640                assert_eq!(exp_langs, languages.as_slice());
641            }
642        };
643    }
