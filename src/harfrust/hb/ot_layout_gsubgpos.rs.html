29pub fn match_input(
30    ctx: &mut hb_ot_apply_context_t,
31    input_len: u16,
32    match_func: impl Fn(&mut GlyphInfo, u16) -> bool,
33    end_position: &mut usize,
34    p_total_component_count: Option<&mut u8>,
35) -> bool {
36    // This is perhaps the trickiest part of OpenType...  Remarks:
37    //
38    // - If all components of the ligature were marks, we call this a mark ligature.
39    //
40    // - If there is no GDEF, and the ligature is NOT a mark ligature, we categorize
41    //   it as a ligature glyph.
42    //
43    // - Ligatures cannot be formed across glyphs attached to different components
44    //   of previous ligatures.  Eg. the sequence is LAM,SHADDA,LAM,FATHA,HEH, and
45    //   LAM,LAM,HEH form a ligature, leaving SHADDA,FATHA next to eachother.
46    //   However, it would be wrong to ligate that SHADDA,FATHA sequence.
47    //   There are a couple of exceptions to this:
48    //
49    //   o If a ligature tries ligating with marks that belong to it itself, go ahead,
50    //     assuming that the font designer knows what they are doing (otherwise it can
51    //     break Indic stuff when a matra wants to ligate with a conjunct,
52    //
53    //   o If two marks want to ligate and they belong to different components of the
54    //     same ligature glyph, and said ligature glyph is to be ignored according to
55    //     mark-filtering rules, then allow.
56    //     https://github.com/harfbuzz/harfbuzz/issues/545
57
58    #[derive(PartialEq)]
59    enum Ligbase {
60        NotChecked,
61        MayNotSkip,
62        MaySkip,
63    }
64
65    let count = usize::from(input_len) + 1;
66    if count > MAX_CONTEXT_LENGTH {
67        return false;
68    }
69    ctx.match_positions_len = count;
70
71    let mut iter = skipping_iterator_t::with_match_fn(ctx, false, Some(match_func));
72    iter.reset(iter.buffer.idx);
73    iter.set_glyph_data(0);
74
75    let first = *iter.buffer.cur(0);
76    let first_lig_id = first.lig_id();
77    let first_lig_comp = first.lig_comp();
78    let mut total_component_count = 0;
79    let mut ligbase = Ligbase::NotChecked;
80
81    for i in 1..count {
82        let mut unsafe_to = 0;
83        if !iter.next(Some(&mut unsafe_to)) {
84            *end_position = unsafe_to;
85            return false;
86        }
87
88        iter.set_match_position(i, iter.index());
89
90        let this = iter.buffer.info[iter.index()];
91        let this_lig_id = this.lig_id();
92        let this_lig_comp = this.lig_comp();
93
94        if first_lig_id != 0 && first_lig_comp != 0 {
95            // If first component was attached to a previous ligature component,
96            // all subsequent components should be attached to the same ligature
97            // component, otherwise we shouldn't ligate them...
98            if first_lig_id != this_lig_id || first_lig_comp != this_lig_comp {
99                // ...unless, we are attached to a base ligature and that base
100                // ligature is ignorable.
101                if ligbase == Ligbase::NotChecked {
102                    let out = iter.buffer.out_info();
103                    let mut j = iter.buffer.out_len;
104                    let mut found = false;
105                    while j > 0 && out[j - 1].lig_id() == first_lig_id {
106                        if out[j - 1].lig_comp() == 0 {
107                            j -= 1;
108                            found = true;
109                            break;
110                        }
111                        j -= 1;
112                    }
113
114                    ligbase = if found && iter.may_skip(&out[j]) == may_skip_t::SKIP_YES {
115                        Ligbase::MaySkip
116                    } else {
117                        Ligbase::MayNotSkip
118                    };
119                }
120
121                if ligbase == Ligbase::MayNotSkip {
122                    return false;
123                }
124            }
125        } else {
126            // If first component was NOT attached to a previous ligature component,
127            // all subsequent components should also NOT be attached to any ligature
128            // component, unless they are attached to the first component itself!
129            if this_lig_id != 0 && this_lig_comp != 0 && (this_lig_id != first_lig_id) {
130                return false;
131            }
132        }
133
134        total_component_count += this.lig_num_comps();
135    }
136
137    *end_position = iter.index() + 1;
138
139    if let Some(p_total_component_count) = p_total_component_count {
140        total_component_count += first.lig_num_comps();
141        *p_total_component_count = total_component_count;
142    }
143
144    ctx.match_positions[0] = iter.buffer.idx as u32;
145
146    true
147}

149pub fn match_backtrack(
150    ctx: &mut hb_ot_apply_context_t,
151    backtrack_len: u16,
152    match_func: impl Fn(&mut GlyphInfo, u16) -> bool,
153    match_start: &mut usize,
154) -> bool {
155    let mut iter = skipping_iterator_t::with_match_fn(ctx, true, Some(match_func));
156    iter.reset(iter.buffer.backtrack_len());
157    iter.set_glyph_data(0);
158
159    for _ in 0..backtrack_len {
160        let mut unsafe_from = 0;
161        if !iter.prev(Some(&mut unsafe_from)) {
162            *match_start = unsafe_from;
163            return false;
164        }
165    }
166
167    *match_start = iter.index();
168    true
169}

171pub fn match_lookahead(
172    ctx: &mut hb_ot_apply_context_t,
173    lookahead_len: u16,
174    match_func: impl Fn(&mut GlyphInfo, u16) -> bool,
175    start_index: usize,
176    end_index: &mut usize,
177) -> bool {
178    // Function should always be called with a non-zero starting index
179    // c.f. https://github.com/harfbuzz/rustybuzz/issues/142
180    debug_assert!(start_index >= 1);
181    let mut iter = skipping_iterator_t::with_match_fn(ctx, true, Some(match_func));
182    iter.reset(start_index - 1);
183    iter.set_glyph_data(0);
184
185    for _ in 0..lookahead_len {
186        let mut unsafe_to = 0;
187        if !iter.next(Some(&mut unsafe_to)) {
188            *end_index = unsafe_to;
189            return false;
190        }
191    }
192
193    *end_index = iter.index() + 1;
194    true
195}

218#[derive(Default)]
219pub struct matcher_t {
220    lookup_props: u32,
221    mask: hb_mask_t,
222    ignore_zwnj: bool,
223    ignore_zwj: bool,
224    ignore_hidden: bool,
225    per_syllable: bool,
226    cached_props: u32, // Cached glyph properties for the current lookup.
227}
228
229impl matcher_t {
230    fn new(ctx: &hb_ot_apply_context_t, context_match: bool) -> Self {
231        matcher_t {
232            lookup_props: ctx.lookup_props,
233            // Ignore ZWNJ if we are matching GPOS, or matching GSUB context and asked to.
234            ignore_zwnj: ctx.table_index == TableIndex::GPOS || (context_match && ctx.auto_zwnj),
235            // Ignore ZWJ if we are matching context, or asked to.
236            ignore_zwj: context_match || ctx.auto_zwj,
237            // Ignore hidden glyphs (like CGJ) during GPOS.
238            ignore_hidden: ctx.table_index == TableIndex::GPOS,
239            mask: if context_match {
240                u32::MAX
241            } else {
242                ctx.lookup_mask()
243            },
244            /* Per syllable matching is only for GSUB. */
245            per_syllable: ctx.table_index == TableIndex::GSUB && ctx.per_syllable,
246            cached_props: ctx.cached_props,
247        }
248    }
249
250    fn may_match(
251        &self,
252        info: &mut GlyphInfo,
253        glyph_data: u16,
254        match_func: Option<&impl Fn(&mut GlyphInfo, u16) -> bool>,
255        syllable: u8,
256    ) -> may_match_t {
257        if (info.mask & self.mask) == 0
258            || (self.per_syllable && syllable != 0 && syllable != info.syllable())
259        {
260            return may_match_t::MATCH_NO;
261        }
262
263        if let Some(match_func) = match_func {
264            return if match_func(info, glyph_data) {
265                may_match_t::MATCH_YES
266            } else {
267                may_match_t::MATCH_NO
268            };
269        }
270
271        may_match_t::MATCH_MAYBE
272    }
273
274    #[inline(always)]
275    fn may_skip(&self, info: &GlyphInfo, face: &hb_font_t, lookup_props: u32) -> may_skip_t {
276        if !check_glyph_property(face, info, lookup_props, self.cached_props) {
277            return may_skip_t::SKIP_YES;
278        }
279
280        if info.is_default_ignorable()
281            && (self.ignore_zwnj || !info.is_zwnj())
282            && (self.ignore_zwj || !info.is_zwj())
283            && (self.ignore_hidden || !info.is_hidden())
284        {
285            return may_skip_t::SKIP_MAYBE;
286        }
287
288        may_skip_t::SKIP_NO
289    }
290}

292// In harfbuzz, skipping iterator works quite differently than it works here. In harfbuzz,
293// hb_ot_apply_context contains a skipping iterator that itself contains references to font
294// and buffer, meaning that we multiple borrows issue. Due to ownership rules in Rust,
295// we cannot copy this approach. Because of this, we basically create a new skipping iterator
296// when needed, and we do not have `init` method that exist in harfbuzz. This has a performance
297// cost, and makes backporting related changes very hard, but it seems unavoidable, unfortunately.
298pub struct skipping_iterator_t<'f, 'c, F> {
299    pub(crate) buffer: &'c mut hb_buffer_t,
300    face: &'c hb_font_t<'f>,
301    matcher: &'c matcher_t,
302    match_positions: &'c mut MatchPositions,
303    buf_len: usize,
304    glyph_data: u16,
305    pub(crate) buf_idx: usize,
306    match_func: Option<F>,
307    lookup_props: u32,
308    syllable: u8,
309}
310
311impl<'f, 'c> skipping_iterator_t<'f, 'c, fn(&mut GlyphInfo, u16) -> bool> {
312    pub fn new(ctx: &'c mut hb_ot_apply_context_t<'f>, context_match: bool) -> Self {
313        Self::with_match_fn(ctx, context_match, None)
314    }
315}

317pub(crate) enum MatchSource {
318    Info,
319    OutInfo,

321
322impl<'f, 'c, F> skipping_iterator_t<'f, 'c, F>
323where
324    F: Fn(&mut GlyphInfo, u16) -> bool,
325{
326    pub fn with_match_fn(
327        ctx: &'c mut hb_ot_apply_context_t<'f>,
328        context_match: bool,
329        match_fn: Option<F>,
330    ) -> Self {
331        let matcher = if context_match {
332            &ctx.context_matcher
333        } else {
334            &ctx.matcher
335        };
336        let buf_len = ctx.buffer.len;
337        skipping_iterator_t {
338            buffer: ctx.buffer,
339            face: ctx.face,
340            glyph_data: 0,
341            buf_len,
342            buf_idx: 0,
343            matcher,
344            match_func: match_fn,
345            match_positions: &mut ctx.match_positions,
346            lookup_props: matcher.lookup_props,
347            syllable: 0,
348        }
349    }
350
351    pub fn set_match_position(&mut self, idx: usize, position: usize) {
352        let count = idx + 1;
353        if count > self.match_positions.len() {
354            self.match_positions.resize(count, 0);
355        }
356
357        self.match_positions[idx] = position as u32;
358    }
359
360    pub fn set_glyph_data(&mut self, glyph_data: u16) {
361        self.glyph_data = glyph_data;
362    }
363
364    fn advance_glyph_data(&mut self) {
365        self.glyph_data += 1;
366    }
367
368    pub fn set_lookup_props(&mut self, lookup_props: u32) {
369        self.lookup_props = lookup_props;
370    }
371
372    pub fn index(&self) -> usize {
373        self.buf_idx
374    }
375
376    #[inline]
377    pub fn next(&mut self, unsafe_to: Option<&mut usize>) -> bool {
378        let stop = self.buf_len as i32 - 1;
379
380        while (self.buf_idx as i32) < stop {
381            self.buf_idx += 1;
382
383            match self.match_at(self.buf_idx, MatchSource::Info) {
384                match_t::MATCH => {
385                    self.advance_glyph_data();
386                    return true;
387                }
388                match_t::NOT_MATCH => {
389                    if let Some(unsafe_to) = unsafe_to {
390                        *unsafe_to = self.buf_idx + 1;
391                    }
392
393                    return false;
394                }
395                match_t::SKIP => continue,
396            }
397        }
398
399        if let Some(unsafe_to) = unsafe_to {
400            *unsafe_to = self.buf_idx + 1;
401        }
402
403        false
404    }
405
406    #[inline]
407    pub fn prev(&mut self, unsafe_from: Option<&mut usize>) -> bool {
408        let stop: usize = 0;
409
410        while self.buf_idx > stop {
411            self.buf_idx -= 1;
412
413            match self.match_at(self.buf_idx, MatchSource::OutInfo) {
414                match_t::MATCH => {
415                    self.advance_glyph_data();
416                    return true;
417                }
418                match_t::NOT_MATCH => {
419                    if let Some(unsafe_from) = unsafe_from {
420                        *unsafe_from = self.buf_idx.max(1) - 1;
421                    }
422
423                    return false;
424                }
425                match_t::SKIP => {
426                    continue;
427                }
428            }
429        }
430
431        if let Some(unsafe_from) = unsafe_from {
432            *unsafe_from = 0;
433        }
434
435        false
436    }
437
438    pub fn reset(&mut self, start_index: usize) {
439        self.buf_idx = start_index;
440        self.buf_len = self.buffer.len;
441        self.syllable = if self.buf_idx == self.buffer.idx {
442            self.buffer.cur(0).syllable()
443        } else {
444            0
445        };
446    }
447
448    pub fn reset_fast(&mut self, start_index: usize) {
449        // Doesn't set end or syllable. Used by GPOS which doesn't care / change.
450        self.buf_idx = start_index;
451    }
452
453    pub fn may_skip(&self, info: &GlyphInfo) -> may_skip_t {
454        self.matcher.may_skip(info, self.face, self.lookup_props)
455    }
456
457    #[inline]
458    pub fn match_at(&mut self, idx: usize, source: MatchSource) -> match_t {
459        let info = match source {
460            MatchSource::Info => &mut self.buffer.info[idx],
461            MatchSource::OutInfo => &mut self.buffer.out_info_mut()[idx],
462        };
463        let skip = self.matcher.may_skip(info, self.face, self.lookup_props);
464
465        if skip == may_skip_t::SKIP_YES {
466            return match_t::SKIP;
467        }
468
469        let _match = self.matcher.may_match(
470            info,
471            self.glyph_data,
472            self.match_func.as_ref(),
473            self.syllable,
474        );
475
476        if _match == may_match_t::MATCH_YES
477            || (_match == may_match_t::MATCH_MAYBE && skip == may_skip_t::SKIP_NO)
478        {
479            return match_t::MATCH;
480        }
481
482        if skip == may_skip_t::SKIP_NO {
483            return match_t::NOT_MATCH;
484        }
485
486        match_t::SKIP
487    }
488}

490pub(crate) fn apply_lookup(
491    ctx: &mut hb_ot_apply_context_t,
492    input_len: usize,
493    match_end: usize,
494    lookups: &[SequenceLookupRecord],
495) {
496    let mut count = input_len + 1;
497
498    debug_assert!(count <= ctx.match_positions.len(), "");
499
500    // All positions are distance from beginning of *output* buffer.
501    // Adjust.
502    let mut end: isize = {
503        let backtrack_len = ctx.buffer.backtrack_len();
504        let delta = backtrack_len as isize - ctx.buffer.idx as isize;
505
506        // Convert positions to new indexing.
507        for j in 0..count {
508            ctx.match_positions[j] = (ctx.match_positions[j] as isize + delta) as _;
509        }
510
511        backtrack_len as isize + match_end as isize - ctx.buffer.idx as isize
512    };
513
514    for record in lookups {
515        if !ctx.buffer.successful {
516            break;
517        }
518
519        let idx = usize::from(record.sequence_index.get());
520        if idx >= count {
521            continue;
522        }
523
524        let orig_len = ctx.buffer.backtrack_len() + ctx.buffer.lookahead_len();
525
526        // This can happen if earlier recursed lookups deleted many entries.
527        if ctx.match_positions[idx] as usize >= orig_len {
528            continue;
529        }
530
531        if !ctx.buffer.move_to(ctx.match_positions[idx] as usize) {
532            break;
533        }
534
535        if ctx.buffer.max_ops <= 0 {
536            break;
537        }
538
539        if ctx.recurse(record.lookup_list_index.get()).is_none() {
540            continue;
541        }
542
543        let new_len = ctx.buffer.backtrack_len() + ctx.buffer.lookahead_len();
544        let mut delta = new_len as isize - orig_len as isize;
545        if delta == 0 {
546            continue;
547        }
548
549        // Recursed lookup changed buffer len.  Adjust.
550        //
551        // TODO:
552        //
553        // Right now, if buffer length increased by n, we assume n new glyphs
554        // were added right after the current position, and if buffer length
555        // was decreased by n, we assume n match positions after the current
556        // one where removed.  The former (buffer length increased) case is
557        // fine, but the decrease case can be improved in at least two ways,
558        // both of which are significant:
559        //
560        //   - If recursed-to lookup is MultipleSubst and buffer length
561        //     decreased, then it's current match position that was deleted,
562        //     NOT the one after it.
563        //
564        //   - If buffer length was decreased by n, it does not necessarily
565        //     mean that n match positions where removed, as there recursed-to
566        //     lookup might had a different LookupFlag.  Here's a constructed
567        //     case of that:
568        //     https://github.com/harfbuzz/harfbuzz/discussions/3538
569        //
570        // It should be possible to construct tests for both of these cases.
571
572        end += delta;
573        if end < ctx.match_positions[idx] as isize {
574            // End might end up being smaller than match_positions[idx] if the recursed
575            // lookup ended up removing many items.
576            // Just never rewind end beyond start of current position, since that is
577            // not possible in the recursed lookup.  Also adjust delta as such.
578            //
579            // https://bugs.chromium.org/p/chromium/issues/detail?id=659496
580            // https://github.com/harfbuzz/harfbuzz/issues/1611
581            //
582            delta += ctx.match_positions[idx] as isize - end;
583            end = ctx.match_positions[idx] as isize;
584        }
585
586        // next now is the position after the recursed lookup.
587        let mut next = idx + 1;
588
589        if delta > 0 {
590            if delta as usize + count > MAX_CONTEXT_LENGTH {
591                break;
592            }
593
594            if delta as usize + count > ctx.match_positions.len() {
595                ctx.match_positions.resize(delta as usize + count, 0);
596            }
597        } else {
598            // NOTE: delta is non-positive.
599            delta = delta.max(next as isize - count as isize);
600            next = (next as isize - delta) as _;
601        }
602
603        // Shift!
604        ctx.match_positions
605            .copy_within(next..count, (next as isize + delta) as _);
606        next = (next as isize + delta) as _;
607        count = (count as isize + delta) as _;
608        ctx.match_positions_len = count;
609
610        // Fill in new entries.
611        for j in idx + 1..next {
612            ctx.match_positions[j] = ctx.match_positions[j - 1] + 1;
613        }
614
615        // And fixup the rest.
616        while next < count {
617            ctx.match_positions[next] = (ctx.match_positions[next] as isize + delta) as _;
618            next += 1;
619        }
620    }
621
622    ctx.buffer.move_to(end.try_into().unwrap());
623}

727/// Apply a lookup.
728pub trait Apply {
729    fn apply(&self, ctx: &mut hb_ot_apply_context_t) -> Option<()> {
730        // Default implementation just calls `apply_with_external_cache`.
731        self.apply_with_external_cache(ctx, &SubtableExternalCache::None)
732    }
733
734    // The rest are relevant to subtables only
735
736    fn apply_with_external_cache(
737        &self,
738        ctx: &mut hb_ot_apply_context_t,
739        _external_cache: &SubtableExternalCache,
740    ) -> Option<()> {
741        // Default implementation just calls `apply`.
742        self.apply(ctx)
743    }
744
745    fn apply_cached(
746        &self,
747        ctx: &mut hb_ot_apply_context_t,
748        external_cache: &SubtableExternalCache,
749    ) -> Option<()> {
750        // Default implementation just calls `apply_with_external_cache`.
751        // This is used to apply the lookup with glyph-info caching.
752        self.apply_with_external_cache(ctx, external_cache)
753    }
754
755    fn cache_cost(&self) -> u32 {
756        // Default implementation returns 0, meaning no cache cost.
757        // This is used to determine the cost of caching the subtable.
758        0
759    }
760
761    fn external_cache_create(&self, mode: SubtableExternalCacheMode) -> SubtableExternalCache {
762        // Default implementation returns None, meaning no external cache.
763        // This is used to create an external cache for the subtable.
764        let _ = mode;
765        SubtableExternalCache::None
766    }
767}

777    fn match_properties_mark(
778        face: &hb_font_t,
779        info: &GlyphInfo,
780        glyph_props: u16,
781        match_props: u32,
782        cached_props: u32,
783    ) -> bool {
784        // If using mark filtering sets, the high short of
785        // match_props has the set index.
786        if match_props as u16 & lookup_flags::USE_MARK_FILTERING_SET != 0 {
787            if match_props == cached_props {
788                return info.matches();
789            }
790
791            let set_index = (match_props >> 16) as u16;
792            return face
793                .ot_tables
794                .is_mark_glyph(info.as_glyph().to_u32(), set_index);
795        }
796
797        // The second byte of match_props has the meaning
798        // "ignore marks of attachment type different than
799        // the attachment type specified."
800        if match_props as u16 & lookup_flags::MARK_ATTACHMENT_TYPE_MASK != 0 {
801            return (match_props as u16 & lookup_flags::MARK_ATTACHMENT_TYPE_MASK)
802                == (glyph_props & lookup_flags::MARK_ATTACHMENT_TYPE_MASK);
803        }
804
805        true
806    }
807
808    #[inline(always)]
809    pub fn check_glyph_property(
810        face: &hb_font_t,
811        info: &GlyphInfo,
812        match_props: u32,
813        cached_props: u32,
814    ) -> bool {
815        let glyph_props = info.glyph_props();
816
817        // Not covered, if, for example, glyph class is ligature and
818        // match_props includes LookupFlags::IgnoreLigatures
819        if glyph_props & match_props as u16 & lookup_flags::IGNORE_FLAGS != 0 {
820            return false;
821        }
822
823        if glyph_props & GlyphPropsFlags::MARK.bits() != 0 {
824            return match_properties_mark(face, info, glyph_props, match_props, cached_props);
825        }
826
827        true
828    }

904        pub fn recurse(&mut self, sub_lookup_index: u16) -> Option<()> {
905            if self.nesting_level_left == 0 {
906                self.buffer.successful = false;
907                return None;
908            }
909
910            self.buffer.max_ops -= 1;
911            if self.buffer.max_ops < 0 {
912                self.buffer.successful = false;
913                return None;
914            }
915
916            self.nesting_level_left -= 1;
917            let saved_props = self.lookup_props;
918            let saved_index = self.lookup_index;
919
920            self.match_positions.resize(self.match_positions_len, 0);
921            let saved_match_positions = self.match_positions.clone();
922            let saved_match_positions_len = self.match_positions_len;
923
924            self.lookup_index = sub_lookup_index;
925            let applied = self
926                .face
927                .ot_tables
928                .table_data_and_lookup(self.table_index, sub_lookup_index)
929                .and_then(|(table_data, lookup)| {
930                    self.lookup_props = lookup.props();
931                    self.update_matchers();
932                    lookup.apply(self, table_data, false)
933                });
934            self.lookup_props = saved_props;
935            self.lookup_index = saved_index;
936            self.update_matchers();
937            self.match_positions = saved_match_positions;
938            self.match_positions_len = saved_match_positions_len;
939            self.nesting_level_left += 1;
940            applied
941        }

943        fn set_glyph_class(
944            &mut self,
945            glyph_id: GlyphId,
946            class_guess: GlyphPropsFlags,
947            ligature: bool,
948            component: bool,
949        ) {
950            self.buffer.digest.add(glyph_id.into());
951
952            if let Some(syllable) = self.new_syllables {
953                self.buffer.cur_mut(0).set_syllable(syllable);
954            }
955
956            let cur = self.buffer.cur_mut(0);
957            let mut props = cur.glyph_props();
958
959            props |= GlyphPropsFlags::SUBSTITUTED.bits();
960
961            if ligature {
962                props |= GlyphPropsFlags::LIGATED.bits();
963                // In the only place that the MULTIPLIED bit is used, Uniscribe
964                // seems to only care about the "last" transformation between
965                // Ligature and Multiple substitutions.  Ie. if you ligate, expand,
966                // and ligate again, it forgives the multiplication and acts as
967                // if only ligation happened.  As such, clear MULTIPLIED bit.
968                props &= !GlyphPropsFlags::MULTIPLIED.bits();
969            }
970
971            if component {
972                props |= GlyphPropsFlags::MULTIPLIED.bits();
973            }
974
975            let has_glyph_classes = self.face.ot_tables.has_glyph_classes();
976
977            if has_glyph_classes {
978                props &= GlyphPropsFlags::PRESERVE.bits();
979                cur.set_glyph_props(props | self.face.ot_tables.glyph_props(glyph_id));
980            } else if !class_guess.is_empty() {
981                props &= GlyphPropsFlags::PRESERVE.bits();
982                cur.set_glyph_props(props | class_guess.bits());
983            } else {
984                cur.set_glyph_props(props);
985            }
986
987            if self.cached_props != u32::MAX {
988                let matches = check_glyph_property(self.face, cur, self.lookup_props, u32::MAX);
989                cur.set_matches(matches);
990            }
991        }
992
993        pub fn replace_glyph(&mut self, glyph_id: GlyphId) {
994            self.set_glyph_class(glyph_id, GlyphPropsFlags::empty(), false, false);
995            self.buffer.replace_glyph(u32::from(glyph_id));
996        }
997
998        pub fn replace_glyph_inplace(&mut self, glyph_id: GlyphId) {
999            self.set_glyph_class(glyph_id, GlyphPropsFlags::empty(), false, false);
1000            self.buffer.cur_mut(0).glyph_id = u32::from(glyph_id);
1001        }

1004            &mut self,
1005            glyph_id: GlyphId,
1006            class_guess: GlyphPropsFlags,
1007        ) {
1008            self.set_glyph_class(glyph_id, class_guess, true, false);
1009            self.buffer.replace_glyph(u32::from(glyph_id));
1010        }

1013            &mut self,

1019        }

1025pub fn ligate_input(
1026    ctx: &mut hb_ot_apply_context_t,
1027    // Including the first glyph
1028    count: usize,
1029    // Including the first glyph
1030    match_end: usize,
1031    total_component_count: u8,
1032    lig_glyph: GlyphId,
1033) {
1034    // - If a base and one or more marks ligate, consider that as a base, NOT
1035    //   ligature, such that all following marks can still attach to it.
1036    //   https://github.com/harfbuzz/harfbuzz/issues/1109
1037    //
1038    // - If all components of the ligature were marks, we call this a mark ligature.
1039    //   If it *is* a mark ligature, we don't allocate a new ligature id, and leave
1040    //   the ligature to keep its old ligature id.  This will allow it to attach to
1041    //   a base ligature in GPOS.  Eg. if the sequence is: LAM,LAM,SHADDA,FATHA,HEH,
1042    //   and LAM,LAM,HEH for a ligature, they will leave SHADDA and FATHA with a
1043    //   ligature id and component value of 2.  Then if SHADDA,FATHA form a ligature
1044    //   later, we don't want them to lose their ligature id/component, otherwise
1045    //   GPOS will fail to correctly position the mark ligature on top of the
1046    //   LAM,LAM,HEH ligature.  See:
1047    //     https://bugzilla.gnome.org/show_bug.cgi?id=676343
1048    //
1049    // - If a ligature is formed of components that some of which are also ligatures
1050    //   themselves, and those ligature components had marks attached to *their*
1051    //   components, we have to attach the marks to the new ligature component
1052    //   positions!  Now *that*'s tricky!  And these marks may be following the
1053    //   last component of the whole sequence, so we should loop forward looking
1054    //   for them and update them.
1055    //
1056    //   Eg. the sequence is LAM,LAM,SHADDA,FATHA,HEH, and the font first forms a
1057    //   'calt' ligature of LAM,HEH, leaving the SHADDA and FATHA with a ligature
1058    //   id and component == 1.  Now, during 'liga', the LAM and the LAM-HEH ligature
1059    //   form a LAM-LAM-HEH ligature.  We need to reassign the SHADDA and FATHA to
1060    //   the new ligature with a component value of 2.
1061    //
1062    //   This in fact happened to a font...  See:
1063    //   https://bugzilla.gnome.org/show_bug.cgi?id=437633
1064    //
1065
1066    let mut buffer = &mut ctx.buffer;
1067    buffer.merge_clusters(buffer.idx, match_end);
1068
1069    let mut is_base_ligature = buffer.info[ctx.match_positions[0] as usize].is_base_glyph();
1070    let mut is_mark_ligature = buffer.info[ctx.match_positions[0] as usize].is_mark();
1071    for i in 1..count {
1072        if !buffer.info[ctx.match_positions[i] as usize].is_mark() {
1073            is_base_ligature = false;
1074            is_mark_ligature = false;
1075        }
1076    }
1077
1078    let is_ligature = !is_base_ligature && !is_mark_ligature;
1079    let class = if is_ligature {
1080        GlyphPropsFlags::LIGATURE
1081    } else {
1082        GlyphPropsFlags::empty()
1083    };
1084    let lig_id = if is_ligature {
1085        buffer.allocate_lig_id()
1086    } else {
1087        0
1088    };
1089    let first = buffer.cur_mut(0);
1090    let mut last_lig_id = first.lig_id();
1091    let mut last_num_comps = first.lig_num_comps();
1092    let mut comps_so_far = last_num_comps;
1093
1094    if is_ligature {
1095        first.set_lig_props_for_ligature(lig_id, total_component_count);
1096        if first.general_category() == GeneralCategory::NON_SPACING_MARK {
1097            first.set_general_category(GeneralCategory::OTHER_LETTER);
1098        }
1099    }
1100
1101    ctx.replace_glyph_with_ligature(lig_glyph, class);
1102    buffer = &mut ctx.buffer;
1103
1104    for i in 1..count {
1105        while buffer.idx < ctx.match_positions[i] as usize && buffer.successful {
1106            if is_ligature {
1107                let cur = buffer.cur_mut(0);
1108                let mut this_comp = cur.lig_comp();
1109                if this_comp == 0 {
1110                    this_comp = last_num_comps;
1111                }
1112                // Avoid the potential for a wrap-around bug when subtracting from an unsigned integer
1113                // c.f. https://github.com/harfbuzz/rustybuzz/issues/142
1114                debug_assert!(comps_so_far >= last_num_comps);
1115                let new_lig_comp = comps_so_far - last_num_comps + this_comp.min(last_num_comps);
1116                cur.set_lig_props_for_mark(lig_id, new_lig_comp);
1117            }
1118            buffer.next_glyph();
1119        }
1120
1121        let cur = buffer.cur(0);
1122        last_lig_id = cur.lig_id();
1123        last_num_comps = cur.lig_num_comps();
1124        comps_so_far += last_num_comps;
1125
1126        // Skip the base glyph.
1127        buffer.idx += 1;
1128    }
1129
1130    if !is_mark_ligature && last_lig_id != 0 {
1131        // Re-adjust components for any marks following.
1132        for i in buffer.idx..buffer.len {
1133            let info = &mut buffer.info[i];
1134            if last_lig_id != info.lig_id() {
1135                break;
1136            }
1137
1138            let this_comp = info.lig_comp();
1139            if this_comp == 0 {
1140                break;
1141            }
1142
1143            // Avoid the potential for a wrap-around bug when subtracting from an unsigned integer
1144            // c.f. https://github.com/harfbuzz/rustybuzz/issues/142
1145            debug_assert!(comps_so_far >= last_num_comps);
1146            let new_lig_comp = comps_so_far - last_num_comps + this_comp.min(last_num_comps);
1147            info.set_lig_props_for_mark(lig_id, new_lig_comp);
1148        }
1149    }
