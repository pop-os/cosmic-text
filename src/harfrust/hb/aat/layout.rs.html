498/// HB: hb_aat_layout_substitute
499///
500/// See <https://github.com/harfbuzz/harfbuzz/blob/main/src/hb-aat-layout.cc#L285>
501#[doc(alias = "hb_aat_layout_substitute")]
502pub fn substitute(
503    plan: &hb_ot_shape_plan_t,
504    face: &hb_font_t,
505    buffer: &mut hb_buffer_t,
506    features: &[Feature],
507) {
508    let mut aat_map = map::AatMap::default();
509    if !features.is_empty() {
510        let mut builder = map::AatMapBuilder::default();
511        for feature in features {
512            builder.add_feature(face, feature);
513        }
514        builder.compile(face, &mut aat_map);
515    }
516
517    let mut c = AatApplyContext::new(plan, face, buffer);
518    layout_morx_table::apply(
519        &mut c,
520        if features.is_empty() {
521            &plan.aat_map
522        } else {
523            &aat_map
524        },
525    );
526}
