22pub fn compile_flags(face: &hb_font_t, builder: &AatMapBuilder, map: &mut AatMap) -> Option<()> {
23    let has_feature = |kind: u16, setting: u16| {
24        builder
25            .current_features
26            .binary_search_by(|probe| {
27                if probe.kind != kind {
28                    probe.kind.cmp(&kind)
29                } else {
30                    probe.setting.cmp(&setting)
31                }
32            })
33            .is_ok()
34    };
35
36    let chains = face.aat_tables.morx.as_ref()?.0.chains();
37    let chain_len = chains.iter().count();
38    map.chain_flags.resize(chain_len, vec![]);
39
40    for (chain, chain_flags) in chains.iter().zip(map.chain_flags.iter_mut()) {
41        let Ok(chain) = chain else {
42            continue;
43        };
44        let mut flags = chain.default_flags();
45        for feature in chain.features() {
46            // Check whether this type/setting pair was requested in the map,
47            // and if so, apply its flags.
48
49            if has_feature(feature.feature_type(), feature.feature_settings()) {
50                flags &= feature.disable_flags();
51                flags |= feature.enable_flags();
52            } else if feature.feature_type() == FEATURE_TYPE_LETTER_CASE as u16
53                && feature.feature_settings() == u16::from(FEATURE_SELECTOR_SMALL_CAPS)
54            {
55                // Deprecated. https://github.com/harfbuzz/harfbuzz/issues/1342
56                let ok = has_feature(
57                    FEATURE_TYPE_LOWER_CASE as u16,
58                    u16::from(FEATURE_SELECTOR_LOWER_CASE_SMALL_CAPS),
59                );
60                if ok {
61                    flags &= feature.disable_flags();
62                    flags |= feature.enable_flags();
63                }
64            }
65            // TODO: Port the following commit: https://github.com/harfbuzz/harfbuzz/commit/2124ad890
66        }
67
68        chain_flags.push(RangeFlags {
69            flags,
70            cluster_first: builder.range_first as u32,
71            cluster_last: builder.range_last as u32,
72        });
73    }
74
75    Some(())
76}

78// Chain::apply in harfbuzz
79pub fn apply<'a>(c: &mut AatApplyContext<'a>, map: &'a AatMap) -> Option<()> {
80    c.buffer.unsafe_to_concat(None, None);
81
82    c.setup_buffer_glyph_set();
83
84    let (morx, subtable_caches) = c.face.aat_tables.morx.as_ref()?;
85
86    let chains = morx.chains();
87
88    let mut subtable_idx = 0;
89
90    let mut buffer_is_reversed = false;
91    'outer: for (chain, chain_flags) in chains.iter().zip(map.chain_flags.iter()) {
92        let Ok(chain) = chain else {
93            continue;
94        };
95        c.range_flags = Some(chain_flags.as_slice());
96        for subtable in chain.subtables().iter() {
97            let Ok(subtable) = subtable else {
98                continue;
99            };
100
101            let subtable_cache = subtable_caches.get(subtable_idx);
102            let Some(subtable_cache) = subtable_cache.as_ref() else {
103                break 'outer;
104            };
105            subtable_idx += 1;
106
107            if let Some(range_flags) = c.range_flags.as_ref() {
108                if range_flags.len() == 1
109                    && (subtable.sub_feature_flags() & range_flags[0].flags == 0)
110                {
111                    continue;
112                }
113            }
114
115            if !subtable.is_all_directions()
116                && c.buffer.direction.is_vertical() != subtable.is_vertical()
117            {
118                continue;
119            }
120
121            c.subtable_flags = subtable.sub_feature_flags();
122            c.first_set = Some(&subtable_cache.glyph_set);
123            c.machine_class_cache = Some(&subtable_cache.class_cache);
124            c.start_end_safe_to_break = subtable_cache.start_end_safe_to_break;
125
126            if !c.buffer_intersects_machine() {
127                continue;
128            }
129
130            // Buffer contents is always in logical direction.  Determine if
131            // we need to reverse before applying this subtable.  We reverse
132            // back after if we did reverse indeed.
133            //
134            // Quoting the spec:
135            // """
136            // Bits 28 and 30 of the coverage field control the order in which
137            // glyphs are processed when the subtable is run by the layout engine.
138            // Bit 28 is used to indicate if the glyph processing direction is
139            // the same as logical order or layout order. Bit 30 is used to
140            // indicate whether glyphs are processed forwards or backwards within
141            // that order.
142            //
143            // Bit 30   Bit 28   Interpretation for Horizontal Text
144            //      0        0   The subtable is processed in layout order
145            //                   (the same order as the glyphs, which is
146            //                   always left-to-right).
147            //      1        0   The subtable is processed in reverse layout order
148            //                   (the order opposite that of the glyphs, which is
149            //                   always right-to-left).
150            //      0        1   The subtable is processed in logical order
151            //                   (the same order as the characters, which may be
152            //                   left-to-right or right-to-left).
153            //      1        1   The subtable is processed in reverse logical order
154            //                   (the order opposite that of the characters, which
155            //                   may be right-to-left or left-to-right).
156
157            let reverse = if subtable.is_logical() {
158                subtable.is_backwards()
159            } else {
160                subtable.is_backwards() != c.buffer.direction.is_backward()
161            };
162
163            if reverse != buffer_is_reversed {
164                c.buffer.reverse();
165                buffer_is_reversed = reverse;
166            }
167
168            if let Ok(kind) = subtable.kind() {
169                apply_subtable(kind, c);
170            }
171        }
172        if buffer_is_reversed {
173            c.buffer.reverse();
174        }
175    }
176
177    Some(())
178}

180fn collect_initial_glyphs<T, Ctx: DriverContext<T>>(
181    machine: &ExtendedStateTable<T>,
182    glyphs: &mut U32Set,
183    num_glyphs: u32,
184) where
185    T: FixedSize + bytemuck::AnyBitPattern,
186{
187    let mut classes = U32Set::default();
188
189    let class_table = &machine.class_table;
190    for i in 0..machine.n_classes {
191        if let Ok(entry) = machine.entry(START_OF_TEXT, i as u16) {
192            if entry.new_state == START_OF_TEXT
193                && !Ctx::is_action_initiable(&entry)
194                && !Ctx::is_actionable(&entry)
195            {
196                continue;
197            }
198            classes.insert(i as u32);
199        }
200    }
201
202    // And glyphs in those classes.
203
204    let filter = |class: u16| classes.contains(class as u32);
205
206    if filter(aat::class::DELETED_GLYPH as u16) {
207        glyphs.insert(DELETED_GLYPH);
208    }
209
210    class_table.collect_glyphs_filtered(glyphs, num_glyphs, filter);
211}

213fn collect_start_end_safe_to_break<T, Ctx: DriverContext<T>>(machine: &ExtendedStateTable<T>) -> u64
214where
215    T: FixedSize + bytemuck::AnyBitPattern,
216{
217    let mut result = 0u64;
218    for state in 0..64 {
219        let bit = if let Ok(entry) = machine.entry(state, aat::class::END_OF_TEXT as u16) {
220            !Ctx::is_actionable(&entry)
221        } else {
222            true
223        };
224        if bit {
225            result |= 1 << state;
226        }
227    }
228    result
229}

234    fn is_action_initiable(entry: &StateEntry<T>) -> bool;

239fn drive<T: bytemuck::AnyBitPattern + FixedSize + core::fmt::Debug, Ctx: DriverContext<T>>(
240    machine: &ExtendedStateTable<'_, T>,
241    c: &mut Ctx,
242    ac: &mut AatApplyContext,
243) {
244    if !Ctx::in_place() {
245        ac.buffer.clear_output();
246    }
247
248    let mut state = START_OF_TEXT;
249    let mut last_range = ac.range_flags.as_ref().and_then(|rf| {
250        if rf.len() > 1 {
251            rf.first().map(|_| 0usize)
252        } else {
253            // If there's only one range, we already checked the flag.
254            None
255        }
256    });
257    ac.buffer.idx = 0;
258    loop {
259        // This block copied from NoncontextualSubtable::apply. Keep in sync.
260        if let Some(range_flags) = ac.range_flags.as_ref() {
261            if let Some(last_range) = last_range.as_mut() {
262                let mut range = *last_range;
263                if ac.buffer.idx < ac.buffer.len {
264                    let cluster = ac.buffer.cur(0).cluster;
265                    while cluster < range_flags[range].cluster_first {
266                        range -= 1;
267                    }
268
269                    while cluster > range_flags[range].cluster_last {
270                        range += 1;
271                    }
272
273                    *last_range = range;
274                }
275
276                if range_flags[range].flags & ac.subtable_flags == 0 {
277                    if ac.buffer.idx == ac.buffer.len || !ac.buffer.successful {
278                        break;
279                    }
280
281                    state = START_OF_TEXT;
282
283                    ac.buffer.next_glyph();
284                    continue;
285                }
286            }
287        }
288
289        let class = if ac.buffer.idx < ac.buffer.len {
290            get_class(
291                machine,
292                ac.buffer.cur(0).as_glyph(),
293                ac.machine_class_cache.unwrap(),
294            )
295        } else {
296            u16::from(aat::class::END_OF_TEXT)
297        };
298
299        let Ok(entry) = machine.entry(state, class) else {
300            break;
301        };
302
303        let next_state = entry.new_state;
304
305        // Conditions under which it's guaranteed safe-to-break before current glyph:
306        //
307        // 1. There was no action in this transition; and
308        //
309        // 2. If we break before current glyph, the results will be the same. That
310        //    is guaranteed if:
311        //
312        //    2a. We were already in start-of-text state; or
313        //
314        //    2b. We are epsilon-transitioning to start-of-text state; or
315        //
316        //    2c. Starting from start-of-text state seeing current glyph:
317        //
318        //        2c'. There won't be any actions; and
319        //
320        //        2c". We would end up in the same state that we were going to end up
321        //             in now, including whether epsilon-transitioning.
322        //
323        //    and
324        //
325        // 3. If we break before current glyph, there won't be any end-of-text action
326        //    after previous glyph.
327        //
328        // This triples the transitions we need to look up, but is worth returning
329        // granular unsafe-to-break results. See eg.:
330        //
331        //   https://github.com/harfbuzz/harfbuzz/issues/2860
332
333        let is_safe_to_break =
334            // 1
335            !Ctx::is_actionable(&entry) &&
336
337            // 2
338            (
339                state == START_OF_TEXT
340                || (!Ctx::can_advance(&entry) && next_state == START_OF_TEXT)
341                ||
342                {
343                    // 2c
344                    if let Ok(wouldbe_entry) = machine.entry(START_OF_TEXT, class) {
345                        // 2c'
346                        !Ctx::is_actionable(&wouldbe_entry) &&
347
348                        // 2c"
349                        (
350                            next_state == wouldbe_entry.new_state &&
351                            Ctx::can_advance(&entry) == Ctx::can_advance(&wouldbe_entry)
352                        )
353                    } else {
354                        false
355                    }
356                }
357            ) &&
358
359            // 3
360            (
361                if state < 64 {
362                    (ac.start_end_safe_to_break & (1 << state)) != 0
363                } else {
364                    if let Ok(end_entry) = machine.entry(state, u16::from(aat::class::END_OF_TEXT)) {
365                        !Ctx::is_actionable(&end_entry)
366                    } else {
367                        false
368                    }
369                }
370            )
371        ;
372
373        if !is_safe_to_break && ac.buffer.backtrack_len() > 0 && ac.buffer.idx < ac.buffer.len {
374            ac.buffer.unsafe_to_break_from_outbuffer(
375                Some(ac.buffer.backtrack_len() - 1),
376                Some(ac.buffer.idx + 1),
377            );
378        }
379
380        c.transition(&entry, ac);
381
382        state = next_state;
383
384        if ac.buffer.idx >= ac.buffer.len || !ac.buffer.successful {
385            break;
386        }
387
388        if Ctx::can_advance(&entry) {
389            ac.buffer.next_glyph();
390        } else {
391            if ac.buffer.max_ops <= 0 {
392                ac.buffer.next_glyph();
393            }
394            ac.buffer.max_ops -= 1;
395        }
396    }
397
398    if !Ctx::in_place() {
399        ac.buffer.sync();
400    }
401}

403fn apply_subtable<'a>(kind: SubtableKind<'a>, ac: &mut AatApplyContext<'a>) {
404    match kind {
405        SubtableKind::Rearrangement(table) => {
406            let mut c = RearrangementCtx { start: 0, end: 0 };
407            drive(&table, &mut c, ac);
408        }
409        SubtableKind::Contextual(table) => {
410            let mut c = ContextualCtx {
411                mark_set: false,
412                mark: 0,
413                table: table.clone(),
414            };
415            drive(&table.state_table, &mut c, ac);
416        }
417        SubtableKind::Ligature(table) => {
418            let mut c = LigatureCtx {
419                table: table.clone(),
420                match_length: 0,
421                match_positions: [0; LIGATURE_MAX_MATCHES],
422            };
423            drive(&table.state_table, &mut c, ac);
424        }
425        SubtableKind::NonContextual(ref lookup) => {
426            let mut last_range = ac.range_flags.as_ref().and_then(|rf| {
427                if rf.len() > 1 {
428                    rf.first().map(|_| 0usize)
429                } else {
430                    // If there's only one range, we already checked the flag.
431                    None
432                }
433            });
434
435            for i in 0..ac.buffer.len {
436                // This block copied from StateTableDriver::drive. Keep in sync.
437                if let Some(range_flags) = ac.range_flags.as_ref() {
438                    if let Some(last_range) = last_range.as_mut() {
439                        let mut range = *last_range;
440                        if ac.buffer.idx < ac.buffer.len {
441                            // We need to access info
442                            let cluster = ac.buffer.cur(0).cluster;
443                            while cluster < range_flags[range].cluster_first {
444                                range -= 1;
445                            }
446
447                            while cluster > range_flags[range].cluster_last {
448                                range += 1;
449                            }
450
451                            *last_range = range;
452                        }
453
454                        if range_flags[range].flags & ac.subtable_flags == 0 {
455                            continue;
456                        }
457                    }
458                }
459
460                if let Some(glyph) = ac.buffer.info[i].as_gid16() {
461                    if let Ok(replacement) = lookup.value(glyph.to_u16()) {
462                        ac.replace_glyph_inplace(i, replacement.into());
463                    }
464                }
465            }
466        }
467        SubtableKind::Insertion(table) => {
468            let mut c = InsertionCtx {
469                mark: 0,
470                glyphs: table.glyphs,
471            };
472            drive(&table.state_table, &mut c, ac);
473        }
474    }
475}

477struct RearrangementCtx {
478    start: usize,
479    end: usize,
480}
481
482impl RearrangementCtx {
483    const MARK_FIRST: u16 = 0x8000;
484    const DONT_ADVANCE: u16 = 0x4000;
485    const MARK_LAST: u16 = 0x2000;
486    const VERB: u16 = 0x000F;
487}
488
489impl DriverContext<NoPayload> for RearrangementCtx {
490    fn in_place() -> bool {
491        true
492    }
493
494    fn can_advance(entry: &StateEntry) -> bool {
495        entry.flags & Self::DONT_ADVANCE == 0
496    }
497
498    fn is_action_initiable(entry: &StateEntry) -> bool {
499        entry.flags & Self::MARK_FIRST != 0
500    }
501
502    fn is_actionable(entry: &StateEntry) -> bool {
503        entry.flags & Self::VERB != 0
504    }
505
506    #[inline(always)]
507    fn transition(&mut self, entry: &StateEntry, ac: &mut AatApplyContext) -> Option<()> {
508        let buffer = &mut ac.buffer;
509        let flags = entry.flags;
510
511        if flags & Self::MARK_FIRST != 0 {
512            self.start = buffer.idx;
513        }
514
515        if flags & Self::MARK_LAST != 0 {
516            self.end = (buffer.idx + 1).min(buffer.len);
517        }
518
519        if flags & Self::VERB != 0 && self.start < self.end {
520            // The following map has two nibbles, for start-side
521            // and end-side. Values of 0,1,2 mean move that many
522            // to the other side. Value of 3 means move 2 and
523            // flip them.
524            const MAP: [u8; 16] = [
525                0x00, // 0  no change
526                0x10, // 1  Ax => xA
527                0x01, // 2  xD => Dx
528                0x11, // 3  AxD => DxA
529                0x20, // 4  ABx => xAB
530                0x30, // 5  ABx => xBA
531                0x02, // 6  xCD => CDx
532                0x03, // 7  xCD => DCx
533                0x12, // 8  AxCD => CDxA
534                0x13, // 9  AxCD => DCxA
535                0x21, // 10 ABxD => DxAB
536                0x31, // 11 ABxD => DxBA
537                0x22, // 12 ABxCD => CDxAB
538                0x32, // 13 ABxCD => CDxBA
539                0x23, // 14 ABxCD => DCxAB
540                0x33, // 15 ABxCD => DCxBA
541            ];
542
543            let m = MAP[usize::from(flags & Self::VERB)];
544            let l = 2.min(m >> 4) as usize;
545            let r = 2.min(m & 0x0F) as usize;
546            let reverse_l = 3 == (m >> 4);
547            let reverse_r = 3 == (m & 0x0F);
548
549            if (self.end - self.start >= l + r) && (self.end - self.start <= MAX_CONTEXT_LENGTH) {
550                buffer.merge_clusters(self.start, (buffer.idx + 1).min(buffer.len));
551                buffer.merge_clusters(self.start, self.end);
552
553                let mut buf = [GlyphInfo::default(); 4];
554
555                for (i, glyph_info) in buf[..l].iter_mut().enumerate() {
556                    *glyph_info = buffer.info[self.start + i];
557                }
558
559                for i in 0..r {
560                    buf[i + 2] = buffer.info[self.end - r + i];
561                }
562
563                if l > r {
564                    for i in 0..(self.end - self.start - l - r) {
565                        buffer.info[self.start + r + i] = buffer.info[self.start + l + i];
566                    }
567                } else if l < r {
568                    for i in (0..(self.end - self.start - l - r)).rev() {
569                        buffer.info[self.start + r + i] = buffer.info[self.start + l + i];
570                    }
571                }
572
573                for i in 0..r {
574                    buffer.info[self.start + i] = buf[2 + i];
575                }
576
577                for i in 0..l {
578                    buffer.info[self.end - l + i] = buf[i];
579                }
580
581                if reverse_l {
582                    buffer.info.swap(self.end - 1, self.end - 2);
583                }
584
585                if reverse_r {
586                    buffer.info.swap(self.start, self.start + 1);
587                }
588            }
589        }
590
591        Some(())
592    }
593}

595struct ContextualCtx<'a> {
596    mark_set: bool,
597    mark: usize,
598    table: ContextualSubtable<'a>,
599}
600
601impl ContextualCtx<'_> {
602    const SET_MARK: u16 = 0x8000;
603    const DONT_ADVANCE: u16 = 0x4000;
604}
605
606impl DriverContext<ContextualEntryData> for ContextualCtx<'_> {
607    fn in_place() -> bool {
608        true
609    }
610
611    fn can_advance(entry: &StateEntry<ContextualEntryData>) -> bool {
612        entry.flags & Self::DONT_ADVANCE == 0
613    }
614
615    fn is_action_initiable(entry: &StateEntry<ContextualEntryData>) -> bool {
616        entry.flags & Self::SET_MARK != 0
617    }
618
619    fn is_actionable(entry: &StateEntry<ContextualEntryData>) -> bool {
620        entry.payload.mark_index.get() != 0xFFFF || entry.payload.current_index.get() != 0xFFFF
621    }
622
623    #[inline(always)]
624    fn transition(
625        &mut self,
626        entry: &StateEntry<ContextualEntryData>,
627        ac: &mut AatApplyContext,
628    ) -> Option<()> {
629        // Looks like CoreText applies neither mark nor current substitution for
630        // end-of-text if mark was not explicitly set.
631        if ac.buffer.idx == ac.buffer.len && !self.mark_set {
632            return Some(());
633        }
634
635        let mut replacement = None;
636
637        if entry.payload.mark_index.get() != 0xFFFF {
638            let lookup = self
639                .table
640                .lookups
641                .get(usize::from(entry.payload.mark_index.get()))
642                .ok()?;
643            if let Some(gid) = ac.buffer.info[self.mark].as_gid16() {
644                replacement = lookup.value(gid.to_u16()).ok();
645            }
646        }
647
648        if let Some(replacement) = replacement {
649            ac.buffer.unsafe_to_break(
650                Some(self.mark),
651                Some((ac.buffer.idx + 1).min(ac.buffer.len)),
652            );
653            ac.replace_glyph_inplace(self.mark, replacement.into());
654        }
655
656        replacement = None;
657        let idx = ac.buffer.idx.min(ac.buffer.len - 1);
658        if entry.payload.current_index.get() != 0xFFFF {
659            let lookup = self
660                .table
661                .lookups
662                .get(usize::from(entry.payload.current_index.get()))
663                .ok()?;
664            if let Some(gid) = ac.buffer.info[idx].as_gid16() {
665                replacement = lookup.value(gid.to_u16()).ok();
666            }
667        }
668
669        if let Some(replacement) = replacement {
670            ac.replace_glyph_inplace(idx, replacement.into());
671        }
672
673        if entry.flags & Self::SET_MARK != 0 {
674            self.mark_set = true;
675            self.mark = ac.buffer.idx;
676        }
677
678        Some(())
679    }
680}

693    const MARKED_INSERT_COUNT: u16 = 0x001F;
694}
695
696impl DriverContext<InsertionEntryData> for InsertionCtx<'_> {
697    fn in_place() -> bool {
698        false
699    }
700
701    fn can_advance(entry: &StateEntry<InsertionEntryData>) -> bool {
702        entry.flags & Self::DONT_ADVANCE == 0
703    }
704
705    fn is_action_initiable(entry: &StateEntry<InsertionEntryData>) -> bool {
706        entry.flags & Self::SET_MARK != 0
707    }
708
709    fn is_actionable(entry: &StateEntry<InsertionEntryData>) -> bool {
710        (entry.flags & (Self::CURRENT_INSERT_COUNT | Self::MARKED_INSERT_COUNT) != 0)
711            && (entry.payload.current_insert_index.get() != 0xFFFF
712                || entry.payload.marked_insert_index.get() != 0xFFFF)
713    }
714
715    #[inline(always)]
716    fn transition(
717        &mut self,
718        entry: &StateEntry<InsertionEntryData>,
719        ac: &mut AatApplyContext,
720    ) -> Option<()> {
721        let flags = entry.flags;
722        let mark_loc = ac.buffer.out_len;
723
724        if entry.payload.marked_insert_index.get() != 0xFFFF {
725            let count = flags & Self::MARKED_INSERT_COUNT;
726            ac.buffer.max_ops -= i32::from(count);
727            if ac.buffer.max_ops <= 0 {
728                return Some(());
729            }
730
731            let start = entry.payload.marked_insert_index.get();
732            let before = flags & Self::MARKED_INSERT_BEFORE != 0;
733
734            let end = ac.buffer.out_len;
735            if !ac.buffer.move_to(self.mark as usize) {
736                return Some(());
737            }
738
739            if ac.buffer.idx < ac.buffer.len && !before {
740                ac.buffer.copy_glyph();
741            }
742
743            // TODO We ignore KashidaLike setting.
744            for i in 0..count {
745                let i = usize::from(start + i);
746                ac.output_glyph(u32::from(self.glyphs.get(i)?.get().to_u16()));
747            }
748
749            if ac.buffer.idx < ac.buffer.len && !before {
750                ac.buffer.skip_glyph();
751            }
752
753            if !ac.buffer.move_to(end + usize::from(count)) {
754                return Some(());
755            }
756
757            ac.buffer.unsafe_to_break_from_outbuffer(
758                Some(self.mark as usize),
759                Some((ac.buffer.idx + 1).min(ac.buffer.len)),
760            );
761        }
762
763        if flags & Self::SET_MARK != 0 {
764            self.mark = mark_loc as u32;
765        }
766
767        if entry.payload.current_insert_index.get() != 0xFFFF {
768            let count = (flags & Self::CURRENT_INSERT_COUNT) >> 5;
769            ac.buffer.max_ops -= i32::from(count);
770            if ac.buffer.max_ops < 0 {
771                return Some(());
772            }
773
774            let start = entry.payload.current_insert_index.get();
775            let before = flags & Self::CURRENT_INSERT_BEFORE != 0;
776            let end = ac.buffer.out_len;
777
778            if ac.buffer.idx < ac.buffer.len && !before {
779                ac.buffer.copy_glyph();
780            }
781
782            // TODO We ignore KashidaLike setting.
783            for i in 0..count {
784                let i = usize::from(start + i);
785                ac.output_glyph(u32::from(self.glyphs.get(i)?.get().to_u16()));
786            }
787
788            if ac.buffer.idx < ac.buffer.len && !before {
789                ac.buffer.skip_glyph();
790            }
791
792            // Humm. Not sure where to move to. There's this wording under
793            // DontAdvance flag:
794            //
795            // "If set, don't update the glyph index before going to the new state.
796            // This does not mean that the glyph pointed to is the same one as
797            // before. If you've made insertions immediately downstream of the
798            // current glyph, the next glyph processed would in fact be the first
799            // one inserted."
800            //
801            // This suggests that if DontAdvance is NOT set, we should move to
802            // end+count. If it *was*, then move to end, such that newly inserted
803            // glyphs are now visible.
804            //
805            // https://github.com/harfbuzz/harfbuzz/issues/1224#issuecomment-427691417
806            if !ac.buffer.move_to(if flags & Self::DONT_ADVANCE != 0 {
807                end
808            } else {
809                end + usize::from(count)
810            }) {
811                return Some(());
812            }
813        }
814
815        Some(())
816    }
817}
818
819const LIGATURE_MAX_MATCHES: usize = 64;
820
821struct LigatureCtx<'a> {
822    table: LigatureSubtable<'a>,
823    match_length: usize,
824    match_positions: [usize; LIGATURE_MAX_MATCHES],
825}
826
827impl LigatureCtx<'_> {
828    const SET_COMPONENT: u16 = 0x8000;
829    const DONT_ADVANCE: u16 = 0x4000;
830    const PERFORM_ACTION: u16 = 0x2000;
831
832    const LIG_ACTION_LAST: u32 = 0x8000_0000;
833    const LIG_ACTION_STORE: u32 = 0x4000_0000;
834    const LIG_ACTION_OFFSET: u32 = 0x3FFF_FFFF;
835}
836
837impl DriverContext<BigEndian<u16>> for LigatureCtx<'_> {
838    fn in_place() -> bool {
839        false
840    }
841
842    fn can_advance(entry: &StateEntry<BigEndian<u16>>) -> bool {
843        entry.flags & Self::DONT_ADVANCE == 0
844    }
845
846    fn is_action_initiable(entry: &StateEntry<BigEndian<u16>>) -> bool {
847        entry.flags & Self::SET_COMPONENT != 0
848    }
849
850    fn is_actionable(entry: &StateEntry<BigEndian<u16>>) -> bool {
851        entry.flags & Self::PERFORM_ACTION != 0
852    }
853
854    #[inline(always)]
855    fn transition(
856        &mut self,
857        entry: &StateEntry<BigEndian<u16>>,
858        ac: &mut AatApplyContext,
859    ) -> Option<()> {
860        if entry.flags & Self::SET_COMPONENT != 0 {
861            // Never mark same index twice, in case DONT_ADVANCE was used...
862            if self.match_length != 0
863                && self.match_positions[(self.match_length - 1) % LIGATURE_MAX_MATCHES]
864                    == ac.buffer.out_len
865            {
866                self.match_length -= 1;
867            }
868
869            self.match_positions[self.match_length % LIGATURE_MAX_MATCHES] = ac.buffer.out_len;
870            self.match_length += 1;
871        }
872
873        if entry.flags & Self::PERFORM_ACTION != 0 {
874            let end = ac.buffer.out_len;
875
876            if self.match_length == 0 {
877                return Some(());
878            }
879
880            if ac.buffer.idx >= ac.buffer.len {
881                return Some(()); // TODO: Work on previous instead?
882            }
883
884            let mut cursor = self.match_length;
885
886            let mut ligature_actions_index = entry.payload.get();
887            let mut ligature_idx = 0;
888            loop {
889                if cursor == 0 {
890                    // Stack underflow. Clear the stack.
891                    self.match_length = 0;
892                    break;
893                }
894
895                cursor -= 1;
896                if !ac
897                    .buffer
898                    .move_to(self.match_positions[cursor % LIGATURE_MAX_MATCHES])
899                {
900                    return Some(());
901                }
902
903                // We cannot use ? in this loop, because we must call
904                // ac.buffer.move_to(end) in the end.
905                let action = match self
906                    .table
907                    .ligature_actions
908                    .get(usize::from(ligature_actions_index))
909                {
910                    Some(v) => v.get(),
911                    None => break,
912                };
913
914                let mut uoffset = action & Self::LIG_ACTION_OFFSET;
915                if uoffset & 0x2000_0000 != 0 {
916                    uoffset |= 0xC000_0000; // Sign-extend.
917                }
918
919                let offset = uoffset as i32;
920                let component_idx = (ac.buffer.cur(0).glyph_id as i32 + offset) as usize;
921                ligature_idx += match self.table.components.get(component_idx) {
922                    Some(v) => v.get(),
923                    None => break,
924                };
925
926                if (action & (Self::LIG_ACTION_STORE | Self::LIG_ACTION_LAST)) != 0 {
927                    let lig = match self.table.ligatures.get(usize::from(ligature_idx)) {
928                        Some(v) => v.get(),
929                        None => break,
930                    };
931
932                    ac.replace_glyph(u32::from(lig.to_u16()));
933
934                    let lig_end =
935                        self.match_positions[(self.match_length - 1) % LIGATURE_MAX_MATCHES] + 1;
936                    // Now go and delete all subsequent components.
937                    while self.match_length - 1 > cursor {
938                        self.match_length -= 1;
939                        if !ac
940                            .buffer
941                            .move_to(self.match_positions[self.match_length % LIGATURE_MAX_MATCHES])
942                        {
943                            return Some(());
944                        }
945                        ac.delete_glyph();
946                    }
947
948                    if !ac.buffer.move_to(lig_end) {
949                        return Some(());
950                    }
951                    ac.buffer.merge_out_clusters(
952                        self.match_positions[cursor % LIGATURE_MAX_MATCHES],
953                        ac.buffer.out_len,
954                    );
955                }
956
957                ligature_actions_index += 1;
958
959                if action & Self::LIG_ACTION_LAST != 0 {
960                    break;
961                }
962            }
963
964            if !ac.buffer.move_to(end) {
965                return Some(());
966            }
967        }
968
969        Some(())
970    }
971}

973pub(crate) struct MorxSubtableCache {
974    start_end_safe_to_break: u64,
975    glyph_set: U32Set,
976    class_cache: ClassCache,
977}
978
979impl MorxSubtableCache {
980    pub(crate) fn new(subtable: &Subtable, num_glyphs: u32) -> Self {
981        let mut start_end_safe_to_break = 0u64;
982        let mut glyph_set = U32Set::default();
983        if let Ok(kind) = subtable.kind() {
984            match &kind {
985                SubtableKind::Rearrangement(table) => {
986                    start_end_safe_to_break =
987                        collect_start_end_safe_to_break::<_, RearrangementCtx>(table);
988                    collect_initial_glyphs::<_, RearrangementCtx>(
989                        table,
990                        &mut glyph_set,
991                        num_glyphs,
992                    );
993                }
994                SubtableKind::Contextual(table) => {
995                    start_end_safe_to_break =
996                        collect_start_end_safe_to_break::<_, ContextualCtx>(&table.state_table);
997                    collect_initial_glyphs::<_, ContextualCtx>(
998                        &table.state_table,
999                        &mut glyph_set,
1000                        num_glyphs,
1001                    );
1002                }
1003                SubtableKind::Ligature(table) => {
1004                    start_end_safe_to_break =
1005                        collect_start_end_safe_to_break::<_, LigatureCtx>(&table.state_table);
1006                    collect_initial_glyphs::<_, LigatureCtx>(
1007                        &table.state_table,
1008                        &mut glyph_set,
1009                        num_glyphs,
1010                    );
1011                }
1012                SubtableKind::NonContextual(ref lookup) => {
1013                    lookup.collect_glyphs(&mut glyph_set, num_glyphs);
1014                }
1015                SubtableKind::Insertion(table) => {
1016                    start_end_safe_to_break =
1017                        collect_start_end_safe_to_break::<_, InsertionCtx>(&table.state_table);
1018                    collect_initial_glyphs::<_, InsertionCtx>(
1019                        &table.state_table,
1020                        &mut glyph_set,
1021                        num_glyphs,
1022                    );
1023                }
1024            }
1025        };
1026        MorxSubtableCache {
1027            start_end_safe_to_break,
1028            glyph_set,
1029            class_cache: ClassCache::new(),
1030        }
1031    }
