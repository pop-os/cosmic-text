27pub(crate) fn apply(c: &mut AatApplyContext) -> Option<()> {
28    c.buffer.unsafe_to_concat(None, None);
29
30    c.setup_buffer_glyph_set();
31
32    let (kerx, subtable_caches) = c.face.aat_tables.kerx.as_ref()?;
33
34    let mut subtable_idx = 0;
35
36    let mut seen_cross_stream = false;
37    for subtable in kerx.subtables().iter() {
38        let Ok(subtable) = subtable else {
39            continue;
40        };
41
42        let subtable_cache = subtable_caches.get(subtable_idx);
43        let Some(subtable_cache) = subtable_cache.as_ref() else {
44            break;
45        };
46        subtable_idx += 1;
47
48        // We don't handle variations
49        if subtable.is_variable() {
50            continue;
51        }
52
53        if c.buffer.direction.is_horizontal() != subtable.is_horizontal() {
54            continue;
55        }
56
57        c.first_set = Some(&subtable_cache.first_set);
58        c.second_set = Some(&subtable_cache.second_set);
59        c.machine_class_cache = Some(&subtable_cache.class_cache);
60        c.start_end_safe_to_break = subtable_cache.start_end_safe_to_break;
61
62        if !c.buffer_intersects_machine() {
63            continue;
64        }
65
66        let reverse = c.buffer.direction.is_backward();
67
68        if !seen_cross_stream && subtable.is_cross_stream() {
69            seen_cross_stream = true;
70
71            // Attach all glyphs into a chain.
72            for pos in &mut c.buffer.pos {
73                pos.set_attach_type(attach_type::CURSIVE);
74                pos.set_attach_chain(if c.buffer.direction.is_forward() {
75                    -1
76                } else {
77                    1
78                });
79                // We intentionally don't set BufferScratchFlags::HAS_GPOS_ATTACHMENT,
80                // since there needs to be a non-zero attachment for post-positioning to
81                // be needed.
82            }
83        }
84
85        let Ok(kind) = subtable.kind() else {
86            continue;
87        };
88
89        if reverse != c.buffer_is_reversed {
90            c.reverse_buffer();
91        }
92
93        match &kind {
94            SubtableKind::Format0(format0) => {
95                if !c.plan.requested_kerning {
96                    continue;
97                }
98                apply_simple_kerning(c, &subtable, format0);
99            }
100            SubtableKind::Format1(format1) => {
101                let mut driver = Driver1 {
102                    stack: [0; 8],
103                    depth: 0,
104                };
105                apply_state_machine_kerning(
106                    c,
107                    &subtable,
108                    format1,
109                    &format1.state_table,
110                    &mut driver,
111                );
112            }
113            SubtableKind::Format2(format2) => {
114                if !c.plan.requested_kerning {
115                    continue;
116                }
117                apply_simple_kerning(c, &subtable, format2);
118            }
119            SubtableKind::Format4(format4) => {
120                let mut driver = Driver4 {
121                    mark_set: false,
122                    mark: 0,
123                    ankr_table: c.face.aat_tables.ankr.clone(),
124                };
125                apply_state_machine_kerning(
126                    c,
127                    &subtable,
128                    format4,
129                    &format4.state_table,
130                    &mut driver,
131                );
132            }
133            SubtableKind::Format6(format6) => {
134                if !c.plan.requested_kerning {
135                    continue;
136                }
137                apply_simple_kerning(c, &subtable, format6);
138            }
139        }
140    }
141    if c.buffer_is_reversed {
142        c.reverse_buffer();
143    }
144
145    Some(())
146}

148pub trait SimpleKerning {
149    fn simple_kerning(&self, left: GlyphId, right: GlyphId) -> Option<i32>;
150    fn collect_glyphs(&self, _first_set: &mut U32Set, _second_set: &mut U32Set, _num_glyphs: u32);
151}

153impl SimpleKerning for Subtable0<'_> {
154    fn simple_kerning(&self, left: GlyphId, right: GlyphId) -> Option<i32> {
155        self.kerning(left, right)
156    }
157    fn collect_glyphs(&self, first_set: &mut U32Set, second_set: &mut U32Set, _num_glyphs: u32) {
158        for &pair in self.pairs() {
159            first_set.insert(pair.left.get().to_u32());
160            second_set.insert(pair.right.get().to_u32());
161        }
162    }

196fn apply_simple_kerning<T: SimpleKerning>(c: &mut AatApplyContext, subtable: &Subtable, kind: &T) {
197    let mut ctx = hb_ot_apply_context_t::new(TableIndex::GPOS, c.face, c.buffer);
198    ctx.set_lookup_mask(c.plan.kern_mask);
199    ctx.lookup_props = u32::from(lookup_flags::IGNORE_MARKS);
200    ctx.update_matchers();
201
202    let horizontal = ctx.buffer.direction.is_horizontal();
203    let cross_stream = subtable.is_cross_stream();
204
205    let first_set = c.first_set.as_ref().unwrap();
206    let second_set = c.second_set.as_ref().unwrap();
207
208    let mut i = 0;
209    let mut iter = skipping_iterator_t::new(&mut ctx, false);
210    while i < iter.buffer.len {
211        if (iter.buffer.info[i].mask & c.plan.kern_mask) == 0 {
212            i += 1;
213            continue;
214        }
215
216        iter.reset_fast(i);
217
218        let mut unsafe_to = 0;
219        if !iter.next(Some(&mut unsafe_to)) {
220            iter.buffer.unsafe_to_concat(Some(i), Some(unsafe_to));
221            i += 1;
222            continue;
223        }
224
225        let j = iter.index();
226
227        let info = &iter.buffer.info;
228        let a = info[i].as_glyph();
229        let b = info[j].as_glyph();
230        let kern = if !first_set.contains(a.to_u32()) || !second_set.contains(b.to_u32()) {
231            0
232        } else {
233            kind.simple_kerning(a, b).unwrap_or(0)
234        };
235
236        let pos = &mut iter.buffer.pos;
237        if kern != 0 {
238            if horizontal {
239                if cross_stream {
240                    pos[j].y_offset = kern;
241                    iter.buffer.scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_GPOS_ATTACHMENT;
242                } else {
243                    let kern1 = kern >> 1;
244                    let kern2 = kern - kern1;
245                    pos[i].x_advance += kern1;
246                    pos[j].x_advance += kern2;
247                    pos[j].x_offset += kern2;
248                }
249            } else {
250                if cross_stream {
251                    pos[j].x_offset = kern;
252                    iter.buffer.scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_GPOS_ATTACHMENT;
253                } else {
254                    let kern1 = kern >> 1;
255                    let kern2 = kern - kern1;
256                    pos[i].y_advance += kern1;
257                    pos[j].y_advance += kern2;
258                    pos[j].y_offset += kern2;
259                }
260            }
261
262            iter.buffer.unsafe_to_break(Some(i), Some(j + 1));
263        }
264
265        i = j;
266    }
267}

269pub(crate) trait KerxStateEntryExt {
270    fn flags(&self) -> u16;
271    fn action_index(&self) -> u16;
272
273    fn is_action_initiable(&self) -> bool {
274        self.flags() & 0x8000 != 0
275    }
276
277    fn is_actionable(&self) -> bool {
278        self.action_index() != 0xFFFF
279    }
280
281    fn has_advance(&self) -> bool {
282        self.flags() & 0x4000 == 0
283    }
284
285    fn has_reset(&self) -> bool {
286        self.flags() & 0x2000 != 0
287    }
288
289    fn has_push(&self) -> bool {
290        self.flags() & 0x8000 != 0
291    }
292
293    fn has_mark(&self) -> bool {
294        self.flags() & 0x8000 != 0
295    }
296}

308fn collect_initial_glyphs<T>(
309    machine: &aat::ExtendedStateTable<T>,
310    glyphs: &mut U32Set,
311    num_glyphs: u32,
312) where
313    T: FixedSize + bytemuck::AnyBitPattern,
314    aat::StateEntry<T>: KerxStateEntryExt,
315{
316    let mut classes = U32Set::default();
317
318    let class_table = &machine.class_table;
319    for i in 0..machine.n_classes {
320        if let Ok(entry) = machine.entry(START_OF_TEXT, i as u16) {
321            if entry.new_state == START_OF_TEXT
322                && !entry.is_action_initiable()
323                && !entry.is_actionable()
324            {
325                continue;
326            }
327            classes.insert(i as u32);
328        }
329    }
330
331    // And glyphs in those classes.
332
333    let filter = |class: u16| classes.contains(class as u32);
334
335    if filter(aat::class::DELETED_GLYPH as u16) {
336        glyphs.insert(DELETED_GLYPH);
337    }
338
339    class_table.collect_glyphs_filtered(glyphs, num_glyphs, filter);
340}

342fn collect_start_end_safe_to_break<T>(machine: &aat::ExtendedStateTable<T>) -> u64
343where
344    T: FixedSize + bytemuck::AnyBitPattern,
345    aat::StateEntry<T>: KerxStateEntryExt,
346{
347    let mut result = 0u64;
348    for state in 0..64 {
349        let bit = if let Ok(entry) = machine.entry(state, aat::class::END_OF_TEXT as u16) {
350            !entry.is_actionable()
351        } else {
352            true
353        };
354        if bit {
355            result |= 1 << state;
356        }
357    }
358    result
359}

361fn apply_state_machine_kerning<T, E, Driver: StateTableDriver<T, E>>(
362    c: &mut AatApplyContext,
363    subtable: &Subtable,
364    kind: &T,
365    state_table: &aat::ExtendedStateTable<E>,
366    driver: &mut Driver,
367) where
368    E: FixedSize + bytemuck::AnyBitPattern,
369    aat::StateEntry<E>: KerxStateEntryExt,
370{
371    let mut state = START_OF_TEXT;
372    c.buffer.idx = 0;
373    loop {
374        let class = if c.buffer.idx < c.buffer.len {
375            get_class(
376                state_table,
377                c.buffer.cur(0).as_glyph(),
378                c.machine_class_cache.unwrap(),
379            )
380        } else {
381            u16::from(aat::class::END_OF_TEXT)
382        };
383
384        let Ok(entry) = state_table.entry(state, class) else {
385            break;
386        };
387
388        let next_state = entry.new_state;
389
390        // Conditions under which it's guaranteed safe-to-break before current glyph:
391        //
392        // 1. There was no action in this transition; and
393        //
394        // 2. If we break before current glyph, the results will be the same. That
395        //    is guaranteed if:
396        //
397        //    2a. We were already in start-of-text state; or
398        //
399        //    2b. We are epsilon-transitioning to start-of-text state; or
400        //
401        //    2c. Starting from start-of-text state seeing current glyph:
402        //
403        //        2c'. There won't be any actions; and
404        //
405        //        2c". We would end up in the same state that we were going to end up
406        //             in now, including whether epsilon-transitioning.
407        //
408        //    and
409        //
410        // 3. If we break before current glyph, there won't be any end-of-text action
411        //    after previous glyph.
412        //
413        // This triples the transitions we need to look up, but is worth returning
414        // granular unsafe-to-break results. See eg.:
415        //
416        //   https://github.com/harfbuzz/harfbuzz/issues/2860
417
418        let is_safe_to_break =
419            // 1
420            !entry.is_actionable() &&
421
422            // 2
423            (
424                state == START_OF_TEXT
425                || (!entry.has_advance() && next_state == START_OF_TEXT)
426                ||
427                {
428                    // 2c
429                    if let Ok(wouldbe_entry) = state_table.entry(START_OF_TEXT, class) {
430                        // 2c'
431                        !wouldbe_entry.is_actionable() &&
432
433                        // 2c"
434                        (
435                            next_state == wouldbe_entry.new_state &&
436                            entry.has_advance() == wouldbe_entry.has_advance()
437                        )
438                    } else {
439                        false
440                    }
441                }
442            ) &&
443
444            // 3
445            (
446                if state < 64 {
447                    (c.start_end_safe_to_break & (1 << state)) != 0
448                } else {
449                    if let Ok(end_entry) = state_table.entry(state, u16::from(aat::class::END_OF_TEXT)) {
450                        !end_entry.is_actionable()
451                    } else {
452                        false
453                    }
454                }
455            )
456        ;
457
458        if !is_safe_to_break && c.buffer.backtrack_len() > 0 && c.buffer.idx < c.buffer.len {
459            c.buffer.unsafe_to_break_from_outbuffer(
460                Some(c.buffer.backtrack_len() - 1),
461                Some(c.buffer.idx + 1),
462            );
463        }
464
465        let _ = driver.transition(
466            kind,
467            &entry,
468            subtable.is_cross_stream(),
469            subtable.tuple_count(),
470            c,
471        );
472
473        state = next_state;
474
475        if c.buffer.idx >= c.buffer.len {
476            break;
477        }
478
479        if entry.has_advance() || c.buffer.max_ops <= 0 {
480            c.buffer.next_glyph();
481        }
482        c.buffer.max_ops -= 1;
483    }
484}

486trait StateTableDriver<T, E> {
487    fn transition(
488        &mut self,
489        aat: &T,
490        entry: &aat::StateEntry<E>,
491        has_cross_stream: bool,
492        tuple_count: u32,
493        c: &mut AatApplyContext,
494    ) -> Option<()>;
495}

497struct Driver1 {
498    stack: [usize; 8],
499    depth: usize,
500}
501
502impl StateTableDriver<Subtable1<'_>, BigEndian<u16>> for Driver1 {
503    #[inline(always)]
504    fn transition(
505        &mut self,
506        aat: &Subtable1,
507        entry: &aat::StateEntry<BigEndian<u16>>,
508        has_cross_stream: bool,
509        tuple_count: u32,
510        c: &mut AatApplyContext,
511    ) -> Option<()> {
512        if entry.has_reset() {
513            self.depth = 0;
514        }
515
516        if entry.has_push() {
517            if self.depth < self.stack.len() {
518                self.stack[self.depth] = c.buffer.idx;
519                self.depth += 1;
520            } else {
521                self.depth = 0; // Probably not what CoreText does, but better?
522            }
523        }
524
525        if entry.is_actionable() && self.depth != 0 {
526            let tuple_count = u16::try_from(tuple_count.max(1)).ok()?;
527
528            let mut action_index = entry.action_index();
529
530            // From Apple 'kern' spec:
531            // "Each pops one glyph from the kerning stack and applies the kerning value to it.
532            // The end of the list is marked by an odd value...
533            let mut last = false;
534            while !last && self.depth != 0 {
535                self.depth -= 1;
536                let idx = self.stack[self.depth];
537                let mut v = aat.values.get(action_index as usize)?.get() as i32;
538                action_index = action_index.checked_add(tuple_count)?;
539                if idx >= c.buffer.len {
540                    continue;
541                }
542
543                // "The end of the list is marked by an odd value..."
544                last = v & 1 != 0;
545                v &= !1;
546
547                // Testing shows that CoreText only applies kern (cross-stream or not)
548                // if none has been applied by previous subtables. That is, it does
549                // NOT seem to accumulate as otherwise implied by specs.
550
551                let mut has_gpos_attachment = false;
552                let glyph_mask = c.buffer.info[idx].mask;
553                let pos = &mut c.buffer.pos[idx];
554
555                if c.buffer.direction.is_horizontal() {
556                    if has_cross_stream {
557                        // The following flag is undocumented in the spec, but described
558                        // in the 'kern' table example.
559                        if v == -0x8000 {
560                            pos.set_attach_type(0);
561                            pos.set_attach_chain(0);
562                            pos.y_offset = 0;
563                        } else if pos.attach_type() != 0 {
564                            pos.y_offset += v;
565                            has_gpos_attachment = true;
566                        }
567                    } else if glyph_mask & c.plan.kern_mask != 0 {
568                        pos.x_advance += v;
569                        pos.x_offset += v;
570                    }
571                } else {
572                    if has_cross_stream {
573                        // CoreText doesn't do crossStream kerning in vertical. We do.
574                        if v == -0x8000 {
575                            pos.set_attach_type(0);
576                            pos.set_attach_chain(0);
577                            pos.x_offset = 0;
578                        } else if pos.attach_type() != 0 {
579                            pos.x_offset += v;
580                            has_gpos_attachment = true;
581                        }
582                    } else if glyph_mask & c.plan.kern_mask != 0 {
583                        if pos.y_offset == 0 {
584                            pos.y_advance += v;
585                            pos.y_offset += v;
586                        }
587                    }
588                }
589
590                if has_gpos_attachment {
591                    c.buffer.scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_GPOS_ATTACHMENT;
592                }
593            }
594        }
595
596        Some(())
597    }
598}
599struct Driver4<'a> {
600    mark_set: bool,
601    mark: usize,
602    ankr_table: Option<Ankr<'a>>,
603}
604
605impl StateTableDriver<Subtable4<'_>, BigEndian<u16>> for Driver4<'_> {
606    #[inline(always)]
607    fn transition(
608        &mut self,
609        aat: &Subtable4,
610        entry: &aat::StateEntry<BigEndian<u16>>,
611        _has_cross_stream: bool,
612        _tuple_count: u32,
613        c: &mut AatApplyContext,
614    ) -> Option<()> {
615        if self.mark_set && entry.is_actionable() && c.buffer.idx < c.buffer.len {
616            match (self.ankr_table.as_ref(), &aat.actions) {
617                (Some(ankr_table), Subtable4Actions::AnchorPoints(ankr_data)) => {
618                    let action_idx = entry.action_index() as usize * 2;
619                    let mark_action_idx = ankr_data.get(action_idx)?.get() as usize;
620                    let curr_action_idx = ankr_data.get(action_idx + 1)?.get() as usize;
621                    let mark_idx = c.buffer.info[self.mark].as_glyph();
622                    let mark_anchor = ankr_table
623                        .anchor_points(mark_idx)
624                        .ok()
625                        .and_then(|list| list.get(mark_action_idx))
626                        .map(|point| (point.x(), point.y()))
627                        .unwrap_or_default();
628
629                    let curr_idx = c.buffer.cur(0).as_glyph();
630                    let curr_anchor = ankr_table
631                        .anchor_points(curr_idx)
632                        .ok()
633                        .and_then(|list| list.get(curr_action_idx))
634                        .map(|point| (point.x(), point.y()))
635                        .unwrap_or_default();
636
637                    let pos = c.buffer.cur_pos_mut();
638                    pos.x_offset = i32::from(mark_anchor.0 - curr_anchor.0);
639                    pos.y_offset = i32::from(mark_anchor.1 - curr_anchor.1);
640                }
641                (_, Subtable4Actions::ControlPointCoords(coords)) => {
642                    let action_idx = entry.action_index() as usize * 4;
643                    let mark_x = coords.get(action_idx)?.get() as i32;
644                    let mark_y = coords.get(action_idx + 1)?.get() as i32;
645                    let curr_x = coords.get(action_idx + 2)?.get() as i32;
646                    let curr_y = coords.get(action_idx + 3)?.get() as i32;
647                    let pos = c.buffer.cur_pos_mut();
648                    pos.x_offset = mark_x - curr_x;
649                    pos.y_offset = mark_y - curr_y;
650                }
651                _ => {}
652            }
653
654            c.buffer.cur_pos_mut().set_attach_type(attach_type::MARK);
655            let idx = c.buffer.idx;
656            let mut attach_chain = self.mark as i16 - idx as i16;
657            if c.buffer_is_reversed {
658                attach_chain = -attach_chain;
659            }
660            c.buffer.cur_pos_mut().set_attach_chain(attach_chain);
661            c.buffer.scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_GPOS_ATTACHMENT;
662        }
663
664        if entry.has_mark() {
665            self.mark_set = true;
666            self.mark = c.buffer.idx;
667        }
668
669        Some(())
670    }
671}

673pub(crate) struct KerxSubtableCache {

675    first_set: U32Set,

680impl KerxSubtableCache {
681    pub(crate) fn new(subtable: &Subtable, num_glyphs: u32) -> Self {
682        let mut start_end_safe_to_break = 0u64;
683        let mut first_set = U32Set::default();
684        let mut second_set = U32Set::default();
685        if let Ok(kind) = subtable.kind() {
686            match &kind {
687                SubtableKind::Format0(format0) => {
688                    format0.collect_glyphs(&mut first_set, &mut second_set, num_glyphs);

690                SubtableKind::Format1(format1) => {
691                    start_end_safe_to_break = collect_start_end_safe_to_break(&format1.state_table);
692                    collect_initial_glyphs(&format1.state_table, &mut first_set, num_glyphs);
693                }
694                SubtableKind::Format2(format2) => {
695                    format2.collect_glyphs(&mut first_set, &mut second_set, num_glyphs);
696                }
697                SubtableKind::Format4(format4) => {
698                    start_end_safe_to_break = collect_start_end_safe_to_break(&format4.state_table);
699                    collect_initial_glyphs(&format4.state_table, &mut first_set, num_glyphs);
700                }
701                SubtableKind::Format6(format6) => {
702                    format6.collect_glyphs(&mut first_set, &mut second_set, num_glyphs);
703                }
704            }
705        };
706        KerxSubtableCache {

711        }
