81impl TableOffsets {
82    pub fn new(font: &FontRef) -> Self {
83        let num_glyphs = font
84            .maxp()
85            .map(|maxp| maxp.num_glyphs() as u32)
86            .unwrap_or_default();
87        let (units_per_em, loca_long) = font
88            .head()
89            .map(|head| (head.units_per_em(), head.index_to_loc_format() == 1))
90            .unwrap_or((1000, false));
91        let os2 = font.os2().ok();
92        let hhea = font.hhea().ok();
93        let (ascent, descent) = if let Some(os2) = &os2 {
94            (os2.s_typo_ascender(), os2.s_typo_descender())
95        } else if let Some(hhea) = &hhea {
96            (hhea.ascender().to_i16(), hhea.descender().to_i16())
97        } else {
98            (0, 0) // TODO
99        };
100        let num_h_metrics = hhea
101            .map(|hhea| hhea.number_of_h_metrics())
102            .unwrap_or_default();
103        let num_v_metrics = font
104            .vhea()
105            .map(|vhea| vhea.number_of_long_ver_metrics())
106            .unwrap_or_default();
107        let offset = |tag| TableOffset::new(font, tag).unwrap_or_default();
108        let loca = offset(Loca::TAG);
109        let glyf = offset(Glyf::TAG);
110        let gvar = offset(Gvar::TAG);
111        let hmtx = offset(Hmtx::TAG);
112        let hvar = offset(Hvar::TAG);
113        let vmtx = offset(Vmtx::TAG);
114        let vvar = offset(Vvar::TAG);
115        let vorg = offset(Vorg::TAG);
116        let mvar = offset(Mvar::TAG);
117        let cmap = offset(Cmap::TAG);
118        let cmap_table: Option<Cmap> = cmap.resolve_table(font);
119        let cmap_subtable = cmap_table
120            .as_ref()
121            .and_then(|cmap| find_best_cmap_subtable(cmap))
122            .map(|(index, platform, encoding, _)| SelectedCmapSubtable {
123                index,
124                is_mac_roman: platform == PlatformId::Macintosh,
125                is_symbol: platform == PlatformId::Windows && encoding == WINDOWS_SYMBOL_ENCODING,
126            });
127        let cmap_vs_subtable = cmap_table.and_then(|cmap| {
128            let data = cmap.offset_data();
129            cmap.encoding_records()
130                .iter()
131                .enumerate()
132                .filter_map(|(index, record)| Some((index, record.subtable(data).ok()?)))
133                .find_map(|(index, subtable)| match subtable {
134                    CmapSubtable::Format14(_) => Some(index as u16),
135                    _ => None,
136                })
137        });
138        let gdef = offset(Gdef::TAG);
139        let gsub = offset(Gsub::TAG);
140        let gpos = offset(Gpos::TAG);
141        let morx = offset(Morx::TAG);
142        let kerx = offset(Kerx::TAG);
143        let ankr = offset(Ankr::TAG);
144        let kern = offset(Kern::TAG);
145        let feat = offset(Feat::TAG);
146        let trak = offset(Trak::TAG);
147        Self {
148            num_glyphs,
149            units_per_em,
150            loca_long,
151            num_h_metrics,
152            num_v_metrics,
153            ascent,
154            descent,
155            loca,
156            glyf,
157            gvar,
158            hmtx,
159            hvar,
160            vmtx,
161            vvar,
162            vorg,
163            mvar,
164            cmap,
165            cmap_subtable,
166            cmap_vs_subtable,
167            gdef,
168            gsub,
169            gpos,
170            morx,
171            kerx,
172            ankr,
173            kern,
174            feat,
175            trak,
176        }
177    }
178}

180#[derive(Copy, Clone, Default, Debug)]
181pub struct TableOffset(u32);
182
183impl TableOffset {
184    fn new(font: &FontRef, tag: Tag) -> Option<Self> {
185        let records = font.table_directory().table_records();
186        records
187            .binary_search_by_key(&tag, |rec| rec.tag())
188            .ok()
189            .and_then(|ix| records.get(ix))
190            .map(|rec| Self(rec.offset()))
191    }
192
193    pub fn resolve(self) -> Option<usize> {
194        (self.0 != 0).then_some(self.0 as usize)
195    }
196
197    pub fn resolve_data<'a>(self, font: &FontRef<'a>) -> Option<FontData<'a>> {
198        font.data().slice(self.resolve()?..)
199    }
200
201    pub fn resolve_table<'a, T: FontRead<'a>>(self, font: &FontRef<'a>) -> Option<T> {
202        T::read(self.resolve_data(font)?).ok()
203    }
204}

206fn find_best_cmap_subtable<'a>(
207    cmap: &Cmap<'a>,
208) -> Option<(u16, PlatformId, u16, CmapSubtable<'a>)> {
209    // Symbol subtable.
210    // Prefer symbol if available.
211    // https://github.com/harfbuzz/harfbuzz/issues/1918
212    find_cmap_subtable(cmap, PlatformId::Windows, WINDOWS_SYMBOL_ENCODING)
213        // 32-bit subtables:
214        .or_else(|| find_cmap_subtable(cmap, PlatformId::Windows, WINDOWS_UNICODE_FULL_ENCODING))
215        .or_else(|| find_cmap_subtable(cmap, PlatformId::Unicode, UNICODE_FULL_ENCODING))
216        .or_else(|| find_cmap_subtable(cmap, PlatformId::Unicode, UNICODE_2_0_FULL_ENCODING))
217        // 16-bit subtables:
218        .or_else(|| find_cmap_subtable(cmap, PlatformId::Windows, WINDOWS_UNICODE_BMP_ENCODING))
219        .or_else(|| find_cmap_subtable(cmap, PlatformId::Unicode, UNICODE_2_0_BMP_ENCODING))
220        .or_else(|| find_cmap_subtable(cmap, PlatformId::Unicode, UNICODE_ISO_ENCODING))
221        .or_else(|| find_cmap_subtable(cmap, PlatformId::Unicode, UNICODE_1_1_ENCODING))
222        .or_else(|| find_cmap_subtable(cmap, PlatformId::Unicode, UNICODE_1_0_ENCODING))
223        // MacRoman subtable:
224        .or_else(|| find_cmap_subtable(cmap, PlatformId::Macintosh, 0))
225}
226
227fn find_cmap_subtable<'a>(
228    cmap: &Cmap<'a>,
229    platform_id: PlatformId,
230    encoding_id: u16,
231) -> Option<(u16, PlatformId, u16, CmapSubtable<'a>)> {
232    let offset_data = cmap.offset_data();
233    for (index, record) in cmap.encoding_records().iter().enumerate() {
234        if record.platform_id() != platform_id || record.encoding_id() != encoding_id {
235            continue;
236        }
237        if let Ok(subtable) = record.subtable(offset_data) {
238            match subtable {
239                CmapSubtable::Format0(_)
240                | CmapSubtable::Format4(_)
241                | CmapSubtable::Format6(_)
242                | CmapSubtable::Format10(_)
243                | CmapSubtable::Format12(_)
244                | CmapSubtable::Format13(_) => {
245                    return Some((index as u16, platform_id, encoding_id, subtable))
246                }
247                _ => {}
248            }
249        }
250    }
251    None
