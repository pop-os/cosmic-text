144    #[inline]
145    pub fn stage_lookup_range(&self, table_index: TableIndex, stage: usize) -> Range<usize> {
146        let stages = &self.stages[table_index];
147        let lookups = &self.lookups[table_index];
148        let start = stage
149            .checked_sub(1)
150            .map_or(0, |prev| stages[prev].last_lookup);
151        let end = stages
152            .get(stage)
153            .map_or(lookups.len(), |curr| curr.last_lookup);
154        start..end
155    }

162pub type hb_ot_map_feature_flags_t = u32;
163pub const F_NONE: u32 = 0x0000;
164pub const F_GLOBAL: u32 = 0x0001; /* Feature applies to all characters; results in no mask allocated for it. */
165pub const F_HAS_FALLBACK: u32 = 0x0002; /* Has fallback implementation, so include mask bit even if feature not found. */
166pub const F_MANUAL_ZWNJ: u32 = 0x0004; /* Don't skip over ZWNJ when matching **context**. */
167pub const F_MANUAL_ZWJ: u32 = 0x0008; /* Don't skip over ZWJ when matching **input**. */
168pub const F_MANUAL_JOINERS: u32 = F_MANUAL_ZWNJ | F_MANUAL_ZWJ;
169pub const F_GLOBAL_MANUAL_JOINERS: u32 = F_GLOBAL | F_MANUAL_JOINERS;
170pub const F_GLOBAL_HAS_FALLBACK: u32 = F_GLOBAL | F_HAS_FALLBACK;
171pub const F_GLOBAL_SEARCH: u32 = 0x0010; /* If feature not found in LangSys, look for it in global feature list and pick one. */
172pub const F_RANDOM: u32 = 0x0020; /* Randomly select a glyph from an AlternateSubstFormat1 subtable. */
173pub const F_PER_SYLLABLE: u32 = 0x0040; /* Contain lookup application to within syllable. */

210    pub fn new(
211        face: &'a hb_font_t<'a>,
212        script: Option<Script>,
213        language: Option<&Language>,
214    ) -> Self {
215        // Fetch script/language indices for GSUB/GPOS.  We need these later to skip
216        // features not available in either table and not waste precious bits for them.
217        let (script_tags, lang_tags) = tag::tags_from_script_and_language(script, language);
218
219        let mut found_script = [false; 2];
220        let mut script_index = [None; 2];
221        let mut chosen_script = [None; 2];
222        let mut lang_index = [None; 2];
223
224        for (table_index, table) in face.layout_tables() {
225            if let Some((found, idx, tag)) = table.select_script(&script_tags) {
226                chosen_script[table_index] = Some(tag);
227                found_script[table_index] = found;
228                script_index[table_index] = Some(idx);
229
230                if let Some(idx) = table.select_script_language(idx, &lang_tags) {
231                    lang_index[table_index] = Some(idx);
232                }
233            }
234        }
235
236        Self {
237            face,
238            found_script,
239            script_index,
240            chosen_script,
241            lang_index,
242            current_stage: [0, 0],
243            feature_infos: Vec::new(),
244            stages: [Vec::new(), Vec::new()],
245            is_simple: false,
246        }
247    }

256        for (table_index, table) in self.face.layout_tables() {
257            if let Some(script_index) = self.script_index[table_index] {
258                if table
259                    .find_language_feature(script_index, self.lang_index[table_index], tag)
260                    .is_some()
261                {
262                    return true;
263                }
264            }
265        }
266
267        false

314    pub fn compile(&mut self) -> hb_ot_map_t {
315        // We default to applying required feature in stage 0.  If the required
316        // feature has a tag that is known to the shaper, we apply required feature
317        // in the stage for that tag.
318        let mut required_index = [None; 2];
319        let mut required_tag = [None; 2];
320
321        for (table_index, table) in self.face.layout_tables() {
322            if let Some(script) = self.script_index[table_index] {
323                let lang = self.lang_index[table_index];
324                if let Some((idx, tag)) = table.get_required_language_feature(script, lang) {
325                    required_index[table_index] = Some(idx);
326                    required_tag[table_index] = Some(tag);
327                }
328            }
329        }
330
331        let (features, required_stage, global_mask) = self.collect_feature_maps(required_tag);
332
333        self.add_gsub_pause(None);
334        self.add_gpos_pause(None);
335
336        let (lookups, stages) =
337            self.collect_lookup_stages(&features, required_index, required_stage);
338
339        hb_ot_map_t {
340            found_script: self.found_script,
341            chosen_script: self.chosen_script,
342            global_mask,
343            features,
344            lookups,
345            stages,
346            feature_variations: self.face.ot_tables.feature_variations,
347        }
348    }

350    fn collect_feature_maps(
351        &mut self,
352        required_tag: [Option<hb_tag_t>; 2],
353    ) -> (Vec<feature_map_t>, [usize; 2], hb_mask_t) {
354        let mut map_features = Vec::new();
355        let mut required_stage = [0; 2];
356        let mut global_mask = GLOBAL_BIT_MASK;
357        let mut next_bit = glyph_flag::DEFINED.count_ones() + 1;
358
359        // Sort features and merge duplicates.
360        self.dedup_feature_infos();
361
362        for info in &self.feature_infos {
363            let bits_needed = if info.flags & F_GLOBAL != 0 && info.max_value == 1 {
364                // Uses the global bit.
365                0
366            } else {
367                // Limit bits per feature.
368                let v = info.max_value;
369                let num_bits = 8 * size_of_val(&v) as u32 - v.leading_zeros();
370                hb_ot_map_t::MAX_BITS.min(num_bits)
371            };
372
373            if info.max_value == 0 || next_bit + bits_needed >= GLOBAL_BIT_SHIFT {
374                // Feature disabled, or not enough bits.
375                continue;
376            }
377
378            let mut found = false;
379            let mut feature_index = [None; 2];
380
381            for (table_index, table) in self.face.layout_tables() {
382                if required_tag[table_index] == Some(info.tag) {
383                    required_stage[table_index] = info.stage[table_index];
384                }
385
386                if let Some(script) = self.script_index[table_index] {
387                    let lang = self.lang_index[table_index];
388                    if let Some(idx) = table.find_language_feature(script, lang, info.tag) {
389                        feature_index[table_index] = Some(idx);
390                        found = true;
391                    }
392                }
393            }
394
395            if !found && info.flags & F_GLOBAL_SEARCH != 0 {
396                // hb_ot_layout_table_find_feature
397                for (table_index, table) in self.face.layout_tables() {
398                    if let Some(idx) = table.feature_index(info.tag) {
399                        feature_index[table_index] = Some(idx);
400                        found = true;
401                    }
402                }
403            }
404
405            if !found && !info.flags & F_HAS_FALLBACK != 0 {
406                continue;
407            }
408
409            let (shift, mask) = if info.flags & F_GLOBAL != 0 && info.max_value == 1 {
410                // Uses the global bit
411                (GLOBAL_BIT_SHIFT, GLOBAL_BIT_MASK)
412            } else {
413                let shift = next_bit;
414                let mask = (1 << (next_bit + bits_needed)) - (1 << next_bit);
415                next_bit += bits_needed;
416                global_mask |= (info.default_value << shift) & mask;
417                (shift, mask)
418            };
419
420            map_features.push(feature_map_t {
421                tag: info.tag,
422                index: feature_index,
423                stage: info.stage,
424                shift,
425                mask,
426                one_mask: (1 << shift) & mask,
427                auto_zwnj: info.flags & F_MANUAL_ZWNJ == 0,
428                auto_zwj: info.flags & F_MANUAL_ZWJ == 0,
429                random: info.flags & F_RANDOM != 0,
430                per_syllable: info.flags & F_PER_SYLLABLE != 0,
431            });
432        }
433
434        if self.is_simple {
435            map_features.sort();
436        }
437
438        (map_features, required_stage, global_mask)
439    }

441    fn dedup_feature_infos(&mut self) {
442        let feature_infos = &mut self.feature_infos;
443        if feature_infos.is_empty() {
444            return;
445        }
446
447        if !self.is_simple {
448            feature_infos.sort();
449        }
450
451        let mut j = 0;
452        for i in 1..feature_infos.len() {
453            if feature_infos[i].tag != feature_infos[j].tag {
454                j += 1;
455                feature_infos[j] = feature_infos[i];
456            } else {
457                if feature_infos[i].flags & F_GLOBAL != 0 {
458                    feature_infos[j].flags |= F_GLOBAL;
459                    feature_infos[j].max_value = feature_infos[i].max_value;
460                    feature_infos[j].default_value = feature_infos[i].default_value;
461                } else {
462                    if feature_infos[j].flags & F_GLOBAL != 0 {
463                        feature_infos[j].flags ^= F_GLOBAL;
464                    }
465                    feature_infos[j].max_value =
466                        feature_infos[j].max_value.max(feature_infos[i].max_value);
467                    // Inherit default_value from j
468                }
469                let flags = feature_infos[i].flags & F_HAS_FALLBACK;
470                feature_infos[j].flags |= flags;
471                feature_infos[j].stage[0] =
472                    feature_infos[j].stage[0].min(feature_infos[i].stage[0]);
473                feature_infos[j].stage[1] =
474                    feature_infos[j].stage[1].min(feature_infos[i].stage[1]);
475            }
476        }
477
478        feature_infos.truncate(j + 1);
479    }

481    fn collect_lookup_stages(
482        &self,
483        map_features: &[feature_map_t],
484        required_feature_index: [Option<u16>; 2],
485        required_feature_stage: [usize; 2],
486    ) -> ([Vec<lookup_map_t>; 2], [Vec<StageMap>; 2]) {
487        let mut map_lookups = [Vec::new(), Vec::new()];
488        let mut map_stages = [Vec::new(), Vec::new()];
489
490        for table_index in TableIndex::iter() {
491            // Collect lookup indices for features.
492            let mut stage_index = 0;
493            let mut last_lookup = 0;
494
495            let variation_index = self.face.ot_tables.feature_variations[table_index as usize];
496
497            for stage in 0..self.current_stage[table_index] {
498                if let Some(feature_index) = required_feature_index[table_index] {
499                    if required_feature_stage[table_index] == stage {
500                        self.add_lookups(
501                            &mut map_lookups[table_index],
502                            table_index,
503                            feature_index,
504                            variation_index,
505                            GLOBAL_BIT_MASK,
506                            true,
507                            true,
508                            false,
509                            false,
510                        );
511                    }
512                }
513
514                for feature in map_features {
515                    if let Some(feature_index) = feature.index[table_index] {
516                        if feature.stage[table_index] == stage {
517                            self.add_lookups(
518                                &mut map_lookups[table_index],
519                                table_index,
520                                feature_index,
521                                variation_index,
522                                feature.mask,
523                                feature.auto_zwnj,
524                                feature.auto_zwj,
525                                feature.random,
526                                feature.per_syllable,
527                            );
528                        }
529                    }
530                }
531
532                // Sort lookups and merge duplicates.
533                let lookups = &mut map_lookups[table_index];
534                let len = lookups.len();
535
536                if last_lookup + 1 < len {
537                    lookups[last_lookup..].sort();
538
539                    let mut j = last_lookup;
540                    for i in j + 1..len {
541                        if lookups[i].index != lookups[j].index {
542                            j += 1;
543                            lookups[j] = lookups[i];
544                        } else {
545                            lookups[j].mask |= lookups[i].mask;
546                            lookups[j].auto_zwnj &= lookups[i].auto_zwnj;
547                            lookups[j].auto_zwj &= lookups[i].auto_zwj;
548                        }
549                    }
550
551                    lookups.truncate(j + 1);
552                }
553
554                last_lookup = lookups.len();
555
556                if let Some(info) = self.stages[table_index].get(stage_index) {
557                    if info.index == stage {
558                        map_stages[table_index].push(StageMap {
559                            last_lookup,
560                            pause_func: info.pause_func,
561                        });
562
563                        stage_index += 1;
564                    }
565                }
566            }
567        }
568
569        (map_lookups, map_stages)
570    }

572    fn add_lookups(
573        &self,
574        lookups: &mut Vec<lookup_map_t>,
575        table_index: TableIndex,
576        feature_index: u16,
577        variation_index: Option<u32>,
578        mask: hb_mask_t,
579        auto_zwnj: bool,
580        auto_zwj: bool,
581        random: bool,
582        per_syllable: bool,
583    ) -> Option<()> {
584        let table = self.face.layout_table(table_index)?;
585
586        let lookup_count = table.lookup_count();
587        let feature = match variation_index {
588            Some(idx) => table
589                .feature_substitution(idx, feature_index)
590                .or_else(|| table.feature(feature_index))?,
591            None => table.feature(feature_index)?,
592        };
593
594        for index in feature.lookup_list_indices() {
595            let index = index.get();
596            if index < lookup_count {
597                lookups.push(lookup_map_t {
598                    mask,
599                    index,
600                    auto_zwnj,
601                    auto_zwj,
602                    random,
603                    per_syllable,
604                });
605            }
606        }
607
608        Some(())
609    }
