184fn collect_features(planner: &mut hb_ot_shape_planner_t) {
185    // Do this before any lookups have been applied.
186    planner.ot_map.add_gsub_pause(Some(setup_syllables));
187
188    // Default glyph pre-processing group
189    planner
190        .ot_map
191        .enable_feature(hb_tag_t::new(b"locl"), F_PER_SYLLABLE, 1);
192    planner
193        .ot_map
194        .enable_feature(hb_tag_t::new(b"ccmp"), F_PER_SYLLABLE, 1);
195    planner
196        .ot_map
197        .enable_feature(hb_tag_t::new(b"nukt"), F_PER_SYLLABLE, 1);
198    planner
199        .ot_map
200        .enable_feature(hb_tag_t::new(b"akhn"), F_MANUAL_ZWJ | F_PER_SYLLABLE, 1);
201
202    // Reordering group
203    planner
204        .ot_map
205        .add_gsub_pause(Some(_hb_clear_substitution_flags));
206    planner
207        .ot_map
208        .add_feature(hb_tag_t::new(b"rphf"), F_MANUAL_ZWJ | F_PER_SYLLABLE, 1);
209    planner.ot_map.add_gsub_pause(Some(record_rphf));
210    planner
211        .ot_map
212        .add_gsub_pause(Some(_hb_clear_substitution_flags));
213    planner
214        .ot_map
215        .enable_feature(hb_tag_t::new(b"pref"), F_MANUAL_ZWJ | F_PER_SYLLABLE, 1);
216    planner.ot_map.add_gsub_pause(Some(record_pref));
217
218    // Orthographic unit shaping group
219    for feature in BASIC_FEATURES {
220        planner
221            .ot_map
222            .enable_feature(*feature, F_MANUAL_ZWJ | F_PER_SYLLABLE, 1);
223    }
224
225    planner.ot_map.add_gsub_pause(Some(reorder_use));
226    planner.ot_map.add_gsub_pause(Some(syllabic_clear_var)); // Don't need syllables anymore.
227
228    // Topographical features
229    for feature in TOPOGRAPHICAL_FEATURES {
230        planner.ot_map.add_feature(*feature, F_NONE, 1);
231    }
232    planner.ot_map.add_gsub_pause(None);
233
234    // Standard typographic presentation
235    for feature in OTHER_FEATURES {
236        planner.ot_map.enable_feature(*feature, F_MANUAL_ZWJ, 1);
237    }
238}

240fn setup_syllables(plan: &hb_ot_shape_plan_t, _: &hb_font_t, buffer: &mut hb_buffer_t) -> bool {
241    buffer.allocate_var(GlyphInfo::SYLLABLE_VAR);
242
243    super::ot_shaper_use_machine::find_syllables(buffer);
244
245    foreach_syllable!(buffer, start, end, {
246        buffer.unsafe_to_break(Some(start), Some(end));
247    });
248
249    setup_rphf_mask(plan, buffer);
250    setup_topographical_masks(plan, buffer);
251
252    false
253}

255fn setup_rphf_mask(plan: &hb_ot_shape_plan_t, buffer: &mut hb_buffer_t) -> bool {
256    let universal_plan = plan.data::<UniversalShapePlan>();
257
258    let mask = universal_plan.rphf_mask;
259    if mask == 0 {
260        return false;
261    }
262
263    let mut start = 0;
264    let mut end = buffer.next_syllable(0);
265    while start < buffer.len {
266        let limit = if buffer.info[start].use_category() == category::R {
267            1
268        } else {
269            core::cmp::min(3, end - start)
270        };
271
272        for i in start..start + limit {
273            buffer.info[i].mask |= mask;
274        }
275
276        start = end;
277        end = buffer.next_syllable(start);
278    }
279
280    false
281}

283fn setup_topographical_masks(plan: &hb_ot_shape_plan_t, buffer: &mut hb_buffer_t) {
284    use super::ot_shaper_use_machine::SyllableType;
285
286    if plan.data::<UniversalShapePlan>().arabic_plan.is_some() {
287        return;
288    }
289
290    let mut masks = [0; 4];
291    let mut all_masks = 0;
292    for i in 0..4 {
293        masks[i] = plan.ot_map.get_1_mask(TOPOGRAPHICAL_FEATURES[i]);
294        if masks[i] == plan.ot_map.get_global_mask() {
295            masks[i] = 0;
296        }
297
298        all_masks |= masks[i];
299    }
300
301    if all_masks == 0 {
302        return;
303    }
304
305    let other_masks = !all_masks;
306
307    let mut last_start = 0;
308    let mut last_form = None;
309    let mut start = 0;
310    let mut end = buffer.next_syllable(0);
311    while start < buffer.len {
312        let syllable = buffer.info[start].syllable() & 0x0F;
313        if syllable == SyllableType::HieroglyphCluster as u8
314            || syllable == SyllableType::NonCluster as u8
315        {
316            last_form = None;
317        } else {
318            let join = last_form == Some(JoiningForm::Terminal)
319                || last_form == Some(JoiningForm::Isolated);
320
321            if join {
322                // Fixup previous syllable's form.
323                let form = if last_form == Some(JoiningForm::Terminal) {
324                    JoiningForm::Medial
325                } else {
326                    JoiningForm::Initial
327                };
328
329                for i in last_start..start {
330                    buffer.info[i].mask =
331                        (buffer.info[i].mask & other_masks) | masks[form as usize];
332                }
333            }
334
335            // Form for this syllable.
336            let form = if join {
337                JoiningForm::Terminal
338            } else {
339                JoiningForm::Isolated
340            };
341            last_form = Some(form);
342            for i in start..end {
343                buffer.info[i].mask = (buffer.info[i].mask & other_masks) | masks[form as usize];
344            }
345        }
346
347        last_start = start;
348        start = end;
349        end = buffer.next_syllable(start);
350    }
351}

353fn record_rphf(plan: &hb_ot_shape_plan_t, _: &hb_font_t, buffer: &mut hb_buffer_t) -> bool {
354    let universal_plan = plan.data::<UniversalShapePlan>();
355
356    let mask = universal_plan.rphf_mask;
357    if mask == 0 {
358        return false;
359    }
360
361    let mut start = 0;
362    let mut end = buffer.next_syllable(0);
363    while start < buffer.len {
364        // Mark a substituted repha as USE_R.
365        for i in start..end {
366            if buffer.info[i].mask & mask == 0 {
367                break;
368            }
369
370            if buffer.info[i].substituted() {
371                buffer.info[i].set_use_category(category::R);
372                break;
373            }
374        }
375
376        start = end;
377        end = buffer.next_syllable(start);
378    }
379
380    false
381}

383fn reorder_use(_: &hb_ot_shape_plan_t, face: &hb_font_t, buffer: &mut hb_buffer_t) -> bool {
384    use super::ot_shaper_use_machine::SyllableType;
385
386    let mut ret = false;
387
388    if insert_dotted_circles(
389        face,
390        buffer,
391        SyllableType::BrokenCluster as u8,
392        category::B,
393        Some(category::R),
394        None,
395    ) {
396        ret = true;
397    }
398
399    let mut start = 0;
400    let mut end = buffer.next_syllable(0);
401    while start < buffer.len {
402        reorder_syllable_use(start, end, buffer);
403        start = end;
404        end = buffer.next_syllable(start);
405    }
406
407    buffer.deallocate_var(GlyphInfo::USE_CATEGORY_VAR);
408
409    ret
410}

439fn reorder_syllable_use(start: usize, end: usize, buffer: &mut hb_buffer_t) {
440    use super::ot_shaper_use_machine::SyllableType;
441
442    let syllable_type = (buffer.info[start].syllable() & 0x0F) as u32;
443    // Only a few syllable types need reordering.
444    if (rb_flag_unsafe(syllable_type)
445        & (rb_flag(SyllableType::ViramaTerminatedCluster as u32)
446            | rb_flag(SyllableType::SakotTerminatedCluster as u32)
447            | rb_flag(SyllableType::StandardCluster as u32)
448            | rb_flag(SyllableType::BrokenCluster as u32)
449            | 0))
450        == 0
451    {
452        return;
453    }
454
455    // Move things forward.
456    if buffer.info[start].use_category() == category::R && end - start > 1 {
457        // Got a repha.  Reorder it towards the end, but before the first post-base glyph.
458        for i in start + 1..end {
459            let is_post_base_glyph =
460                (rb_flag64_unsafe(buffer.info[i].use_category() as u32) & POST_BASE_FLAGS) != 0
461                    || buffer.info[i].is_halant_use();
462
463            if is_post_base_glyph || i == end - 1 {
464                // If we hit a post-base glyph, move before it; otherwise move to the
465                // end. Shift things in between backward.
466
467                let mut i = i;
468                if is_post_base_glyph {
469                    i -= 1;
470                }
471
472                buffer.merge_clusters(start, i + 1);
473                let t = buffer.info[start];
474                for k in 0..i - start {
475                    buffer.info[k + start] = buffer.info[k + start + 1];
476                }
477                buffer.info[i] = t;
478
479                break;
480            }
481        }
482    }
483
484    // Move things back.
485    let mut j = start;
486    for i in start..end {
487        let flag = rb_flag_unsafe(buffer.info[i].use_category() as u32);
488        if buffer.info[i].is_halant_use() {
489            // If we hit a halant, move after it; otherwise move to the beginning, and
490            // shift things in between forward.
491            j = i + 1;
492        } else if (flag & (category_flag(category::VPre) | category_flag(category::VMPre))) != 0
493            && buffer.info[i].lig_comp() == 0
494            && j < i
495        {
496            // Only move the first component of a MultipleSubst.
497            buffer.merge_clusters(j, i + 1);
498            let t = buffer.info[i];
499            for k in (0..i - j).rev() {
500                buffer.info[k + j + 1] = buffer.info[k + j];
501            }
502            buffer.info[j] = t;
503        }
504    }
505}

507fn record_pref(_: &hb_ot_shape_plan_t, _: &hb_font_t, buffer: &mut hb_buffer_t) -> bool {
508    let mut start = 0;
509    let mut end = buffer.next_syllable(0);
510    while start < buffer.len {
511        // Mark a substituted pref as VPre, as they behave the same way.
512        for i in start..end {
513            if buffer.info[i].substituted() {
514                buffer.info[i].set_use_category(category::VPre);
515                break;
516            }
517        }
518
519        start = end;
520        end = buffer.next_syllable(start);
521    }
522
523    false
524}

550fn compose(_: &hb_ot_shape_normalize_context_t, a: Codepoint, b: Codepoint) -> Option<Codepoint> {
551    // Avoid recomposing split matras.
552    if a.general_category().is_mark() {
553        return None;
554    }
555
556    crate::hb::unicode::compose(a, b)
557}

559fn setup_masks(plan: &hb_ot_shape_plan_t, _: &hb_font_t, buffer: &mut hb_buffer_t) {
560    let universal_plan = plan.data::<UniversalShapePlan>();
561
562    // Do this before allocating use_category().
563    if let Some(ref arabic_plan) = universal_plan.arabic_plan {
564        crate::hb::ot_shaper_arabic::setup_masks_inner(arabic_plan, plan.script, buffer);
565    }
566
567    buffer.allocate_var(GlyphInfo::USE_CATEGORY_VAR);
568
569    // We cannot setup masks here. We save information about characters
570    // and setup masks later on in a pause-callback.
571    for info in buffer.info_slice_mut() {
572        info.set_use_category(super::ot_shaper_use_table::hb_use_get_category(
573            info.glyph_id as usize,
574        ));
575    }
