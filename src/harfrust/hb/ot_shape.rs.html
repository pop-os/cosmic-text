39impl<'a> hb_ot_shape_planner_t<'a> {
40    pub fn new(
41        face: &'a hb_font_t<'a>,
42        direction: Direction,
43        script: Option<Script>,
44        language: Option<&Language>,
45    ) -> Self {
46        let ot_map = hb_ot_map_builder_t::new(face, script, language);
47        let aat_map = AatMapBuilder::default();
48
49        let mut shaper = match script {
50            Some(script) => hb_ot_shape_complex_categorize(
51                script,
52                direction,
53                ot_map.chosen_script(TableIndex::GSUB),
54            ),
55            None => &DEFAULT_SHAPER,
56        };
57
58        let script_zero_marks = shaper.zero_width_marks != HB_OT_SHAPE_ZERO_WIDTH_MARKS_NONE;
59        let script_fallback_position = shaper.fallback_position;
60
61        // https://github.com/harfbuzz/harfbuzz/issues/2124
62        let apply_morx = face.aat_tables.morx.is_some()
63            && (direction.is_horizontal() || face.ot_tables.gsub.is_none());
64
65        // https://github.com/harfbuzz/harfbuzz/issues/1528
66        if apply_morx && !ptr::eq(ptr::from_ref(shaper), ptr::from_ref(&DEFAULT_SHAPER)) {
67            shaper = &DUMBER_SHAPER;
68        }
69
70        hb_ot_shape_planner_t {
71            face,
72            direction,
73            script,
74            language: language.cloned(),
75            ot_map,
76            aat_map,
77            apply_morx,
78            script_zero_marks,
79            script_fallback_position,
80            shaper,
81        }
82    }

84    pub fn collect_features(&mut self, user_features: &[Feature]) {
85        static COMMON_FEATURES: &[(hb_tag_t, hb_ot_map_feature_flags_t)] = &[
86            (hb_tag_t::new(b"abvm"), F_GLOBAL),
87            (hb_tag_t::new(b"blwm"), F_GLOBAL),
88            (hb_tag_t::new(b"ccmp"), F_GLOBAL),
89            (hb_tag_t::new(b"locl"), F_GLOBAL),
90            (hb_tag_t::new(b"mark"), F_GLOBAL_MANUAL_JOINERS),
91            (hb_tag_t::new(b"mkmk"), F_GLOBAL_MANUAL_JOINERS),
92            (hb_tag_t::new(b"rlig"), F_GLOBAL),
93        ];
94
95        static HORIZONTAL_FEATURES: &[(hb_tag_t, hb_ot_map_feature_flags_t)] = &[
96            (hb_tag_t::new(b"calt"), F_GLOBAL),
97            (hb_tag_t::new(b"clig"), F_GLOBAL),
98            (hb_tag_t::new(b"curs"), F_GLOBAL),
99            (hb_tag_t::new(b"dist"), F_GLOBAL),
100            (hb_tag_t::new(b"kern"), F_GLOBAL_HAS_FALLBACK),
101            (hb_tag_t::new(b"liga"), F_GLOBAL),
102            (hb_tag_t::new(b"rclt"), F_GLOBAL),
103        ];
104
105        let empty = F_NONE;
106
107        self.ot_map.is_simple = true;
108
109        self.ot_map.enable_feature(hb_tag_t::new(b"rvrn"), empty, 1);
110        self.ot_map.add_gsub_pause(None);
111
112        match self.direction {
113            Direction::LeftToRight => {
114                self.ot_map.enable_feature(hb_tag_t::new(b"ltra"), empty, 1);
115                self.ot_map.enable_feature(hb_tag_t::new(b"ltrm"), empty, 1);
116            }
117            Direction::RightToLeft => {
118                self.ot_map.enable_feature(hb_tag_t::new(b"rtla"), empty, 1);
119                self.ot_map.add_feature(hb_tag_t::new(b"rtlm"), empty, 1);
120            }
121            _ => {}
122        }
123
124        // Automatic fractions.
125        self.ot_map.add_feature(hb_tag_t::new(b"frac"), empty, 1);
126        self.ot_map.add_feature(hb_tag_t::new(b"numr"), empty, 1);
127        self.ot_map.add_feature(hb_tag_t::new(b"dnom"), empty, 1);
128
129        // Random!
130        self.ot_map
131            .enable_feature(hb_tag_t::new(b"rand"), F_RANDOM, hb_ot_map_t::MAX_VALUE);
132
133        // Tracking.  We enable dummy feature here just to allow disabling
134        // AAT 'trak' table using features.
135        // https://github.com/harfbuzz/harfbuzz/issues/1303
136        self.ot_map
137            .enable_feature(hb_tag_t::new(b"trak"), F_HAS_FALLBACK, 1);
138
139        self.ot_map.enable_feature(hb_tag_t::new(b"Harf"), empty, 1); // Considered required.
140        self.ot_map.enable_feature(hb_tag_t::new(b"HARF"), empty, 1); // Considered discretionary.
141
142        if let Some(func) = self.shaper.collect_features {
143            self.ot_map.is_simple = false;
144            func(self);
145        }
146
147        self.ot_map.enable_feature(hb_tag_t::new(b"Buzz"), empty, 1); // Considered required.
148        self.ot_map.enable_feature(hb_tag_t::new(b"BUZZ"), empty, 1); // Considered discretionary.
149
150        for &(tag, flags) in COMMON_FEATURES {
151            self.ot_map.add_feature(tag, flags, 1);
152        }
153
154        if self.direction.is_horizontal() {
155            for &(tag, flags) in HORIZONTAL_FEATURES {
156                self.ot_map.add_feature(tag, flags, 1);
157            }
158        } else {
159            // We only apply `vert` feature. See:
160            // https://github.com/harfbuzz/harfbuzz/commit/d71c0df2d17f4590d5611239577a6cb532c26528
161            // https://lists.freedesktop.org/archives/harfbuzz/2013-August/003490.html
162
163            // We really want to find a 'vert' feature if there's any in the font, no
164            // matter which script/langsys it is listed (or not) under.
165            // See various bugs referenced from:
166            // https://github.com/harfbuzz/harfbuzz/issues/63
167            self.ot_map
168                .enable_feature(hb_tag_t::new(b"vert"), F_GLOBAL_SEARCH, 1);
169        }
170
171        if !user_features.is_empty() {
172            self.ot_map.is_simple = false;
173        }
174
175        for feature in user_features {
176            let flags = if feature.is_global() { F_GLOBAL } else { empty };
177            self.ot_map.add_feature(feature.tag, flags, feature.value);
178        }
179
180        if let Some(func) = self.shaper.override_features {
181            func(self);
182        }
183    }

185    pub fn compile(mut self, features: &[Feature]) -> hb_ot_shape_plan_t {
186        let ot_map = self.ot_map.compile();
187        let mut aat_map = AatMap::default();
188        if self.apply_morx {
189            self.aat_map.compile(self.face, &mut aat_map);
190        }
191
192        let frac_mask = ot_map.get_1_mask(hb_tag_t::new(b"frac"));
193        let numr_mask = ot_map.get_1_mask(hb_tag_t::new(b"numr"));
194        let dnom_mask = ot_map.get_1_mask(hb_tag_t::new(b"dnom"));
195        let has_frac = frac_mask != 0 || (numr_mask != 0 && dnom_mask != 0);
196
197        let rtlm_mask = ot_map.get_1_mask(hb_tag_t::new(b"rtlm"));
198        let has_vert = ot_map.get_1_mask(hb_tag_t::new(b"vert")) != 0;
199
200        let horizontal = self.direction.is_horizontal();
201        let kern_tag = if horizontal {
202            hb_tag_t::new(b"kern")
203        } else {
204            hb_tag_t::new(b"vkrn")
205        };
206        let kern_mask = ot_map.get_mask(kern_tag).0;
207        let requested_kerning = kern_mask != 0;
208
209        let has_gpos_kern = ot_map
210            .get_feature_index(TableIndex::GPOS, kern_tag)
211            .is_some();
212        let disable_gpos = self.shaper.gpos_tag.is_some()
213            && self.shaper.gpos_tag != ot_map.chosen_script(TableIndex::GPOS);
214
215        // Decide who provides glyph classes. GDEF or Unicode.
216        let fallback_glyph_classes = !hb_ot_layout_has_glyph_classes(self.face);
217
218        // Decide who does substitutions. GSUB, morx, or fallback.
219        let apply_morx = self.apply_morx;
220
221        let mut apply_gpos = false;
222        let mut apply_kerx = false;
223        let mut apply_kern = false;
224
225        // Decide who does positioning. GPOS, kerx, kern, or fallback.
226        let has_kerx = self.face.aat_tables.kerx.is_some();
227        let has_gsub = !apply_morx && self.face.ot_tables.gsub.is_some();
228        let has_gpos = !disable_gpos && self.face.ot_tables.gpos.is_some();
229
230        // Prefer GPOS over kerx if GSUB is present;
231        // https://github.com/harfbuzz/harfbuzz/issues/3008
232        if has_kerx && !(has_gsub && has_gpos) {
233            apply_kerx = true;
234        } else if has_gpos {
235            apply_gpos = true;
236        }
237
238        if !apply_kerx && (!has_gpos_kern || !apply_gpos) {
239            if has_kerx {
240                apply_kerx = true;
241            } else if hb_ot_layout_has_kerning(self.face) {
242                apply_kern = self.script_fallback_position;
243            }
244        }
245
246        let apply_fallback_kern = !(apply_gpos || apply_kerx || apply_kern);
247        let zero_marks = self.script_zero_marks
248            && !apply_kerx
249            && (!apply_kern || !hb_ot_layout_has_machine_kerning(self.face));
250
251        let has_gpos_mark = ot_map.get_1_mask(hb_tag_t::new(b"mark")) != 0;
252
253        let mut adjust_mark_positioning_when_zeroing = !apply_gpos
254            && !apply_kerx
255            && (!apply_kern || !hb_ot_layout_has_cross_kerning(self.face));
256
257        let fallback_mark_positioning =
258            adjust_mark_positioning_when_zeroing && self.script_fallback_position;
259
260        // If we're using morx shaping, we cancel mark position adjustment because
261        // Apple Color Emoji assumes this will NOT be done when forming emoji sequences;
262        // https://github.com/harfbuzz/harfbuzz/issues/2967.
263        if apply_morx {
264            adjust_mark_positioning_when_zeroing = false;
265        }
266
267        // According to Ned, trak is applied by default for "modern fonts", as detected by presence of STAT table.
268        // https://github.com/googlefonts/fontations/issues/1492
269        let apply_trak = self.face.font.trak().is_ok()
270            && self
271                .face
272                .font
273                .table_directory
274                .table_records()
275                .iter()
276                .any(|table| table.tag() == "STAT");
277
278        let mut plan = hb_ot_shape_plan_t {
279            direction: self.direction,
280            script: self.script,
281            language: self.language,
282            shaper: self.shaper,
283            ot_map,
284            aat_map,
285            data: None,
286            frac_mask,
287            numr_mask,
288            dnom_mask,
289            rtlm_mask,
290            kern_mask,
291            requested_kerning,
292            has_frac,
293            has_vert,
294            has_gpos_mark,
295            zero_marks,
296            fallback_glyph_classes,
297            fallback_mark_positioning,
298            adjust_mark_positioning_when_zeroing,
299            apply_gpos,
300            apply_kern,
301            apply_fallback_kern,
302            apply_kerx,
303            apply_morx,
304            apply_trak,
305            user_features: features.into(),
306        };
307
308        if let Some(func) = self.shaper.create_data {
309            plan.data = Some(func(&plan));
310        }
311
312        plan
313    }
314}

316pub struct hb_ot_shape_context_t<'a> {
317    pub plan: &'a hb_ot_shape_plan_t,
318    pub face: &'a hb_font_t<'a>,
319    pub buffer: &'a mut hb_buffer_t,
320    // Transient stuff
321    pub target_direction: Direction,
322    pub features: &'a [Feature],
323}

325// Pull it all together!
326pub fn shape_internal(ctx: &mut hb_ot_shape_context_t) {
327    ctx.buffer.allocate_unicode_vars();
328
329    initialize_masks(ctx);
330    set_unicode_props(ctx.buffer);
331    insert_dotted_circle(ctx.buffer, ctx.face);
332
333    form_clusters(ctx.buffer);
334
335    ensure_native_direction(ctx.buffer);
336
337    if let Some(func) = ctx.plan.shaper.preprocess_text {
338        func(ctx.plan, ctx.face, ctx.buffer);
339    }
340
341    substitute_pre(ctx);
342    position(ctx);
343    substitute_post(ctx);
344
345    propagate_flags(ctx.buffer);
346
347    ctx.buffer.deallocate_unicode_vars();
348
349    ctx.buffer.direction = ctx.target_direction;
350}
351
352fn substitute_pre(ctx: &mut hb_ot_shape_context_t) {
353    hb_ot_substitute_default(ctx);
354
355    ctx.buffer.allocate_gsubgpos_vars();
356
357    hb_ot_substitute_plan(ctx);
358
359    if ctx.plan.apply_morx && ctx.plan.apply_gpos {
360        aat::layout::remove_deleted_glyphs(ctx.buffer);
361    }
362}
363
364fn substitute_post(ctx: &mut hb_ot_shape_context_t) {
365    if ctx.plan.apply_morx && !ctx.plan.apply_gpos {
366        aat::layout::remove_deleted_glyphs(ctx.buffer);
367    }
368
369    deal_with_variation_selectors(ctx.buffer);
370    hide_default_ignorables(ctx.buffer, ctx.face);
371
372    if let Some(func) = ctx.plan.shaper.postprocess_glyphs {
373        func(ctx.plan, ctx.face, ctx.buffer);
374    }
375}

377fn hb_ot_substitute_default(ctx: &mut hb_ot_shape_context_t) {
378    rotate_chars(ctx);
379
380    ctx.buffer
381        .allocate_var(GlyphInfo::NORMALIZER_GLYPH_INDEX_VAR);
382
383    ot_shape_normalize::_hb_ot_shape_normalize(ctx.plan, ctx.buffer, ctx.face);
384
385    setup_masks(ctx);
386
387    // This is unfortunate to go here, but necessary...
388    if ctx.plan.fallback_mark_positioning {
389        ot_shape_fallback::_hb_ot_shape_fallback_mark_position_recategorize_marks(
390            ctx.plan, ctx.face, ctx.buffer,
391        );
392    }
393
394    map_glyphs_fast(ctx.buffer);
395
396    ctx.buffer
397        .deallocate_var(GlyphInfo::NORMALIZER_GLYPH_INDEX_VAR);
398}

400fn hb_ot_substitute_plan(ctx: &mut hb_ot_shape_context_t) {
401    hb_ot_layout_substitute_start(ctx.face, ctx.buffer);
402
403    if ctx.plan.fallback_glyph_classes {
404        hb_synthesize_glyph_classes(ctx.buffer);
405    }
406
407    if ctx.plan.apply_morx {
408        aat::layout::substitute(ctx.plan, ctx.face, ctx.buffer, ctx.features);
409        ctx.buffer.update_digest();
410    } else {
411        ctx.buffer.update_digest();
412        ot_layout_gsub_table::substitute(ctx.plan, ctx.face, ctx.buffer);
413    }
414}

416fn position(ctx: &mut hb_ot_shape_context_t) {
417    ctx.buffer.clear_positions();
418
419    position_default(ctx);
420
421    position_complex(ctx);
422
423    if ctx.buffer.direction.is_backward() {
424        ctx.buffer.reverse();
425    }
426
427    ctx.buffer.deallocate_gsubgpos_vars();
428}

430fn position_default(ctx: &mut hb_ot_shape_context_t) {
431    let len = ctx.buffer.len;
432
433    if ctx.buffer.direction.is_horizontal() {
434        ctx.face.glyph_h_advances(ctx.buffer);
435    } else {
436        for (info, pos) in ctx.buffer.info[..len]
437            .iter()
438            .zip(&mut ctx.buffer.pos[..len])
439        {
440            let glyph = info.as_glyph();
441            pos.y_advance = ctx.face.glyph_v_advance(glyph);
442            pos.x_offset -= ctx.face.glyph_h_origin(glyph);
443            pos.y_offset -= ctx.face.glyph_v_origin(glyph);
444        }
445    }
446
447    if ctx.buffer.scratch_flags & HB_BUFFER_SCRATCH_FLAG_HAS_SPACE_FALLBACK != 0 {
448        ot_shape_fallback::_hb_ot_shape_fallback_spaces(ctx.plan, ctx.face, ctx.buffer);
449    }
450}

452fn position_complex(ctx: &mut hb_ot_shape_context_t) {
453    // If the font has no GPOS and direction is forward, then when
454    // zeroing mark widths, we shift the mark with it, such that the
455    // mark is positioned hanging over the previous glyph.  When
456    // direction is backward we don't shift and it will end up
457    // hanging over the next glyph after the final reordering.
458    //
459    // Note: If fallback positioning happens, we don't care about
460    // this as it will be overridden.
461    let adjust_offsets_when_zeroing =
462        ctx.plan.adjust_mark_positioning_when_zeroing && ctx.buffer.direction.is_forward();
463
464    // We change glyph origin to what GPOS expects (horizontal), apply GPOS, change it back.
465
466    GPOS::position_start(ctx.face, ctx.buffer);
467
468    if ctx.plan.zero_marks
469        && ctx.plan.shaper.zero_width_marks == HB_OT_SHAPE_ZERO_WIDTH_MARKS_BY_GDEF_EARLY
470    {
471        zero_mark_widths_by_gdef(ctx.buffer, adjust_offsets_when_zeroing);
472    }
473
474    position_by_plan(ctx.plan, ctx.face, ctx.buffer);
475
476    if ctx.plan.zero_marks
477        && ctx.plan.shaper.zero_width_marks == HB_OT_SHAPE_ZERO_WIDTH_MARKS_BY_GDEF_LATE
478    {
479        zero_mark_widths_by_gdef(ctx.buffer, adjust_offsets_when_zeroing);
480    }
481
482    // Finish off.  Has to follow a certain order.
483    GPOS::position_finish_advances(ctx.face, ctx.buffer);
484    zero_width_default_ignorables(ctx.buffer);
485    GPOS::position_finish_offsets(ctx.face, ctx.buffer);
486
487    if ctx.plan.fallback_mark_positioning {
488        ot_shape_fallback::position_marks(
489            ctx.plan,
490            ctx.face,
491            ctx.buffer,
492            adjust_offsets_when_zeroing,
493        );
494    }
495}

497fn position_by_plan(plan: &hb_ot_shape_plan_t, face: &hb_font_t, buffer: &mut hb_buffer_t) {
498    if plan.apply_gpos {
499        ot_layout_gpos_table::position(plan, face, buffer);
500    } else if plan.apply_kerx {
501        aat::layout::position(plan, face, buffer);
502    }
503    if plan.apply_kern {
504        kerning::hb_ot_layout_kern(plan, face, buffer);
505    } else if plan.apply_fallback_kern {
506        ot_shape_fallback::_hb_ot_shape_fallback_kern(plan, face, buffer);
507    }
508
509    if plan.apply_trak {
510        aat::layout::track(plan, face, buffer);
511    }
512}

514fn initialize_masks(ctx: &mut hb_ot_shape_context_t) {
515    let global_mask = ctx.plan.ot_map.get_global_mask();
516    ctx.buffer.reset_masks(global_mask);
517}
518
519fn setup_masks(ctx: &mut hb_ot_shape_context_t) {
520    setup_masks_fraction(ctx);
521
522    if let Some(func) = ctx.plan.shaper.setup_masks {
523        func(ctx.plan, ctx.face, ctx.buffer);
524    }
525
526    for feature in ctx.features {
527        if !feature.is_global() {
528            let (mask, shift) = ctx.plan.ot_map.get_mask(feature.tag);
529            ctx.buffer
530                .set_masks(feature.value << shift, mask, feature.start, feature.end);
531        }
532    }
533}
534
535fn setup_masks_fraction(ctx: &mut hb_ot_shape_context_t) {
536    let buffer = &mut ctx.buffer;
537    if buffer.scratch_flags & HB_BUFFER_SCRATCH_FLAG_HAS_FRACTION_SLASH == 0 || !ctx.plan.has_frac {
538        return;
539    }
540
541    let (pre_mask, post_mask) = if buffer.direction.is_forward() {
542        (
543            ctx.plan.numr_mask | ctx.plan.frac_mask,
544            ctx.plan.frac_mask | ctx.plan.dnom_mask,
545        )
546    } else {
547        (
548            ctx.plan.frac_mask | ctx.plan.dnom_mask,
549            ctx.plan.numr_mask | ctx.plan.frac_mask,
550        )
551    };
552
553    let len = buffer.len;
554    let mut i = 0;
555    while i < len {
556        // FRACTION SLASH
557        if buffer.info[i].glyph_id == 0x2044 {
558            let mut start = i;
559            while start > 0
560                && buffer.info[start - 1].general_category() == GeneralCategory::DECIMAL_NUMBER
561            {
562                start -= 1;
563            }
564
565            let mut end = i + 1;
566            while end < len
567                && buffer.info[end].general_category() == GeneralCategory::DECIMAL_NUMBER
568            {
569                end += 1;
570            }
571
572            if start == i || end == i + 1 {
573                if start == i {
574                    buffer.unsafe_to_concat(Some(start), Some(start + 1));
575                }
576
577                if end == i + 1 {
578                    buffer.unsafe_to_concat(Some(end - 1), Some(end));
579                }
580
581                i += 1;
582                continue;
583            }
584
585            buffer.unsafe_to_break(Some(start), Some(end));
586
587            for info in &mut buffer.info[start..i] {
588                info.mask |= pre_mask;
589            }
590
591            buffer.info[i].mask |= ctx.plan.frac_mask;
592
593            for info in &mut buffer.info[i + 1..end] {
594                info.mask |= post_mask;
595            }
596
597            i = end;
598        } else {
599            i += 1;
600        }
601    }
602}

604fn set_unicode_props(buffer: &mut hb_buffer_t) {
605    // Implement enough of Unicode Graphemes here that shaping
606    // in reverse-direction wouldn't break graphemes.  Namely,
607    // we mark all marks and ZWJ and ZWJ,Extended_Pictographic
608    // sequences as continuations.  The foreach_grapheme()
609    // macro uses this bit.
610    //
611    // https://www.unicode.org/reports/tr29/#Regex_Definitions
612
613    let len = buffer.len;
614
615    let mut i = 0;
616    while i < len {
617        let info = &mut buffer.info[i];
618        info.init_unicode_props(&mut buffer.scratch_flags);
619
620        if info.glyph_id < 0x80 {
621            i += 1;
622            continue;
623        }
624
625        let gen_cat = info.general_category();
626
627        if (rb_flag_unsafe(gen_cat.to_u8() as u32)
628            & (rb_flag(HB_UNICODE_GENERAL_CATEGORY_LOWERCASE_LETTER)
629                | rb_flag(HB_UNICODE_GENERAL_CATEGORY_UPPERCASE_LETTER)
630                | rb_flag(HB_UNICODE_GENERAL_CATEGORY_TITLECASE_LETTER)
631                | rb_flag(HB_UNICODE_GENERAL_CATEGORY_OTHER_LETTER)
632                | rb_flag(HB_UNICODE_GENERAL_CATEGORY_SPACE_SEPARATOR)))
633            != 0
634        {
635            i += 1;
636            continue;
637        }
638
639        // Mutably borrow buffer.info[i] and immutably borrow
640        // buffer.info[i - 1] (if present) in a way that the borrow
641        // checker can understand.
642        let (prior, later) = buffer.info.split_at_mut(i);
643        let info = &mut later[0];
644
645        // Marks are already set as continuation by the above line.
646        // Handle Emoji_Modifier and ZWJ-continuation.
647        if gen_cat == GeneralCategory::MODIFIER_SYMBOL && matches!(info.glyph_id, 0x1F3FB..=0x1F3FF)
648        {
649            info.set_continuation(&mut buffer.scratch_flags);
650        } else if i != 0 && matches!(info.glyph_id, 0x1F1E6..=0x1F1FF) {
651            // Should never fail because we checked for i > 0.
652            // TODO: use let chains when they become stable
653            let prev = prior.last().unwrap();
654            if matches!(prev.glyph_id, 0x1F1E6..=0x1F1FF) && !prev.is_continuation() {
655                info.set_continuation(&mut buffer.scratch_flags);
656            }
657        } else if info.is_zwj() {
658            info.set_continuation(&mut buffer.scratch_flags);
659            if let Some(next) = buffer.info[..len].get_mut(i + 1) {
660                if next.as_codepoint().is_emoji_extended_pictographic() {
661                    next.init_unicode_props(&mut buffer.scratch_flags);
662                    next.set_continuation(&mut buffer.scratch_flags);
663                    i += 1;
664                }
665            }
666        } else if matches!(info.glyph_id, 0xFF9E..=0xFF9F | 0xE0020..=0xE007F) {
667            // Or part of the Other_Grapheme_Extend that is not marks.
668            // As of Unicode 15 that is just:
669            //
670            // 200C          ; Other_Grapheme_Extend # Cf       ZERO WIDTH NON-JOINER
671            // FF9E..FF9F    ; Other_Grapheme_Extend # Lm   [2] HALFWIDTH KATAKANA VOICED SOUND MARK..HALFWIDTH KATAKANA
672            // SEMI-VOICED SOUND MARK E0020..E007F  ; Other_Grapheme_Extend # Cf  [96] TAG SPACE..CANCEL TAG
673            //
674            // ZWNJ is special, we don't want to merge it as there's no need, and keeping
675            // it separate results in more granular clusters.
676            // Tags are used for Emoji sub-region flag sequences:
677            // https://github.com/harfbuzz/harfbuzz/issues/1556
678            // Katakana ones were requested:
679            // https://github.com/harfbuzz/harfbuzz/issues/3844
680            info.set_continuation(&mut buffer.scratch_flags);
681        } else if info.glyph_id == 0x2044
682        /* FRACTION SLASH */
683        {
684            buffer.scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_FRACTION_SLASH;
685        }
686
687        i += 1;
688    }
689}

691fn insert_dotted_circle(buffer: &mut hb_buffer_t, face: &hb_font_t) {
692    if !buffer
693        .flags
694        .contains(BufferFlags::DO_NOT_INSERT_DOTTED_CIRCLE)
695        && buffer.flags.contains(BufferFlags::BEGINNING_OF_TEXT)
696        && buffer.context_len[0] == 0
697        && buffer.info[0].is_unicode_mark()
698        && face.has_glyph(0x25CC)
699    {
700        let mut info = GlyphInfo {
701            glyph_id: 0x25CC,
702            mask: buffer.cur(0).mask,
703            cluster: buffer.cur(0).cluster,
704            ..GlyphInfo::default()
705        };
706
707        info.init_unicode_props(&mut buffer.scratch_flags);
708        buffer.clear_output();
709        buffer.output_info(info);
710        buffer.sync();
711    }
712}

714fn form_clusters(buffer: &mut hb_buffer_t) {
715    if buffer.scratch_flags & HB_BUFFER_SCRATCH_FLAG_HAS_CONTINUATIONS != 0 {
716        if BufferClusterLevel::new(buffer.cluster_level).is_graphemes() {
717            foreach_grapheme!(buffer, start, end, { buffer.merge_clusters(start, end) });
718        } else {
719            foreach_grapheme!(buffer, start, end, {
720                buffer.unsafe_to_break(Some(start), Some(end));
721            });
722        }
723    }

726fn ensure_native_direction(buffer: &mut hb_buffer_t) {
727    let dir = buffer.direction;
728    let mut hor = buffer
729        .script
730        .and_then(Direction::from_script)
731        .unwrap_or_default();
732
733    // Numeric runs in natively-RTL scripts are actually native-LTR, so we reset
734    // the horiz_dir if the run contains at least one decimal-number char, and no
735    // letter chars (ideally we should be checking for chars with strong
736    // directionality but hb-unicode currently lacks bidi categories).
737    //
738    // This allows digit sequences in Arabic etc to be shaped in "native"
739    // direction, so that features like ligatures will work as intended.
740    //
741    // https://github.com/harfbuzz/harfbuzz/issues/501
742    //
743    // Similar thing about Regional_Indicators; They are bidi=L, but Script=Common.
744    // If they are present in a run of natively-RTL text, they get assigned a script
745    // with natively RTL direction, which would result in wrong shaping if we
746    // assign such native RTL direction to them then. Detect that as well.
747    //
748    // https://github.com/harfbuzz/harfbuzz/issues/3314
749
750    if hor == Direction::RightToLeft && dir == Direction::LeftToRight {
751        let mut found_number = false;
752        let mut found_letter = false;
753        let mut found_ri = false;
754        for info in &buffer.info {
755            let gc = info.general_category();
756            if gc == GeneralCategory::DECIMAL_NUMBER {
757                found_number = true;
758            } else if gc.is_letter() {
759                found_letter = true;
760                break;
761            } else if matches!(info.glyph_id, 0x1F1E6..=0x1F1FF) {
762                found_ri = true;
763            }
764        }
765        if (found_number || found_ri) && !found_letter {
766            hor = Direction::LeftToRight;
767        }
768    }
769
770    // TODO vertical:
771    // The only BTT vertical script is Ogham, but it's not clear to me whether OpenType
772    // Ogham fonts are supposed to be implemented BTT or not.  Need to research that
773    // first.
774    if (dir.is_horizontal() && dir != hor && hor != Direction::Invalid)
775        || (dir.is_vertical() && dir != Direction::TopToBottom)
776    {
777        _hb_ot_layout_reverse_graphemes(buffer);
778        buffer.direction = buffer.direction.reverse();
779    }
780}

782fn rotate_chars(ctx: &mut hb_ot_shape_context_t) {
783    let len = ctx.buffer.len;
784
785    if ctx.target_direction.is_backward() {
786        let rtlm_mask = ctx.plan.rtlm_mask;
787
788        for info in &mut ctx.buffer.info[..len] {
789            if let Some(c) = info.as_codepoint().mirrored() {
790                if ctx.face.has_glyph(c) {
791                    info.glyph_id = c;
792                    continue;
793                }
794            }
795            info.mask |= rtlm_mask;
796        }
797    }
798
799    if ctx.target_direction.is_vertical() && !ctx.plan.has_vert {
800        for info in &mut ctx.buffer.info[..len] {
801            if let Some(c) = info.as_codepoint().vertical() {
802                if ctx.face.has_glyph(c) {
803                    info.glyph_id = c;
804                }
805            }
806        }
807    }
808}

822fn hb_synthesize_glyph_classes(buffer: &mut hb_buffer_t) {
823    let len = buffer.len;
824    for info in &mut buffer.info[..len] {
825        // Never mark default-ignorables as marks.
826        // They won't get in the way of lookups anyway,
827        // but having them as mark will cause them to be skipped
828        // over if the lookup-flag says so, but at least for the
829        // Mongolian variation selectors, looks like Uniscribe
830        // marks them as non-mark.  Some Mongolian fonts without
831        // GDEF rely on this.  Another notable character that
832        // this applies to is COMBINING GRAPHEME JOINER.
833        let class = if info.general_category() != GeneralCategory::NON_SPACING_MARK
834            || info.is_default_ignorable()
835        {
836            GlyphPropsFlags::BASE_GLYPH
837        } else {
838            GlyphPropsFlags::MARK
839        };
840
841        info.set_glyph_props(class.bits());
842    }
843}

845fn zero_width_default_ignorables(buffer: &mut hb_buffer_t) {
846    if buffer.scratch_flags & HB_BUFFER_SCRATCH_FLAG_HAS_DEFAULT_IGNORABLES != 0
847        && !buffer
848            .flags
849            .contains(BufferFlags::PRESERVE_DEFAULT_IGNORABLES)
850        && !buffer
851            .flags
852            .contains(BufferFlags::REMOVE_DEFAULT_IGNORABLES)
853    {
854        let len = buffer.len;
855        for (info, pos) in buffer.info[..len].iter().zip(&mut buffer.pos[..len]) {
856            if info.is_default_ignorable() {
857                pos.x_advance = 0;
858                pos.y_advance = 0;
859                if buffer.direction.is_horizontal() {
860                    pos.x_offset = 0;
861                } else {
862                    pos.y_offset = 0;
863                }
864            }
865        }
866    }
867}

869fn deal_with_variation_selectors(buffer: &mut hb_buffer_t) {
870    if buffer.scratch_flags & HB_BUFFER_SCRATCH_FLAG_HAS_VARIATION_SELECTOR_FALLBACK == 0 {
871        return;
872    }
873
874    // Note: In harfbuzz, this is part of the condition above (with OR), so it needs to stay
875    // in sync.
876    let Some(nf) = buffer.not_found_variation_selector else {
877        return;
878    };
879
880    let count = buffer.len;
881    let info = &mut buffer.info;
882    let pos = &mut buffer.pos;
883
884    for i in 0..count {
885        if info[i].is_variation_selector() {
886            info[i].glyph_id = nf;
887            pos[i].x_advance = 0;
888            pos[i].y_advance = 0;
889            pos[i].x_offset = 0;
890            pos[i].y_offset = 0;
891            info[0].set_variation_selector(false);
892        }
893    }
894}

896fn zero_mark_widths_by_gdef(buffer: &mut hb_buffer_t, adjust_offsets: bool) {
897    let len = buffer.len;
898    for (info, pos) in buffer.info[..len].iter().zip(&mut buffer.pos[..len]) {
899        if info.is_mark() {
900            if adjust_offsets {
901                pos.x_offset -= pos.x_advance;
902                pos.y_offset -= pos.y_advance;
903            }
904
905            pos.x_advance = 0;
906            pos.y_advance = 0;
907        }
908    }
909}

911fn hide_default_ignorables(buffer: &mut hb_buffer_t, face: &hb_font_t) {
912    if buffer.scratch_flags & HB_BUFFER_SCRATCH_FLAG_HAS_DEFAULT_IGNORABLES != 0
913        && !buffer
914            .flags
915            .contains(BufferFlags::PRESERVE_DEFAULT_IGNORABLES)
916    {
917        if !buffer
918            .flags
919            .contains(BufferFlags::REMOVE_DEFAULT_IGNORABLES)
920        {
921            if let Some(invisible) = buffer
922                .invisible
923                .or_else(|| face.get_nominal_glyph(u32::from(' ')))
924            {
925                let len = buffer.len;
926                for info in &mut buffer.info[..len] {
927                    if info.is_default_ignorable() {
928                        info.glyph_id = invisible.to_u32();
929                    }
930                }
931                return;
932            }
933        }
934
935        buffer.delete_glyphs_inplace(GlyphInfo::is_default_ignorable);
936    }
937}

939fn propagate_flags(buffer: &mut hb_buffer_t) {
940    // Propagate cluster-level glyph flags to be the same on all cluster glyphs.
941    // Simplifies using them.
942
943    let mut and_mask = glyph_flag::DEFINED;
944    if !buffer.flags.contains(BufferFlags::PRODUCE_UNSAFE_TO_CONCAT) {
945        and_mask &= !UNSAFE_TO_CONCAT;
946    }
947
948    if !buffer
949        .flags
950        .contains(BufferFlags::PRODUCE_SAFE_TO_INSERT_TATWEEL)
951    {
952        foreach_cluster!(buffer, start, end, {
953            if end - start == 1 {
954                buffer.info[start].mask &= and_mask;
955            } else {
956                let mut mask = 0;
957                for info in &buffer.info[start..end] {
958                    mask |= info.mask;
959                }
960
961                mask &= and_mask;
962
963                for info in &mut buffer.info[start..end] {
964                    info.mask = mask;
965                }
966            }
967        });
968        return;
969    }
970
971    /* If we are producing SAFE_TO_INSERT_TATWEEL, then do two things:
972     *
973     * - If the places that the Arabic shaper marked as SAFE_TO_INSERT_TATWEEL,
974     *   are UNSAFE_TO_BREAK, then clear the SAFE_TO_INSERT_TATWEEL,
975     * - Any place that is SAFE_TO_INSERT_TATWEEL, is also now UNSAFE_TO_BREAK.
976     *
977     * We couldn't make this interaction earlier. It has to be done here.
978     */
979    foreach_cluster!(buffer, start, end, {
980        // We cannot use `continue` in our `for_each_cluster!` macro.
981        if end - start != 1 {
982            let mut mask = 0;
983            for info in &buffer.info[start..end] {
984                mask |= info.mask;
985            }
986
987            mask &= glyph_flag::DEFINED;
988
989            if mask & UNSAFE_TO_BREAK != 0 {
990                mask &= !SAFE_TO_INSERT_TATWEEL;
991            }
992
993            if mask & SAFE_TO_INSERT_TATWEEL != 0 {
994                mask |= UNSAFE_TO_BREAK | UNSAFE_TO_CONCAT;
995            }
996
997            mask &= and_mask;
998
999            for info in &mut buffer.info[start..end] {
1000                info.mask = mask;
1001            }
1002        }
1003    });
