417struct IndicWouldSubstituteFeature {
418    lookups: Range<usize>,
419    zero_context: bool,
420}
421
422impl IndicWouldSubstituteFeature {
423    pub fn new(map: &hb_ot_map_t, feature_tag: hb_tag_t, zero_context: bool) -> Self {
424        IndicWouldSubstituteFeature {
425            lookups: match map.get_feature_stage(TableIndex::GSUB, feature_tag) {
426                Some(stage) => map.stage_lookup_range(TableIndex::GSUB, stage),
427                None => 0..0,
428            },
429            zero_context,
430        }
431    }
432
433    pub fn would_substitute(
434        &self,
435        map: &hb_ot_map_t,
436        face: &hb_font_t,
437        glyphs: &[GlyphId],
438    ) -> bool {
439        for index in self.lookups.clone() {
440            let lookup = map.lookup(TableIndex::GSUB, index);
441            let ctx = WouldApplyContext {
442                glyphs,
443                zero_context: self.zero_context,
444            };
445            if face
446                .ot_tables
447                .gsub
448                .as_ref()
449                .and_then(|table| table.get_lookup(lookup.index))
450                .is_some_and(|lookup| lookup.would_apply(face, &ctx) == Some(true))
451            {
452                return true;
453            }
454        }
455
456        false
457    }
458}

472impl IndicShapePlan {
473    fn new(plan: &hb_ot_shape_plan_t) -> Self {
474        let script = plan.script;
475        let config = if let Some(c) = INDIC_CONFIGS.iter().skip(1).find(|c| c.script == script) {
476            *c
477        } else {
478            INDIC_CONFIGS[0]
479        };
480
481        let is_old_spec = config.has_old_spec
482            && plan
483                .ot_map
484                .chosen_script(TableIndex::GSUB)
485                .is_none_or(|tag| tag.to_be_bytes()[3] != b'2');
486
487        // Use zero-context would_substitute() matching for new-spec of the main
488        // Indic scripts, and scripts with one spec only, but not for old-specs.
489        // The new-spec for all dual-spec scripts says zero-context matching happens.
490        //
491        // However, testing with Malayalam shows that old and new spec both allow
492        // context.  Testing with Bengali new-spec however shows that it doesn't.
493        // So, the heuristic here is the way it is.  It should *only* be changed,
494        // as we discover more cases of what Windows does.  DON'T TOUCH OTHERWISE.
495        let zero_context = is_old_spec && script != Some(script::MALAYALAM);
496
497        let mut mask_array = [0; INDIC_FEATURES.len()];
498        for (i, feature) in INDIC_FEATURES.iter().enumerate() {
499            mask_array[i] = if feature.1 & F_GLOBAL != 0 {
500                0
501            } else {
502                plan.ot_map.get_1_mask(feature.0)
503            }
504        }
505
506        // TODO: what is this?
507        // let mut virama_glyph = None;
508        // if config.virama != 0 {
509        //     if let Some(g) = face.glyph_index(char::try_from(config.virama).unwrap()) {
510        //         virama_glyph = Some(g.0 as u32);
511        //     }
512        // }
513
514        IndicShapePlan {
515            config,
516            is_old_spec,
517            // virama_glyph,
518            rphf: IndicWouldSubstituteFeature::new(
519                &plan.ot_map,
520                hb_tag_t::new(b"rphf"),
521                zero_context,
522            ),
523            pref: IndicWouldSubstituteFeature::new(
524                &plan.ot_map,
525                hb_tag_t::new(b"pref"),
526                zero_context,
527            ),
528            blwf: IndicWouldSubstituteFeature::new(
529                &plan.ot_map,
530                hb_tag_t::new(b"blwf"),
531                zero_context,
532            ),
533            pstf: IndicWouldSubstituteFeature::new(
534                &plan.ot_map,
535                hb_tag_t::new(b"pstf"),
536                zero_context,
537            ),
538            vatu: IndicWouldSubstituteFeature::new(
539                &plan.ot_map,
540                hb_tag_t::new(b"vatu"),
541                zero_context,
542            ),
543            mask_array,
544        }
545    }
546}

548fn collect_features(planner: &mut hb_ot_shape_planner_t) {
549    // Do this before any lookups have been applied.
550    planner.ot_map.add_gsub_pause(Some(setup_syllables));
551
552    planner
553        .ot_map
554        .enable_feature(hb_tag_t::new(b"locl"), F_PER_SYLLABLE, 1);
555    // The Indic specs do not require ccmp, but we apply it here since if
556    // there is a use of it, it's typically at the beginning.
557    planner
558        .ot_map
559        .enable_feature(hb_tag_t::new(b"ccmp"), F_PER_SYLLABLE, 1);
560
561    planner.ot_map.add_gsub_pause(Some(initial_reordering));
562
563    for feature in INDIC_FEATURES.iter().take(11) {
564        planner.ot_map.add_feature(feature.0, feature.1, 1);
565        planner.ot_map.add_gsub_pause(None);
566    }
567
568    planner.ot_map.add_gsub_pause(Some(final_reordering));
569
570    for feature in INDIC_FEATURES.iter().skip(11) {
571        planner.ot_map.add_feature(feature.0, feature.1, 1);
572    }
573}

584fn decompose(_: &hb_ot_shape_normalize_context_t, ab: Codepoint) -> Option<(Codepoint, Codepoint)> {
585    // Don't decompose these.
586    match ab {
587        0x0931 |               // DEVANAGARI LETTER RRA
588        // https://github.com/harfbuzz/harfbuzz/issues/779
589        0x09DC |               // BENGALI LETTER RRA
590        0x09DD |               // BENGALI LETTER RHA
591        0x0B94 => return None, // TAMIL LETTER AU
592        _ => {}
593    }
594
595    crate::hb::unicode::decompose(ab)
596}

598fn compose(_: &hb_ot_shape_normalize_context_t, a: Codepoint, b: Codepoint) -> Option<Codepoint> {
599    // Avoid recomposing split matras.
600    if a.general_category().is_mark() {
601        return None;
602    }
603
604    // Composition-exclusion exceptions that we want to recompose.
605    if a == 0x09AF && b == 0x09BC {
606        return Some(0x09DF);
607    }
608
609    crate::hb::unicode::compose(a, b)
610}

623fn setup_syllables(_: &hb_ot_shape_plan_t, _: &hb_font_t, buffer: &mut hb_buffer_t) -> bool {
624    buffer.allocate_var(GlyphInfo::SYLLABLE_VAR);
625
626    super::ot_shaper_indic_machine::find_syllables_indic(buffer);
627
628    let mut start = 0;
629    let mut end = buffer.next_syllable(0);
630    while start < buffer.len {
631        buffer.unsafe_to_break(Some(start), Some(end));
632        start = end;
633        end = buffer.next_syllable(start);
634    }
635
636    false
637}

640    plan: &hb_ot_shape_plan_t,
641    face: &hb_font_t,
642    buffer: &mut hb_buffer_t,
643) -> bool {
644    use super::ot_shaper_indic_machine::SyllableType;
645
646    let mut ret = false;
647
648    let indic_plan = plan.data::<IndicShapePlan>();
649
650    update_consonant_positions(plan, indic_plan, face, buffer);
651    if insert_dotted_circles(
652        face,
653        buffer,
654        SyllableType::BrokenCluster as u8,
655        ot_category_t::OT_DOTTEDCIRCLE,
656        Some(ot_category_t::OT_Repha),
657        Some(ot_position_t::POS_END),
658    ) {
659        ret = true;
660    }
661
662    let mut start = 0;
663    let mut end = buffer.next_syllable(0);
664    while start < buffer.len {
665        initial_reordering_syllable(plan, indic_plan, face, start, end, buffer);
666        start = end;
667        end = buffer.next_syllable(start);
668    }
669
670    ret
671}

673fn update_consonant_positions(
674    plan: &hb_ot_shape_plan_t,
675    indic_plan: &IndicShapePlan,
676    face: &hb_font_t,
677    buffer: &mut hb_buffer_t,
678) {
679    let mut virama_glyph = None;
680    if indic_plan.config.virama != 0 {
681        virama_glyph = face.get_nominal_glyph(indic_plan.config.virama);
682    }
683
684    if let Some(virama) = virama_glyph {
685        for info in buffer.info_slice_mut() {
686            if info.indic_position() == ot_position_t::POS_BASE_C {
687                let consonant = info.as_glyph();
688                info.set_indic_position(consonant_position_from_face(
689                    plan, indic_plan, face, consonant, virama,
690                ));
691            }
692        }
693    }
694}

696fn consonant_position_from_face(
697    plan: &hb_ot_shape_plan_t,
698    indic_plan: &IndicShapePlan,
699    face: &hb_font_t,
700    consonant: GlyphId,
701    virama: GlyphId,
702) -> u8 {
703    // For old-spec, the order of glyphs is Consonant,Virama,
704    // whereas for new-spec, it's Virama,Consonant.  However,
705    // some broken fonts (like Free Sans) simply copied lookups
706    // from old-spec to new-spec without modification.
707    // And oddly enough, Uniscribe seems to respect those lookups.
708    // Eg. in the sequence U+0924,U+094D,U+0930, Uniscribe finds
709    // base at 0.  The font however, only has lookups matching
710    // 930,94D in 'blwf', not the expected 94D,930 (with new-spec
711    // table).  As such, we simply match both sequences.  Seems
712    // to work.
713    //
714    // Vatu is done as well, for:
715    // https://github.com/harfbuzz/harfbuzz/issues/1587
716
717    if indic_plan
718        .blwf
719        .would_substitute(&plan.ot_map, face, &[virama, consonant])
720        || indic_plan
721            .blwf
722            .would_substitute(&plan.ot_map, face, &[consonant, virama])
723        || indic_plan
724            .vatu
725            .would_substitute(&plan.ot_map, face, &[virama, consonant])
726        || indic_plan
727            .vatu
728            .would_substitute(&plan.ot_map, face, &[consonant, virama])
729    {
730        return ot_position_t::POS_BELOW_C;
731    }
732
733    if indic_plan
734        .pstf
735        .would_substitute(&plan.ot_map, face, &[virama, consonant])
736        || indic_plan
737            .pstf
738            .would_substitute(&plan.ot_map, face, &[consonant, virama])
739    {
740        return ot_position_t::POS_POST_C;
741    }
742
743    if indic_plan
744        .pref
745        .would_substitute(&plan.ot_map, face, &[virama, consonant])
746        || indic_plan
747            .pref
748            .would_substitute(&plan.ot_map, face, &[consonant, virama])
749    {
750        return ot_position_t::POS_POST_C;
751    }
752
753    ot_position_t::POS_BASE_C
754}

756fn initial_reordering_syllable(
757    plan: &hb_ot_shape_plan_t,
758    indic_plan: &IndicShapePlan,
759    face: &hb_font_t,
760    start: usize,
761    end: usize,
762    buffer: &mut hb_buffer_t,
763) {
764    use super::ot_shaper_indic_machine::SyllableType;
765
766    let syllable_type = match buffer.info[start].syllable() & 0x0F {
767        0 => SyllableType::ConsonantSyllable,
768        1 => SyllableType::VowelSyllable,
769        2 => SyllableType::StandaloneCluster,
770        3 => SyllableType::SymbolCluster,
771        4 => SyllableType::BrokenCluster,
772        5 => SyllableType::NonIndicCluster,
773        _ => unreachable!(),
774    };
775
776    match syllable_type {
777        // We made the vowels look like consonants.  So let's call the consonant logic!
778        SyllableType::VowelSyllable | SyllableType::ConsonantSyllable => {
779            initial_reordering_consonant_syllable(plan, indic_plan, face, start, end, buffer);
780        }
781        // We already inserted dotted-circles, so just call the standalone_cluster.
782        SyllableType::BrokenCluster | SyllableType::StandaloneCluster => {
783            initial_reordering_standalone_cluster(plan, indic_plan, face, start, end, buffer);
784        }
785        SyllableType::SymbolCluster | SyllableType::NonIndicCluster => {}
786    }
787}

789// Rules from:
790// https://docs.microsqoft.com/en-us/typography/script-development/devanagari */
791fn initial_reordering_consonant_syllable(
792    plan: &hb_ot_shape_plan_t,
793    indic_plan: &IndicShapePlan,
794    face: &hb_font_t,
795    start: usize,
796    end: usize,
797    buffer: &mut hb_buffer_t,
798) {
799    // https://github.com/harfbuzz/harfbuzz/issues/435#issuecomment-335560167
800    // For compatibility with legacy usage in Kannada,
801    // Ra+h+ZWJ must behave like Ra+ZWJ+h...
802    if buffer.script == Some(script::KANNADA)
803        && start + 3 <= end
804        && buffer.info[start].is_one_of(category_flag(ot_category_t::OT_Ra))
805        && buffer.info[start + 1].is_one_of(category_flag(ot_category_t::OT_H))
806        && buffer.info[start + 2].is_one_of(category_flag(ot_category_t::OT_ZWJ))
807    {
808        buffer.merge_clusters(start + 1, start + 3);
809        buffer.info.swap(start + 1, start + 2);
810    }
811
812    // 1. Find base consonant:
813    //
814    // The shaping engine finds the base consonant of the syllable, using the
815    // following algorithm: starting from the end of the syllable, move backwards
816    // until a consonant is found that does not have a below-base or post-base
817    // form (post-base forms have to follow below-base forms), or that is not a
818    // pre-base-reordering Ra, or arrive at the first consonant. The consonant
819    // stopped at will be the base.
820    //
821    //   - If the syllable starts with Ra + Halant (in a script that has Reph)
822    //     and has more than one consonant, Ra is excluded from candidates for
823    //     base consonants.
824
825    let mut base = end;
826    let mut has_reph = false;
827
828    {
829        // -> If the syllable starts with Ra + Halant (in a script that has Reph)
830        //    and has more than one consonant, Ra is excluded from candidates for
831        //    base consonants.
832        let mut limit = start;
833        if indic_plan.mask_array[indic_feature::RPHF] != 0
834            && start + 3 <= end
835            && ((indic_plan.config.reph_mode == RephMode::Implicit
836                && !buffer.info[start + 2].is_joiner())
837                || (indic_plan.config.reph_mode == RephMode::Explicit
838                    && buffer.info[start + 2].indic_category() == ot_category_t::OT_ZWJ))
839        {
840            // See if it matches the 'rphf' feature.
841            let glyphs = &[
842                buffer.info[start].as_glyph(),
843                buffer.info[start + 1].as_glyph(),
844                if indic_plan.config.reph_mode == RephMode::Explicit {
845                    buffer.info[start + 2].as_glyph()
846                } else {
847                    GlyphId::NOTDEF
848                },
849            ];
850            if indic_plan
851                .rphf
852                .would_substitute(&plan.ot_map, face, &glyphs[0..2])
853                || (indic_plan.config.reph_mode == RephMode::Explicit
854                    && indic_plan.rphf.would_substitute(&plan.ot_map, face, glyphs))
855            {
856                limit += 2;
857                while limit < end && buffer.info[limit].is_joiner() {
858                    limit += 1;
859                }
860                base = start;
861                has_reph = true;
862            }
863        } else if indic_plan.config.reph_mode == RephMode::LogRepha
864            && buffer.info[start].indic_category() == ot_category_t::OT_Repha
865        {
866            limit += 1;
867            while limit < end && buffer.info[limit].is_joiner() {
868                limit += 1;
869            }
870            base = start;
871            has_reph = true;
872        }
873
874        {
875            // -> starting from the end of the syllable, move backwards
876            let mut i = end;
877            let mut seen_below = false;
878            loop {
879                i -= 1;
880                // -> until a consonant is found
881                if buffer.info[i].is_consonant() {
882                    // -> that does not have a below-base or post-base form
883                    // (post-base forms have to follow below-base forms),
884                    if buffer.info[i].indic_position() != ot_position_t::POS_BELOW_C
885                        && (buffer.info[i].indic_position() != ot_position_t::POS_POST_C
886                            || seen_below)
887                    {
888                        base = i;
889                        break;
890                    }
891                    if buffer.info[i].indic_position() == ot_position_t::POS_BELOW_C {
892                        seen_below = true;
893                    }
894
895                    // -> or that is not a pre-base-reordering Ra,
896                    //
897                    // IMPLEMENTATION NOTES:
898                    //
899                    // Our pre-base-reordering Ra's are marked position::PostC, so will be skipped
900                    // by the logic above already.
901
902                    // -> or arrive at the first consonant. The consonant stopped at will
903                    // be the base.
904                    base = i;
905                } else {
906                    // A ZWJ after a Halant stops the base search, and requests an explicit
907                    // half form.
908                    // A ZWJ before a Halant, requests a subjoined form instead, and hence
909                    // search continues.  This is particularly important for Bengali
910                    // sequence Ra,H,Ya that should form Ya-Phalaa by subjoining Ya.
911                    if start < i
912                        && buffer.info[i].indic_category() == ot_category_t::OT_ZWJ
913                        && buffer.info[i - 1].indic_category() == ot_category_t::OT_H
914                    {
915                        break;
916                    }
917                }
918
919                if i <= limit {
920                    break;
921                }
922            }
923        }
924
925        // -> If the syllable starts with Ra + Halant (in a script that has Reph)
926        //    and has more than one consonant, Ra is excluded from candidates for
927        //    base consonants.
928        //
929        //  Only do this for unforced Reph. (ie. not for Ra,H,ZWJ.
930        if has_reph && base == start && limit - base <= 2 {
931            // Have no other consonant, so Reph is not formed and Ra becomes base.
932            has_reph = false;
933        }
934    }
935
936    // 2. Decompose and reorder Matras:
937    //
938    // Each matra and any syllable modifier sign in the syllable are moved to the
939    // appropriate position relative to the consonant(s) in the syllable. The
940    // shaping engine decomposes two- or three-part matras into their constituent
941    // parts before any repositioning. Matra characters are classified by which
942    // consonant in a conjunct they have affinity for and are reordered to the
943    // following positions:
944    //
945    //   - Before first half form in the syllable
946    //   - After subjoined consonants
947    //   - After post-form consonant
948    //   - After main consonant (for above marks)
949    //
950    // IMPLEMENTATION NOTES:
951    //
952    // The normalize() routine has already decomposed matras for us, so we don't
953    // need to worry about that.
954
955    // 3.  Reorder marks to canonical order:
956    //
957    // Adjacent nukta and halant or nukta and vedic sign are always repositioned
958    // if necessary, so that the nukta is first.
959    //
960    // IMPLEMENTATION NOTES:
961    //
962    // We don't need to do this: the normalize() routine already did this for us.
963
964    // Reorder characters
965
966    for i in start..base {
967        let pos = buffer.info[i].indic_position();
968        buffer.info[i].set_indic_position(cmp::min(ot_position_t::POS_PRE_C, pos));
969    }
970
971    if base < end {
972        buffer.info[base].set_indic_position(ot_position_t::POS_BASE_C);
973    }
974
975    // Handle beginning Ra
976    if has_reph {
977        buffer.info[start].set_indic_position(ot_position_t::POS_RA_TO_BECOME_REPH);
978    }
979
980    // For old-style Indic script tags, move the first post-base Halant after
981    // last consonant.
982    //
983    // Reports suggest that in some scripts Uniscribe does this only if there
984    // is *not* a Halant after last consonant already.  We know that is the
985    // case for Kannada, while it reorders unconditionally in other scripts,
986    // eg. Malayalam, Bengali, and Devanagari.  We don't currently know about
987    // other scripts, so we block Kannada.
988    //
989    // Kannada test case:
990    // U+0C9A,U+0CCD,U+0C9A,U+0CCD
991    // With some versions of Lohit Kannada.
992    // https://bugs.freedesktop.org/show_bug.cgi?id=59118
993    //
994    // Malayalam test case:
995    // U+0D38,U+0D4D,U+0D31,U+0D4D,U+0D31,U+0D4D
996    // With lohit-ttf-20121122/Lohit-Malayalam.ttf
997    //
998    // Bengali test case:
999    // U+0998,U+09CD,U+09AF,U+09CD
1000    // With Windows XP vrinda.ttf
1001    // https://github.com/harfbuzz/harfbuzz/issues/1073
1002    //
1003    // Devanagari test case:
1004    // U+091F,U+094D,U+0930,U+094D
1005    // With chandas.ttf
1006    // https://github.com/harfbuzz/harfbuzz/issues/1071
1007    if indic_plan.is_old_spec {
1008        let disallow_double_halants = buffer.script == Some(script::KANNADA);
1009        for i in base + 1..end {
1010            if buffer.info[i].indic_category() == ot_category_t::OT_H {
1011                let mut j = end - 1;
1012                while j > i {
1013                    if buffer.info[j].is_consonant()
1014                        || (disallow_double_halants
1015                            && buffer.info[j].indic_category() == ot_category_t::OT_H)
1016                    {
1017                        break;
1018                    }
1019
1020                    j -= 1;
1021                }
1022
1023                if buffer.info[j].indic_category() != ot_category_t::OT_H && j > i {
1024                    // Move Halant to after last consonant.
1025                    let t = buffer.info[i];
1026                    for k in 0..j - i {
1027                        buffer.info[k + i] = buffer.info[k + i + 1];
1028                    }
1029                    buffer.info[j] = t;
1030                }
1031
1032                break;
1033            }
1034        }
1035    }
1036
1037    // Attach misc marks to previous char to move with them.
1038    {
1039        let mut last_pos = ot_position_t::POS_START;
1040        for i in start..end {
1041            let ok = rb_flag_unsafe(buffer.info[i].indic_category() as u32)
1042                & (category_flag(ot_category_t::OT_ZWJ)
1043                    | category_flag(ot_category_t::OT_ZWNJ)
1044                    | category_flag(ot_category_t::OT_N)
1045                    | category_flag(ot_category_t::OT_RS)
1046                    | category_flag(ot_category_t::OT_CM)
1047                    | category_flag(ot_category_t::OT_H))
1048                != 0;
1049            if ok {
1050                buffer.info[i].set_indic_position(last_pos);
1051
1052                if buffer.info[i].indic_category() == ot_category_t::OT_H
1053                    && buffer.info[i].indic_position() == ot_position_t::POS_PRE_M
1054                {
1055                    // Uniscribe doesn't move the Halant with Left Matra.
1056                    // TEST: U+092B,U+093F,U+094DE
1057                    // We follow.
1058                    for j in (start + 1..=i).rev() {
1059                        if buffer.info[j - 1].indic_position() != ot_position_t::POS_PRE_M {
1060                            let pos = buffer.info[j - 1].indic_position();
1061                            buffer.info[i].set_indic_position(pos);
1062                            break;
1063                        }
1064                    }
1065                }
1066            } else if buffer.info[i].indic_position() != ot_position_t::POS_SMVD {
1067                if buffer.info[i].indic_category() == ot_category_t::OT_MPst
1068                    && i > start
1069                    && buffer.info[i - 1].indic_category() == ot_category_t::OT_SM
1070                {
1071                    let val = buffer.info[i].indic_position();
1072                    buffer.info[i - 1].set_indic_position(val);
1073                }
1074
1075                last_pos = buffer.info[i].indic_position();
1076            }
1077        }
1078    }
1079    // For post-base consonants let them own anything before them
1080    // since the last consonant or matra.
1081    {
1082        let mut last = base;
1083        for i in base + 1..end {
1084            if buffer.info[i].is_consonant() {
1085                for j in last + 1..i {
1086                    if buffer.info[j].indic_position() < ot_position_t::POS_SMVD {
1087                        let pos = buffer.info[i].indic_position();
1088                        buffer.info[j].set_indic_position(pos);
1089                    }
1090                }
1091
1092                last = i;
1093            } else if (rb_flag_unsafe(buffer.info[i].indic_category() as u32)
1094                & (rb_flag(ot_category_t::OT_M as u32) | rb_flag(ot_category_t::OT_MPst as u32)))
1095                != 0
1096            {
1097                last = i;
1098            }
1099        }
1100    }
1101
1102    {
1103        // Use syllable() for sort accounting temporarily.
1104        let syllable = buffer.info[start].syllable();
1105        for i in start..end {
1106            buffer.info[i].set_syllable(u8::try_from(i - start).unwrap());
1107        }
1108
1109        buffer.info[start..end].sort_by_key(|a| a.indic_position());
1110
1111        // Find base again; also flip left-matra sequence.
1112        let mut first_left_mantra = end;
1113        let mut last_left_mantra = end;
1114        base = end;
1115
1116        for i in start..end {
1117            if buffer.info[i].indic_position() == ot_position_t::POS_BASE_C {
1118                base = i;
1119                break;
1120            } else if buffer.info[i].indic_position() == ot_position_t::POS_PRE_M {
1121                if first_left_mantra == end {
1122                    first_left_mantra = i;
1123                }
1124
1125                last_left_mantra = i;
1126            }
1127        }
1128
1129        // https://github.com/harfbuzz/harfbuzz/issues/3863
1130        if first_left_mantra < last_left_mantra {
1131            // No need to merge clusters, handled later.
1132            buffer.reverse_range(first_left_mantra, last_left_mantra + 1);
1133            // Reverse back nuktas, etc.
1134            let mut i = first_left_mantra;
1135
1136            for j in i..=last_left_mantra {
1137                if (rb_flag_unsafe(buffer.info[j].indic_category() as u32)
1138                    & (rb_flag(ot_category_t::OT_M as u32)
1139                        | rb_flag(ot_category_t::OT_MPst as u32)))
1140                    != 0
1141                {
1142                    buffer.reverse_range(i, j + 1);
1143                    i = j + 1;
1144                }
1145            }
1146        }
1147
1148        // Things are out-of-control for post base positions, they may shuffle
1149        // around like crazy.  In old-spec mode, we move halants around, so in
1150        // that case merge all clusters after base.  Otherwise, check the sort
1151        // order and merge as needed.
1152        // For pre-base stuff, we handle cluster issues in final reordering.
1153        //
1154        // We could use buffer->sort() for this, if there was no special
1155        // reordering of pre-base stuff happening later...
1156        // We don't want to merge_clusters all of that, which buffer->sort()
1157        // would.  Here's a concrete example:
1158        //
1159        // Assume there's a pre-base consonant and explicit Halant before base,
1160        // followed by a prebase-reordering (left) Matra:
1161        //
1162        //   C,H,ZWNJ,B,M
1163        //
1164        // At this point in reordering we would have:
1165        //
1166        //   M,C,H,ZWNJ,B
1167        //
1168        // whereas in final reordering we will bring the Matra closer to Base:
1169        //
1170        //   C,H,ZWNJ,M,B
1171        //
1172        // That's why we don't want to merge-clusters anything before the Base
1173        // at this point.  But if something moved from after Base to before it,
1174        // we should merge clusters from base to them.  In final-reordering, we
1175        // only move things around before base, and merge-clusters up to base.
1176        // These two merge-clusters from the two sides of base will interlock
1177        // to merge things correctly.  See:
1178        // https://github.com/harfbuzz/harfbuzz/issues/2272
1179        if indic_plan.is_old_spec || end - start > 127 {
1180            buffer.merge_clusters(base, end);
1181        } else {
1182            // Note! syllable() is a one-byte field.
1183            for i in base..end {
1184                if buffer.info[i].syllable() != 255 {
1185                    let mut min = i;
1186                    let mut max = i;
1187                    let mut j = start + buffer.info[i].syllable() as usize;
1188                    while j != i {
1189                        min = cmp::min(min, j);
1190                        max = cmp::max(max, j);
1191                        let next = start + buffer.info[j].syllable() as usize;
1192                        buffer.info[j].set_syllable(255); // So we don't process j later again.
1193                        j = next;
1194                    }
1195
1196                    buffer.merge_clusters(cmp::max(base, min), max + 1);
1197                }
1198            }
1199        }
1200
1201        // Put syllable back in.
1202        for info in &mut buffer.info[start..end] {
1203            info.set_syllable(syllable);
1204        }
1205    }
1206
1207    // Setup masks now
1208
1209    {
1210        // Reph
1211        for info in &mut buffer.info[start..end] {
1212            if info.indic_position() != ot_position_t::POS_RA_TO_BECOME_REPH {
1213                break;
1214            }
1215
1216            info.mask |= indic_plan.mask_array[indic_feature::RPHF];
1217        }
1218
1219        // Pre-base
1220        let mut mask = indic_plan.mask_array[indic_feature::HALF];
1221        if !indic_plan.is_old_spec && indic_plan.config.blwf_mode == BlwfMode::PreAndPost {
1222            mask |= indic_plan.mask_array[indic_feature::BLWF];
1223        }
1224
1225        for info in &mut buffer.info[start..base] {
1226            info.mask |= mask;
1227        }
1228
1229        // Base
1230        mask = 0;
1231        if base < end {
1232            buffer.info[base].mask |= mask;
1233        }
1234
1235        // Post-base
1236        mask = indic_plan.mask_array[indic_feature::BLWF]
1237            | indic_plan.mask_array[indic_feature::ABVF]
1238            | indic_plan.mask_array[indic_feature::PSTF];
1239        for i in base + 1..end {
1240            buffer.info[i].mask |= mask;
1241        }
1242    }
1243
1244    if indic_plan.is_old_spec && buffer.script == Some(script::DEVANAGARI) {
1245        // Old-spec eye-lash Ra needs special handling.  From the
1246        // spec:
1247        //
1248        // "The feature 'below-base form' is applied to consonants
1249        // having below-base forms and following the base consonant.
1250        // The exception is vattu, which may appear below half forms
1251        // as well as below the base glyph. The feature 'below-base
1252        // form' will be applied to all such occurrences of Ra as well."
1253        //
1254        // Test case: U+0924,U+094D,U+0930,U+094d,U+0915
1255        // with Sanskrit 2003 font.
1256        //
1257        // However, note that Ra,Halant,ZWJ is the correct way to
1258        // request eyelash form of Ra, so we wouldbn't inhibit it
1259        // in that sequence.
1260        //
1261        // Test case: U+0924,U+094D,U+0930,U+094d,U+200D,U+0915
1262        for i in start..base.saturating_sub(1) {
1263            if buffer.info[i].indic_category() == ot_category_t::OT_Ra
1264                && buffer.info[i + 1].indic_category() == ot_category_t::OT_H
1265                && (i + 2 == base || buffer.info[i + 2].indic_category() != ot_category_t::OT_ZWJ)
1266            {
1267                buffer.info[i].mask |= indic_plan.mask_array[indic_feature::BLWF];
1268                buffer.info[i + 1].mask |= indic_plan.mask_array[indic_feature::BLWF];
1269            }
1270        }
1271    }
1272
1273    let pref_len = 2;
1274    if indic_plan.mask_array[indic_feature::PREF] != 0 && base + pref_len < end {
1275        // Find a Halant,Ra sequence and mark it for pre-base-reordering processing.
1276        for i in base + 1..end - pref_len + 1 {
1277            let glyphs = &[buffer.info[i + 0].as_glyph(), buffer.info[i + 1].as_glyph()];
1278            if indic_plan.pref.would_substitute(&plan.ot_map, face, glyphs) {
1279                buffer.info[i + 0].mask |= indic_plan.mask_array[indic_feature::PREF];
1280                buffer.info[i + 1].mask |= indic_plan.mask_array[indic_feature::PREF];
1281                break;
1282            }
1283        }
1284    }
1285
1286    // Apply ZWJ/ZWNJ effects
1287    for i in start + 1..end {
1288        if buffer.info[i].is_joiner() {
1289            let non_joiner = buffer.info[i].indic_category() == ot_category_t::OT_ZWNJ;
1290            let mut j = i;
1291
1292            loop {
1293                j -= 1;
1294
1295                // ZWJ/ZWNJ should disable CJCT.  They do that by simply
1296                // being there, since we don't skip them for the CJCT
1297                // feature (ie. F_MANUAL_ZWJ)
1298
1299                // A ZWNJ disables HALF.
1300                if non_joiner {
1301                    buffer.info[j].mask &= !indic_plan.mask_array[indic_feature::HALF];
1302                }
1303
1304                if j <= start || buffer.info[j].is_consonant() {
1305                    break;
1306                }
1307            }
1308        }
1309    }
1310}

1325fn final_reordering(plan: &hb_ot_shape_plan_t, face: &hb_font_t, buffer: &mut hb_buffer_t) -> bool {
1326    if buffer.is_empty() {
1327        return false;
1328    }

1340fn final_reordering_impl(
1341    plan: &hb_ot_shape_plan_t,
1342    face: &hb_font_t,
1343    start: usize,
1344    end: usize,
1345    buffer: &mut hb_buffer_t,
1346) {
1347    let indic_plan = plan.data::<IndicShapePlan>();
1348
1349    // This function relies heavily on halant glyphs.  Lots of ligation
1350    // and possibly multiple substitutions happened prior to this
1351    // phase, and that might have messed up our properties.  Recover
1352    // from a particular case of that where we're fairly sure that a
1353    // class of OT_H is desired but has been lost.
1354    //
1355    // We don't call load_virama_glyph(), since we know it's already loaded.
1356    let mut virama_glyph = None;
1357    if indic_plan.config.virama != 0 {
1358        if let Some(g) = face.get_nominal_glyph(indic_plan.config.virama) {
1359            virama_glyph = Some(g.to_u32());
1360        }
1361    }
1362
1363    if let Some(virama_glyph) = virama_glyph {
1364        for info in &mut buffer.info[start..end] {
1365            if info.glyph_id == virama_glyph && info.ligated() && info.multiplied() {
1366                // This will make sure that this glyph passes is_halant() test.
1367                info.set_indic_category(ot_category_t::OT_H);
1368                info.clear_ligated_and_multiplied();
1369            }
1370        }
1371    }
1372
1373    // 4. Final reordering:
1374    //
1375    // After the localized forms and basic shaping forms GSUB features have been
1376    // applied (see below), the shaping engine performs some final glyph
1377    // reordering before applying all the remaining font features to the entire
1378    // syllable.
1379
1380    let mut try_pref = indic_plan.mask_array[indic_feature::PREF] != 0;
1381
1382    let mut base = start;
1383    while base < end {
1384        if buffer.info[base].indic_position() as u32 >= ot_position_t::POS_BASE_C as u32 {
1385            if try_pref && base + 1 < end {
1386                for i in base + 1..end {
1387                    if (buffer.info[i].mask & indic_plan.mask_array[indic_feature::PREF]) != 0 {
1388                        if !(buffer.info[i].substituted()
1389                            && buffer.info[i].ligated_and_didnt_multiply())
1390                        {
1391                            // Ok, this was a 'pref' candidate but didn't form any.
1392                            // Base is around here...
1393                            base = i;
1394                            while base < end && buffer.info[base].is_halant() {
1395                                base += 1;
1396                            }
1397
1398                            if base < end {
1399                                buffer.info[base].set_indic_position(ot_position_t::POS_BASE_C);
1400                            }
1401
1402                            try_pref = false;
1403                        }
1404
1405                        break;
1406                    }
1407
1408                    if base == end {
1409                        break;
1410                    }
1411                }
1412            }
1413
1414            // For Malayalam, skip over unformed below- (but NOT post-) forms.
1415            if buffer.script == Some(script::MALAYALAM) {
1416                let mut i = base + 1;
1417                while i < end {
1418                    while i < end && buffer.info[i].is_joiner() {
1419                        i += 1;
1420                    }
1421
1422                    if i == end || !buffer.info[i].is_halant() {
1423                        break;
1424                    }
1425
1426                    i += 1; // Skip halant.
1427
1428                    while i < end && buffer.info[i].is_joiner() {
1429                        i += 1;
1430                    }
1431
1432                    if i < end
1433                        && buffer.info[i].is_consonant()
1434                        && buffer.info[i].indic_position() == ot_position_t::POS_BELOW_C
1435                    {
1436                        base = i;
1437                        buffer.info[base].set_indic_position(ot_position_t::POS_BASE_C);
1438                    }
1439
1440                    i += 1;
1441                }
1442            }
1443
1444            if start < base
1445                && buffer.info[base].indic_position() as u32 > ot_position_t::POS_BASE_C as u32
1446            {
1447                base -= 1;
1448            }
1449
1450            break;
1451        }
1452
1453        base += 1;
1454    }
1455
1456    if base == end
1457        && start < base
1458        && buffer.info[base - 1].is_one_of(rb_flag(ot_category_t::OT_ZWJ as u32))
1459    {
1460        base -= 1;
1461    }
1462
1463    if base < end {
1464        while start < base
1465            && buffer.info[base].is_one_of(
1466                rb_flag(ot_category_t::OT_N as u32) | rb_flag(ot_category_t::OT_H as u32),
1467            )
1468        {
1469            base -= 1;
1470        }
1471    }
1472
1473    // - Reorder matras:
1474    //
1475    //   If a pre-base matra character had been reordered before applying basic
1476    //   features, the glyph can be moved closer to the main consonant based on
1477    //   whether half-forms had been formed. Actual position for the matra is
1478    //   defined as “after last standalone halant glyph, after initial matra
1479    //   position and before the main consonant”. If ZWJ or ZWNJ follow this
1480    //   halant, position is moved after it.
1481    //
1482    // IMPLEMENTATION NOTES:
1483    //
1484    // It looks like the last sentence is wrong.  Testing, with Windows 7 Uniscribe
1485    // and Devanagari shows that the behavior is best described as:
1486    //
1487    // "If ZWJ follows this halant, matra is NOT repositioned after this halant.
1488    //  If ZWNJ follows this halant, position is moved after it."
1489    //
1490    // Test case, with Adobe Devanagari or Nirmala UI:
1491    //
1492    //   U+091F,U+094D,U+200C,U+092F,U+093F
1493    //   (Matra moves to the middle, after ZWNJ.)
1494    //
1495    //   U+091F,U+094D,U+200D,U+092F,U+093F
1496    //   (Matra does NOT move, stays to the left.)
1497    //
1498    // https://github.com/harfbuzz/harfbuzz/issues/1070
1499
1500    // Otherwise there can't be any pre-base matra characters.
1501    if start + 1 < end && start < base {
1502        // If we lost track of base, alas, position before last thingy.
1503        let mut new_pos = if base == end { base - 2 } else { base - 1 };
1504
1505        // Malayalam / Tamil do not have "half" forms or explicit virama forms.
1506        // The glyphs formed by 'half' are Chillus or ligated explicit viramas.
1507        // We want to position matra after them.
1508        if buffer.script != Some(script::MALAYALAM) && buffer.script != Some(script::TAMIL) {
1509            loop {
1510                while new_pos > start
1511                    && !buffer.info[new_pos].is_one_of(
1512                        rb_flag(ot_category_t::OT_M as u32)
1513                            | rb_flag(ot_category_t::OT_MPst as u32)
1514                            | rb_flag(ot_category_t::OT_H as u32),
1515                    )
1516                {
1517                    new_pos -= 1;
1518                }
1519
1520                // If we found no Halant we are done.
1521                // Otherwise only proceed if the Halant does
1522                // not belong to the Matra itself!
1523                if buffer.info[new_pos].is_halant()
1524                    && buffer.info[new_pos].indic_position() != ot_position_t::POS_PRE_M
1525                {
1526                    if new_pos + 1 < end {
1527                        // -> If ZWJ follows this halant, matra is NOT repositioned after this halant.
1528                        if buffer.info[new_pos + 1].indic_category() == ot_category_t::OT_ZWJ {
1529                            // Keep searching.
1530                            if new_pos > start {
1531                                new_pos -= 1;
1532                                continue;
1533                            }
1534                        }
1535
1536                        // -> If ZWNJ follows this halant, position is moved after it.
1537                        //
1538                        // IMPLEMENTATION NOTES:
1539                        //
1540                        // This is taken care of by the state-machine. A Halant,ZWNJ is a terminating
1541                        // sequence for a consonant syllable; any pre-base matras occurring after it
1542                        // will belong to the subsequent syllable.
1543                    }
1544                } else {
1545                    new_pos = start; // No move.
1546                }
1547
1548                break;
1549            }
1550        }
1551
1552        if start < new_pos && buffer.info[new_pos].indic_position() != ot_position_t::POS_PRE_M {
1553            // Now go see if there's actually any matras...
1554            for i in (start + 1..=new_pos).rev() {
1555                if buffer.info[i - 1].indic_position() == ot_position_t::POS_PRE_M {
1556                    let old_pos = i - 1;
1557                    // Shouldn't actually happen.
1558                    if old_pos < base && base <= new_pos {
1559                        base -= 1;
1560                    }
1561
1562                    let tmp = buffer.info[old_pos];
1563                    for i in 0..new_pos - old_pos {
1564                        buffer.info[i + old_pos] = buffer.info[i + old_pos + 1];
1565                    }
1566                    buffer.info[new_pos] = tmp;
1567
1568                    // Note: this merge_clusters() is intentionally *after* the reordering.
1569                    // Indic matra reordering is special and tricky...
1570                    buffer.merge_clusters(new_pos, cmp::min(end, base + 1));
1571
1572                    new_pos -= 1;
1573                }
1574            }
1575        } else {
1576            for i in start..base {
1577                if buffer.info[i].indic_position() == ot_position_t::POS_PRE_M {
1578                    buffer.merge_clusters(i, cmp::min(end, base + 1));
1579                    break;
1580                }
1581            }
1582        }
1583    }
1584
1585    // - Reorder reph:
1586    //
1587    //   Reph’s original position is always at the beginning of the syllable,
1588    //   (i.e. it is not reordered at the character reordering stage). However,
1589    //   it will be reordered according to the basic-forms shaping results.
1590    //   Possible positions for reph, depending on the script, are; after main,
1591    //   before post-base consonant forms, and after post-base consonant forms.
1592
1593    // Two cases:
1594    //
1595    // - If repha is encoded as a sequence of characters (Ra,H or Ra,H,ZWJ), then
1596    //   we should only move it if the sequence ligated to the repha form.
1597    //
1598    // - If repha is encoded separately and in the logical position, we should only
1599    //   move it if it did NOT ligate.  If it ligated, it's probably the font trying
1600    //   to make it work without the reordering.
1601
1602    if start + 1 < end
1603        && buffer.info[start].indic_position() == ot_position_t::POS_RA_TO_BECOME_REPH
1604        && (buffer.info[start].indic_category() == ot_category_t::OT_Repha)
1605            ^ buffer.info[start].ligated_and_didnt_multiply()
1606    {
1607        let mut new_reph_pos;
1608        'reph: {
1609            let reph_pos = indic_plan.config.reph_pos;
1610
1611            // 1. If reph should be positioned after post-base consonant forms,
1612            //    proceed to step 5.
1613            if reph_pos != RephPosition::AfterPost {
1614                // 2. If the reph repositioning class is not after post-base: target
1615                //    position is after the first explicit halant glyph between the
1616                //    first post-reph consonant and last main consonant. If ZWJ or ZWNJ
1617                //    are following this halant, position is moved after it. If such
1618                //    position is found, this is the target position. Otherwise,
1619                //    proceed to the next step.
1620                //
1621                //    Note: in old-implementation fonts, where classifications were
1622                //    fixed in shaping engine, there was no case where reph position
1623                //    will be found on this step.
1624                {
1625                    new_reph_pos = start + 1;
1626                    while new_reph_pos < base && !buffer.info[new_reph_pos].is_halant() {
1627                        new_reph_pos += 1;
1628                    }
1629
1630                    if new_reph_pos < base && buffer.info[new_reph_pos].is_halant() {
1631                        // ->If ZWJ or ZWNJ are following this halant, position is moved after it.
1632                        if new_reph_pos + 1 < base && buffer.info[new_reph_pos + 1].is_joiner() {
1633                            new_reph_pos += 1;
1634                        }
1635
1636                        break 'reph;
1637                    }
1638                }
1639
1640                // 3. If reph should be repositioned after the main consonant: find the
1641                //    first consonant not ligated with main, or find the first
1642                //    consonant that is not a potential pre-base-reordering Ra.
1643                if reph_pos == RephPosition::AfterMain {
1644                    new_reph_pos = base;
1645                    while new_reph_pos + 1 < end
1646                        && buffer.info[new_reph_pos + 1].indic_position()
1647                            <= ot_position_t::POS_AFTER_MAIN
1648                    {
1649                        new_reph_pos += 1;
1650                    }
1651
1652                    if new_reph_pos < end {
1653                        break 'reph;
1654                    }
1655                }
1656
1657                // 4. If reph should be positioned before post-base consonant, find
1658                //    first post-base classified consonant not ligated with main. If no
1659                //    consonant is found, the target position should be before the
1660                //    first matra, syllable modifier sign or vedic sign.
1661                //
1662                // This is our take on what step 4 is trying to say (and failing, BADLY).
1663                if reph_pos == RephPosition::AfterSub {
1664                    new_reph_pos = base;
1665                    while new_reph_pos + 1 < end
1666                        && (rb_flag_unsafe(buffer.info[new_reph_pos + 1].indic_position() as u32)
1667                            & (rb_flag(ot_position_t::POS_POST_C as u32)
1668                                | rb_flag(ot_position_t::POS_AFTER_POST as u32)
1669                                | rb_flag(ot_position_t::POS_SMVD as u32)))
1670                            == 0
1671                    {
1672                        new_reph_pos += 1;
1673                    }
1674
1675                    if new_reph_pos < end {
1676                        break 'reph;
1677                    }
1678                }
1679            }
1680
1681            // 5. If no consonant is found in steps 3 or 4, move reph to a position
1682            //    immediately before the first post-base matra, syllable modifier
1683            //    sign or vedic sign that has a reordering class after the intended
1684            //    reph position. For example, if the reordering position for reph
1685            //    is post-main, it will skip above-base matras that also have a
1686            //    post-main position.
1687            //
1688            // Copied from step 2.
1689            new_reph_pos = start + 1;
1690            while new_reph_pos < base && !buffer.info[new_reph_pos].is_halant() {
1691                new_reph_pos += 1;
1692            }
1693
1694            if new_reph_pos < base && buffer.info[new_reph_pos].is_halant() {
1695                /* ->If ZWJ or ZWNJ are following this halant, position is moved after it. */
1696                if new_reph_pos + 1 < base && buffer.info[new_reph_pos + 1].is_joiner() {
1697                    new_reph_pos += 1;
1698                }
1699
1700                break 'reph;
1701            }
1702            // See https://github.com/harfbuzz/harfbuzz/issues/2298#issuecomment-615318654
1703
1704            // 6. Otherwise, reorder reph to the end of the syllable.
1705            {
1706                new_reph_pos = end - 1;
1707                while new_reph_pos > start
1708                    && buffer.info[new_reph_pos].indic_position() == ot_position_t::POS_SMVD
1709                {
1710                    new_reph_pos -= 1;
1711                }
1712
1713                // If the Reph is to be ending up after a Matra,Halant sequence,
1714                // position it before that Halant so it can interact with the Matra.
1715                // However, if it's a plain Consonant,Halant we shouldn't do that.
1716                // Uniscribe doesn't do this.
1717                // TEST: U+0930,U+094D,U+0915,U+094B,U+094D
1718                if buffer.info[new_reph_pos].is_halant() {
1719                    for info in &buffer.info[base + 1..new_reph_pos] {
1720                        if (rb_flag_unsafe(info.indic_category() as u32)
1721                            & (rb_flag(ot_category_t::OT_M as u32)
1722                                | rb_flag(ot_category_t::OT_MPst as u32)))
1723                            != 0
1724                        {
1725                            // Ok, got it.
1726                            new_reph_pos -= 1;
1727                        }
1728                    }
1729                }
1730            }
1731
1732            break 'reph;
1733        }
1734
1735        // Move
1736        buffer.merge_clusters(start, new_reph_pos + 1);
1737
1738        let reph = buffer.info[start];
1739        for i in 0..new_reph_pos - start {
1740            buffer.info[i + start] = buffer.info[i + start + 1];
1741        }
1742        buffer.info[new_reph_pos] = reph;
1743
1744        if start < base && base <= new_reph_pos {
1745            base -= 1;
1746        }
1747    }
1748
1749    // - Reorder pre-base-reordering consonants:
1750    //
1751    //   If a pre-base-reordering consonant is found, reorder it according to
1752    //   the following rules:
1753
1754    // Otherwise there can't be any pre-base-reordering Ra.
1755    if try_pref && base + 1 < end {
1756        for i in base + 1..end {
1757            if (buffer.info[i].mask & indic_plan.mask_array[indic_feature::PREF]) != 0 {
1758                // 1. Only reorder a glyph produced by substitution during application
1759                //    of the <pref> feature. (Note that a font may shape a Ra consonant with
1760                //    the feature generally but block it in certain contexts.)
1761                //
1762                // Note: We just check that something got substituted.  We don't check that
1763                // the <pref> feature actually did it...
1764                //
1765                // Reorder pref only if it ligated.
1766                if buffer.info[i].ligated_and_didnt_multiply() {
1767                    // 2. Try to find a target position the same way as for pre-base matra.
1768                    //    If it is found, reorder pre-base consonant glyph.
1769                    //
1770                    // 3. If position is not found, reorder immediately before main consonant.
1771
1772                    let mut new_pos = base;
1773                    // Malayalam / Tamil do not have "half" forms or explicit virama forms.
1774                    // The glyphs formed by 'half' are Chillus or ligated explicit viramas.
1775                    // We want to position matra after them.
1776                    if buffer.script != Some(script::MALAYALAM)
1777                        && buffer.script != Some(script::TAMIL)
1778                    {
1779                        while new_pos > start
1780                            && !buffer.info[new_pos - 1].is_one_of(
1781                                rb_flag(ot_category_t::OT_M as u32)
1782                                    | rb_flag(ot_category_t::OT_MPst as u32)
1783                                    | rb_flag(ot_category_t::OT_H as u32),
1784                            )
1785                        {
1786                            new_pos -= 1;
1787                        }
1788                    }
1789
1790                    if new_pos > start && buffer.info[new_pos - 1].is_halant() {
1791                        // -> If ZWJ or ZWNJ follow this halant, position is moved after it.
1792                        if new_pos < end && buffer.info[new_pos].is_joiner() {
1793                            new_pos += 1;
1794                        }
1795                    }
1796
1797                    {
