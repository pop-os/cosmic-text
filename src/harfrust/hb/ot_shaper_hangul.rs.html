27fn collect_features_hangul(planner: &mut hb_ot_shape_planner_t) {
28    planner
29        .ot_map
30        .add_feature(hb_tag_t::new(b"ljmo"), F_NONE, 1);
31    planner
32        .ot_map
33        .add_feature(hb_tag_t::new(b"vjmo"), F_NONE, 1);
34    planner
35        .ot_map
36        .add_feature(hb_tag_t::new(b"tjmo"), F_NONE, 1);
37}

39fn override_features_hangul(planner: &mut hb_ot_shape_planner_t) {
40    // Uniscribe does not apply 'calt' for Hangul, and certain fonts
41    // (Noto Sans CJK, Source Sans Han, etc) apply all of jamo lookups
42    // in calt, which is not desirable.
43    planner.ot_map.disable_feature(hb_tag_t::new(b"calt"));
44}

61const L_BASE: u32 = 0x1100;
62const V_BASE: u32 = 0x1161;
63const T_BASE: u32 = 0x11A7;
64const L_COUNT: u32 = 19;
65const V_COUNT: u32 = 21;
66const T_COUNT: u32 = 28;
67const N_COUNT: u32 = V_COUNT * T_COUNT;
68const S_COUNT: u32 = L_COUNT * N_COUNT;
69const S_BASE: u32 = 0xAC00;
70
71fn is_combining_l(u: u32) -> bool {
72    (L_BASE..L_BASE + L_COUNT).contains(&u)
73}
74
75fn is_combining_v(u: u32) -> bool {
76    (V_BASE..V_BASE + V_COUNT).contains(&u)
77}
78
79fn is_combining_t(u: u32) -> bool {
80    (T_BASE + 1..T_BASE + T_COUNT).contains(&u)
81}
82
83fn is_combined_s(u: u32) -> bool {
84    (S_BASE..S_BASE + S_COUNT).contains(&u)
85}
86
87fn is_l(u: u32) -> bool {
88    (0x1100..=0x115F).contains(&u) || (0xA960..=0xA97C).contains(&u)
89}
90
91fn is_v(u: u32) -> bool {
92    (0x1160..=0x11A7).contains(&u) || (0xD7B0..=0xD7C6).contains(&u)
93}
94
95fn is_t(u: u32) -> bool {
96    (0x11A8..=0x11FF).contains(&u) || (0xD7CB..=0xD7FB).contains(&u)
97}
98
99fn is_hangul_tone(u: u32) -> bool {
100    (0x302E..=0x302F).contains(&u)
101}

103fn is_zero_width_char(face: &hb_font_t, c: Codepoint) -> bool {
104    if let Some(glyph) = face.get_nominal_glyph(c) {
105        face.glyph_h_advance(glyph) == 0
106    } else {
107        false
108    }
109}

111fn preprocess_text_hangul(_: &hb_ot_shape_plan_t, face: &hb_font_t, buffer: &mut hb_buffer_t) {
112    buffer.allocate_var(GlyphInfo::HANGUL_SHAPING_FEATURE_VAR);
113
114    // Hangul syllables come in two shapes: LV, and LVT.  Of those:
115    //
116    //   - LV can be precomposed, or decomposed.  Lets call those
117    //     <LV> and <L,V>,
118    //   - LVT can be fully precomposed, partially precomposed, or
119    //     fully decomposed.  Ie. <LVT>, <LV,T>, or <L,V,T>.
120    //
121    // The composition / decomposition is mechanical.  However, not
122    // all <L,V> sequences compose, and not all <LV,T> sequences
123    // compose.
124    //
125    // Here are the specifics:
126    //
127    //   - <L>: U+1100..115F, U+A960..A97F
128    //   - <V>: U+1160..11A7, U+D7B0..D7C7
129    //   - <T>: U+11A8..11FF, U+D7CB..D7FB
130    //
131    //   - Only the <L,V> sequences for some of the U+11xx ranges combine.
132    //   - Only <LV,T> sequences for some of the Ts in U+11xx range combine.
133    //
134    // Here is what we want to accomplish in this shaper:
135    //
136    //   - If the whole syllable can be precomposed, do that,
137    //   - Otherwise, fully decompose and apply ljmo/vjmo/tjmo features.
138    //   - If a valid syllable is followed by a Hangul tone mark, reorder the tone
139    //     mark to precede the whole syllable - unless it is a zero-width glyph, in
140    //     which case we leave it untouched, assuming it's designed to overstrike.
141    //
142    // That is, of the different possible syllables:
143    //
144    //   <L>
145    //   <L,V>
146    //   <L,V,T>
147    //   <LV>
148    //   <LVT>
149    //   <LV, T>
150    //
151    // - <L> needs no work.
152    //
153    // - <LV> and <LVT> can stay the way they are if the font supports them, otherwise we
154    //   should fully decompose them if font supports.
155    //
156    // - <L,V> and <L,V,T> we should compose if the whole thing can be composed.
157    //
158    // - <LV,T> we should compose if the whole thing can be composed, otherwise we should
159    //   decompose.
160
161    buffer.clear_output();
162    // Extent of most recently seen syllable; valid only if start < end
163    let mut start = 0;
164    let mut end = 0;
165    buffer.idx = 0;
166    while buffer.idx < buffer.len {
167        let u = buffer.cur(0).glyph_id;
168        let c = buffer.cur(0).as_codepoint();
169
170        if is_hangul_tone(u) {
171            // We could cache the width of the tone marks and the existence of dotted-circle,
172            // but the use of the Hangul tone mark characters seems to be rare enough that
173            // I didn't bother for now.
174            if start < end && end == buffer.out_len {
175                // Tone mark follows a valid syllable; move it in front, unless it's zero width.
176                buffer.unsafe_to_break_from_outbuffer(Some(start), Some(buffer.idx));
177                buffer.next_glyph();
178                if !is_zero_width_char(face, c) {
179                    buffer.merge_out_clusters(start, end + 1);
180                    let out_info = buffer.out_info_mut();
181                    let tone = out_info[end];
182                    for i in (0..end - start).rev() {
183                        out_info[i + start + 1] = out_info[i + start];
184                    }
185                    out_info[start] = tone;
186                }
187            } else {
188                // No valid syllable as base for tone mark; try to insert dotted circle.
189                if !buffer
190                    .flags
191                    .contains(BufferFlags::DO_NOT_INSERT_DOTTED_CIRCLE)
192                    && face.has_glyph(0x25CC)
193                {
194                    let mut chars = [0; 2];
195                    if !is_zero_width_char(face, c) {
196                        chars[0] = u;
197                        chars[1] = 0x25CC;
198                    } else {
199                        chars[0] = 0x25CC;
200                        chars[1] = u;
201                    }
202
203                    buffer.replace_glyphs(1, 2, &chars);
204                } else {
205                    // No dotted circle available in the font; just leave tone mark untouched.
206                    buffer.next_glyph();
207                }
208            }
209
210            start = buffer.out_len;
211            end = buffer.out_len;
212            continue;
213        }
214
215        // Remember current position as a potential syllable start;
216        // will only be used if we set end to a later position.
217        start = buffer.out_len;
218
219        if is_l(u) && buffer.idx + 1 < buffer.len {
220            let l = u;
221            let v = buffer.cur(1).glyph_id;
222            if is_v(v) {
223                // Have <L,V> or <L,V,T>.
224                let mut t = 0;
225                let mut tindex = 0;
226                if buffer.idx + 2 < buffer.len {
227                    t = buffer.cur(2).glyph_id;
228                    if is_t(t) {
229                        // Only used if isCombiningT (t); otherwise invalid.
230                        tindex = t - T_BASE;
231                    } else {
232                        // The next character was not a trailing jamo.
233                        t = 0;
234                    }
235                }
236
237                let offset = if t != 0 { 3 } else { 2 };
238                buffer.unsafe_to_break(Some(buffer.idx), Some(buffer.idx + offset));
239
240                // We've got a syllable <L,V,T?>; see if it can potentially be composed.
241                if is_combining_l(l) && is_combining_v(v) && (t == 0 || is_combining_t(t)) {
242                    // Try to compose; if this succeeds, end is set to start+1.
243                    let s = S_BASE + (l - L_BASE) * N_COUNT + (v - V_BASE) * T_COUNT + tindex;
244                    if face.has_glyph(s) {
245                        let n = if t != 0 { 3 } else { 2 };
246                        buffer.replace_glyphs(n, 1, &[s]);
247                        end = start + 1;
248                        continue;
249                    }
250                }
251
252                // We didn't compose, either because it's an Old Hangul syllable without a
253                // precomposed character in Unicode, or because the font didn't support the
254                // necessary precomposed glyph.
255                // Set jamo features on the individual glyphs, and advance past them.
256                buffer.cur_mut(0).set_hangul_shaping_feature(LJMO);
257                buffer.next_glyph();
258                buffer.cur_mut(0).set_hangul_shaping_feature(VJMO);
259                buffer.next_glyph();
260                if t != 0 {
261                    buffer.cur_mut(0).set_hangul_shaping_feature(TJMO);
262                    buffer.next_glyph();
263                    end = start + 3;
264                } else {
265                    end = start + 2;
266                }
267
268                buffer.merge_out_clusters(start, end);
269
270                continue;
271            }
272        } else if is_combined_s(u) {
273            // Have <LV>, <LVT>, or <LV,T>
274            let s = u;
275            let has_glyph = face.has_glyph(s);
276
277            let lindex = (s - S_BASE) / N_COUNT;
278            let nindex = (s - S_BASE) % N_COUNT;
279            let vindex = nindex / T_COUNT;
280            let tindex = nindex % T_COUNT;
281
282            if tindex == 0 && buffer.idx + 1 < buffer.len && is_combining_t(buffer.cur(1).glyph_id)
283            {
284                // <LV,T>, try to combine.
285                let new_tindex = buffer.cur(1).glyph_id - T_BASE;
286                let new_s = s + new_tindex;
287
288                if face.has_glyph(new_s) {
289                    buffer.replace_glyphs(2, 1, &[new_s]);
290                    end = start + 1;
291                    continue;
292                } else {
293                    // Mark unsafe between LV and T.
294                    buffer.unsafe_to_break(Some(buffer.idx), Some(buffer.idx + 2));
295                }
296            }
297
298            // Otherwise, decompose if font doesn't support <LV> or <LVT>,
299            // or if having non-combining <LV,T>.  Note that we already handled
300            // combining <LV,T> above.
301            if !has_glyph
302                || (tindex == 0 && buffer.idx + 1 < buffer.len && is_t(buffer.cur(1).glyph_id))
303            {
304                let decomposed = [L_BASE + lindex, V_BASE + vindex, T_BASE + tindex];
305                if face.has_glyph(decomposed[0])
306                    && face.has_glyph(decomposed[1])
307                    && (tindex == 0 || face.has_glyph(decomposed[2]))
308                {
309                    let mut s_len = if tindex != 0 { 3 } else { 2 };
310                    buffer.replace_glyphs(1, s_len, &decomposed);
311
312                    // If we decomposed an LV because of a non-combining T following,
313                    // we want to include this T in the syllable.
314                    if has_glyph && tindex == 0 {
315                        buffer.next_glyph();
316                        s_len += 1;
317                    }
318
319                    // We decomposed S: apply jamo features to the individual glyphs
320                    // that are now in `buffer.out_info`.
321                    end = start + s_len;
322
323                    buffer.out_info_mut()[start + 0].set_hangul_shaping_feature(LJMO);
324                    buffer.out_info_mut()[start + 1].set_hangul_shaping_feature(VJMO);
325                    if start + 2 < end {
326                        buffer.out_info_mut()[start + 2].set_hangul_shaping_feature(TJMO);
327                    }
328
329                    buffer.merge_out_clusters(start, end);
330
331                    continue;
332                } else if tindex == 0 && buffer.idx + 1 > buffer.len && is_t(buffer.cur(1).glyph_id)
333                {
334                    // Mark unsafe between LV and T.
335                    buffer.unsafe_to_break(Some(buffer.idx), Some(buffer.idx + 2));
336                }
337            }
338
339            if has_glyph {
340                // We didn't decompose the S, so just advance past it.
341                end = start + 1;
342                buffer.next_glyph();
343                continue;
344            }
345        }
346
347        // Didn't find a recognizable syllable, so we leave end <= start;
348        // this will prevent tone-mark reordering happening.
349        buffer.next_glyph();
350    }
351
352    buffer.sync();
353}

355fn setup_masks_hangul(plan: &hb_ot_shape_plan_t, _: &hb_font_t, buffer: &mut hb_buffer_t) {
356    let hangul_plan = plan.data::<hangul_shape_plan_t>();
357    for info in buffer.info_slice_mut() {
358        info.mask |= hangul_plan.mask_array[info.hangul_shaping_feature() as usize];
359    }
360
361    buffer.deallocate_var(GlyphInfo::HANGUL_SHAPING_FEATURE_VAR);
362}
