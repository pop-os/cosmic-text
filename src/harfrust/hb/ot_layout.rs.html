147/// Applies the lookups in the given GSUB or GPOS table.
148pub fn apply_layout_table<T: LayoutTable>(
149    plan: &hb_ot_shape_plan_t,
150    face: &hb_font_t,
151    buffer: &mut hb_buffer_t,
152    table: Option<&T>,
153) {
154    let mut ctx = OT::hb_ot_apply_context_t::new(T::INDEX, face, buffer);
155
156    for (stage_index, stage) in plan.ot_map.stages(T::INDEX).iter().enumerate() {
157        if let Some(table) = table {
158            for lookup_map in plan.ot_map.stage_lookups(T::INDEX, stage_index) {
159                let Some(lookup) = table.get_lookup(lookup_map.index) else {
160                    continue;
161                };
162
163                if lookup.digest().may_intersect(&ctx.buffer.digest) {
164                    ctx.lookup_index = lookup_map.index;
165                    ctx.set_lookup_mask(lookup_map.mask);
166                    ctx.auto_zwj = lookup_map.auto_zwj;
167                    ctx.auto_zwnj = lookup_map.auto_zwnj;
168
169                    ctx.random = lookup_map.random;
170                    ctx.per_syllable = lookup_map.per_syllable;
171
172                    apply_string::<T>(&mut ctx, lookup);
173                }
174            }
175        }
176
177        if let Some(func) = stage.pause_func {
178            if func(plan, face, ctx.buffer) {
179                ctx.buffer.update_digest();
180            }
181        }
182    }
183}

185fn apply_string<T: LayoutTable>(ctx: &mut OT::hb_ot_apply_context_t, lookup: &LookupInfo) {
186    if ctx.buffer.is_empty() || ctx.lookup_mask() == 0 {
187        return;
188    }
189
190    let subtable_count = lookup.subtables.len();
191
192    let lookup_props = lookup.props();
193    if lookup_props != ctx.cached_props {
194        let cache_it =
195            subtable_count > 1 && (lookup_props as u16 & lookup_flags::USE_MARK_FILTERING_SET != 0);
196        if cache_it {
197            ctx.buffer.info.iter_mut().for_each(|info: &mut GlyphInfo| {
198                let matches = check_glyph_property(ctx.face, info, lookup_props, u32::MAX);
199                info.set_matches(matches);
200            });
201            ctx.cached_props = lookup_props;
202        } else {
203            ctx.cached_props = u32::MAX;
204        }
205    }
206    ctx.lookup_props = lookup_props;
207
208    ctx.update_matchers();
209
210    if !lookup.is_reverse() {
211        // in/out forward substitution/positioning
212        if !T::IN_PLACE {
213            ctx.buffer.clear_output();
214        }
215        ctx.buffer.idx = 0;
216        apply_forward(ctx, lookup);
217
218        if !T::IN_PLACE {
219            ctx.buffer.sync();
220        }
221    } else {
222        // in-place backward substitution/positioning
223        debug_assert!(!ctx.buffer.have_output);
224
225        ctx.buffer.idx = ctx.buffer.len - 1;
226        apply_backward(ctx, lookup);
227    }
228}
229
230fn apply_forward(ctx: &mut OT::hb_ot_apply_context_t, lookup: &LookupInfo) -> bool {
231    let mut ret = false;
232    let Some(table_data) = ctx.face.ot_tables.table_data(ctx.table_index) else {
233        return false;
234    };
235
236    let use_hot_subtable_cache = lookup.cache_enter(ctx);
237
238    while ctx.buffer.successful {
239        let mut j = ctx.buffer.idx;
240        while j < ctx.buffer.len && {
241            let info = &ctx.buffer.info[j];
242            !(lookup.digest.may_have(info.glyph_id)
243                && (info.mask & ctx.lookup_mask()) != 0
244                && check_glyph_property(ctx.face, info, ctx.lookup_props, ctx.cached_props))
245        } {
246            j += 1;
247        }
248        if j > ctx.buffer.idx {
249            ctx.buffer.next_glyphs(j - ctx.buffer.idx);
250        }
251        if ctx.buffer.idx >= ctx.buffer.len {
252            break;
253        }
254
255        if lookup
256            .apply(ctx, table_data, use_hot_subtable_cache)
257            .is_some()
258        {
259            ret = true;
260        } else {
261            ctx.buffer.next_glyph();
262        }
263    }
264
265    if use_hot_subtable_cache {
266        lookup.cache_leave(ctx);
267    }
268
269    ret
270}
271
272fn apply_backward(ctx: &mut OT::hb_ot_apply_context_t, lookup: &LookupInfo) -> bool {
273    let mut ret = false;
274    let Some(table_data) = ctx.face.ot_tables.table_data(ctx.table_index) else {
275        return false;
276    };
277    loop {
278        let cur = ctx.buffer.cur(0);
279        ret |= lookup.digest.may_have(cur.glyph_id)
280            && (cur.mask & ctx.lookup_mask()) != 0
281            && check_glyph_property(ctx.face, cur, ctx.lookup_props, ctx.cached_props)
282            && lookup.apply(ctx, table_data, false).is_some();
283
284        if ctx.buffer.idx == 0 {
285            break;
286        }
287
288        ctx.buffer.idx -= 1;
289    }
290    ret
291}

295/* Design:
296 * unicode_props() is a two-byte number.  The low byte includes:
297 * - Modified General_Category: 5 bits.
298 * - A bit each for:
299 *   * Is it Default_Ignorable(); we have a modified Default_Ignorable().
300 *   * Whether it's one of the four Mongolian Free Variation Selectors,
301 *     CGJ, or other characters that are hidden but should not be ignored
302 *     like most other Default_Ignorable()s do during GSUB matching.
303 *   * Whether it's a grapheme continuation.
304 *
305 * The high-byte has different meanings, switched by the Gen-Cat:
306 * - For Mn,Mc,Me: the modified Combining_Class.
307 * - For Cf: whether it's ZWJ, ZWNJ, or something else.
308 * - For Ws: index of which space character this is, if space fallback
309 *   is needed, ie. we don't set this by default, only if asked to.
310 *
311 * Above I said "modified" General_Category. This is because we need to
312 * remember Variation Selectors, and we don't have bits left. So we
313 * change their Gen_Cat from Mn to Cf, and use a bit of the high byte to
314 * remember them.
315 */
316
317//  enum hb_unicode_props_flags_t {
318//     UPROPS_MASK_GEN_CAT	= 0x001Fu,
319//     UPROPS_MASK_IGNORABLE	= 0x0020u,
320//     UPROPS_MASK_HIDDEN	= 0x0040u, /* MONGOLIAN FREE VARIATION SELECTOR 1..4, or TAG characters */
321//     UPROPS_MASK_CONTINUATION=0x0080u,
322
323//     /* If GEN_CAT=FORMAT, top byte masks: */
324//     UPROPS_MASK_Cf_ZWJ	= 0x0100u,
325//     UPROPS_MASK_Cf_ZWNJ	= 0x0200u
326//   };
327//   HB_MARK_AS_FLAG_T (hb_unicode_props_flags_t);

331//   {

333//     unsigned int u = info->codepoint;
334//     unsigned int gen_cat = (unsigned int) unicode->general_category (u);
335//     unsigned int props = gen_cat;

337//     if (u >= 0x80u)
338//     {
339//       if (unlikely (unicode->is_default_ignorable (u)))
340//       {

342//         props |=  UPROPS_MASK_IGNORABLE;
343//         if (u == 0x200Cu) props |= UPROPS_MASK_Cf_ZWNJ;
344//         else if (u == 0x200Du) props |= UPROPS_MASK_Cf_ZWJ;
345//         /* Mongolian Free Variation Selectors need to be remembered
346//          * because although we need to hide them like default-ignorables,
347//          * they need to non-ignorable during shaping.  This is similar to
348//          * what we do for joiners in Indic-like shapers, but since the
349//          * FVSes are GC=Mn, we have use a separate bit to remember them.
350//          * Fixes:
351//          * https://github.com/harfbuzz/harfbuzz/issues/234 */
352//         else if (unlikely (hb_in_ranges<hb_codepoint_t> (u, 0x180Bu, 0x180Du, 0x180Fu, 0x180Fu))) props |= UPROPS_MASK_HIDDEN;
353//         /* TAG characters need similar treatment. Fixes:
354//          * https://github.com/harfbuzz/harfbuzz/issues/463 */
355//         else if (unlikely (hb_in_range<hb_codepoint_t> (u, 0xE0020u, 0xE007Fu))) props |= UPROPS_MASK_HIDDEN;
356//         /* COMBINING GRAPHEME JOINER should not be skipped; at least some times.
357//          * https://github.com/harfbuzz/harfbuzz/issues/554 */
358//         else if (unlikely (u == 0x034Fu))
359//         {
360//       buffer->scratch_flags |= HB_BUFFER_SCRATCH_FLAG_HAS_CGJ;
361//       props |= UPROPS_MASK_HIDDEN;
362//         }
363//       }
364
365//       if (unlikely (HB_UNICODE_GENERAL_CATEGORY_IS_MARK (gen_cat)))
366//       {
367//         props |= UPROPS_MASK_CONTINUATION;
368//         props |= unicode->modified_combining_class (u)<<8;
369//       }

644    //   /* lig_props: aka lig_id / lig_comp
645    //    *
646    //    * When a ligature is formed:
647    //    *
648    //    *   - The ligature glyph and any marks in between all the same newly allocated
649    //    *     lig_id,
650    //    *   - The ligature glyph will get lig_num_comps set to the number of components
651    //    *   - The marks get lig_comp > 0, reflecting which component of the ligature
652    //    *     they were applied to.
653    //    *   - This is used in GPOS to attach marks to the right component of a ligature
654    //    *     in MarkLigPos,
655    //    *   - Note that when marks are ligated together, much of the above is skipped
656    //    *     and the current lig_id reused.
657    //    *
658    //    * When a multiple-substitution is done:
659    //    *
660    //    *   - All resulting glyphs will have lig_id = 0,
661    //    *   - The resulting glyphs will have lig_comp = 0, 1, 2, ... respectively.
662    //    *   - This is used in GPOS to attach marks to the first component of a
663    //    *     multiple substitution in MarkBasePos.
664    //    *
665    //    * The numbers are also used in GPOS to do mark-to-mark positioning only
666    //    * to marks that belong to the same component of the same ligature.
667    //    */
